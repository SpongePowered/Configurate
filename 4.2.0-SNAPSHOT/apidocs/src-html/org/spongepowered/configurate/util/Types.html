090    /**
091     * Get if the provided type is an array type.
092     *
093     * <p>Being an array type means that the provided
094     * type has a component type.</p>
095     *
096     * @param input input type
097     * @return whether the type is an array
098     * @since 4.0.0
099     */
100    public static boolean isArray(final Type input) {
101        if (input instanceof Class<?>) {
102            return ((Class<?>) input).isArray();
103        } else if (input instanceof ParameterizedType) {
104            return isArray(((ParameterizedType) input).getRawType());
105        } else {
106            return input instanceof GenericArrayType;
107        }
108    }

111     * Get whether or not the provided input type is a boxed primitive type.
112     *
113     * <p>This check will <em>not</em> match unboxed primitives.</p>
114     *
115     * @param input type to check
116     * @return if type is a boxed primitive

120        return BOXED_TO_PRIMITIVE.containsKey(input);

125     *

128     *          or the input type if it was already non-primitive
129     * @since 4.0.0

132        final Type ret = BOXED_TO_PRIMITIVE.get(input);
133        return ret == null ? input : ret;

136    /**
137     * Get the default value for a type.
138     *
139     * <p>For all reference types, this is {@code null}. For all primitive
140     * types, this is equivalent to their defined {@code default} value.</p>
141     *
142     * @param type the type to get a default value for
143     * @return the default value, or {@code null} for reference types
144     * @since 4.1.0
145     */
146    public static @Nullable Object defaultValue(final Class<?> type) {
147        requireNonNull(type, "type");
148
149        if (!type.isPrimitive() || type == void.class) {
150            return null;
151        } else if (type == boolean.class) {
152            return Boolean.FALSE;
153        } else if (type == char.class) {
154            return (char) 0;
155        } else if (type == byte.class) {
156            return (byte) 0;
157        } else if (type == short.class) {
158            return (short) 0;
159        } else if (type == int.class) {
160            return 0;
161        } else if (type == long.class) {
162            return 0L;
163        } else if (type == float.class) {
164            return 0F;
165        } else if (type == double.class) {
166            return 0D;
167        } else {
168            // TODO: Verify that this works with Valhalla primitive types
169            return Array.get(Array.newInstance(type, 1), 0);
170        }
171    }

350    private static Stream<Type> calculateSuperTypes(final Type type, final boolean includeInterfaces) {
351        requireNonNull(type, "type");
352        return StreamSupport.stream(Spliterators.spliterator(new SuperTypesIterator(type, includeInterfaces), Long.MAX_VALUE,
353                                                             Spliterator.NONNULL | Spliterator.IMMUTABLE), false);
354    }

356    /**
357     * Recursively iterate through supertypes.
358     */
359    static class SuperTypesIterator implements Iterator<Type> {
360        private final boolean includeInterfaces;
361        private final Deque<Type> types = new ArrayDeque<>();
362        private final Set<Type> seen = new HashSet<>();
363
364        SuperTypesIterator(final Type base, final boolean includeInterfaces) {
365            this.types.add(base);
366            this.includeInterfaces = includeInterfaces;
367        }
368
369        @Override
370        public boolean hasNext() {
371            return !this.types.isEmpty();
372        }
373
374        @Override
375        public Type next() {
376            // Get current type, throws the correct exception if empty
377            final Type head = this.types.removeLast();
378
379            // Calculate the next step depending on the type of Type seen
380            // Arrays, covariant based on component type
381            if ((head instanceof Class<?> && ((Class<?>) head).isArray()) || head instanceof GenericArrayType) {
382                // find a super component-type
383                final Type componentType;
384                if (head instanceof Class<?>) {
385                    componentType = ((Class<?>) head).getComponentType();
386                } else {
387                    componentType = ((GenericArrayType) head).getGenericComponentType();
388                }
389
390                addSuperClassAndInterface(componentType, erase(componentType), TypeFactory::arrayOf);
391            } else if (head instanceof Class<?> || head instanceof ParameterizedType) {
392                final Class<?> clazz;
393                if (head instanceof ParameterizedType) {
394                    final ParameterizedType parameterized = (ParameterizedType) head;
395                    clazz = (Class<?>) parameterized.getRawType();
396                } else {
397                    clazz = (Class<?>) head;
398                }
399                addSuperClassAndInterface(head, clazz, null);
400            } else if (head instanceof TypeVariable<?>) {
401                addAllIfUnseen(head, ((TypeVariable<?>) head).getBounds());
402            } else if (head instanceof WildcardType) {
403                final Type[] upperBounds = ((WildcardType) head).getUpperBounds();
404                if (upperBounds.length == 1) { // single type
405                    final Type upperBound = upperBounds[0];
406                    addSuperClassAndInterface(head, erase(upperBound), TypeFactory::wildcardExtends);
407                } else { // for each bound, add as a single supertype
408                    addAllIfUnseen(head, ((WildcardType) head).getUpperBounds());
409                }
410            }
411            return head;
412        }
413
414        private void addAllIfUnseen(final Type base, final Type... types) {
415            for (final Type type : types) {
416                addIfUnseen(resolveType(type, base));
417            }
418        }
419
420        private void addIfUnseen(final Type type) {
421            if (this.seen.add(type)) {
422                this.types.addLast(type);
423            }
424        }
425
426        private void addSuperClassAndInterface(final Type base, final Class<?> actualClass, final @Nullable UnaryOperator<Type> postProcess) {
427            if (this.includeInterfaces) {
428                for (Type itf : actualClass.getGenericInterfaces()) {
429                    if (postProcess != null) {
430                        addIfUnseen(postProcess.apply(resolveType(itf, base)));
431                    } else {
432                        addIfUnseen(resolveType(itf, base));
433                    }
434                }
435            }
436
437            if (actualClass.getSuperclass() != null) {
438                final Type resolved = resolveType(actualClass.getGenericSuperclass(), base);
439                addIfUnseen(postProcess == null ? resolved : postProcess.apply(resolved));
440            }
441        }
442    }

444    static class CombinedAnnotations implements AnnotatedElement {
445        private static final Annotation[] EMPTY_ANNOTATION_ARRAY = new Annotation[0];
446
447        private final AnnotatedElement[] elements;
448
449        CombinedAnnotations(final AnnotatedElement[] elements) {
450            this.elements = elements;
451        }
452
453        @Override
454        public boolean isAnnotationPresent(final Class<? extends Annotation> annotationClass) {
455            for (AnnotatedElement element : this.elements) {
456                if (element.isAnnotationPresent(annotationClass)) {
457                    return true;
458                }
459            }
460            return false;
461        }
462
463        @Override
464        public <T extends Annotation> @Nullable T getAnnotation(final Class<T> annotationClass) {
465            @Nullable T ret = null;
466            for (AnnotatedElement element : this.elements) {
467                ret = element.getAnnotation(annotationClass);
468                if (ret != null) {
469                    break;
470                }
471            }
472            return ret;
473        }
474
475        @Override
476        public Annotation[] getAnnotations() {
477            final List<Annotation> annotations = new ArrayList<>();
478            for (AnnotatedElement element : this.elements) {
479                final Annotation[] annotation = element.getAnnotations();
480                if (annotation.length > 0) {
481                    annotations.addAll(Arrays.asList(annotation));
482                }
483            }
484            return annotations.toArray(EMPTY_ANNOTATION_ARRAY);
485        }
486
487        @SuppressWarnings("unchecked")
488        @Override
489        public <T extends Annotation> T[] getAnnotationsByType(final Class<T> annotationClass) {
490            final List<T> annotations = new ArrayList<>();
491            for (AnnotatedElement element : this.elements) {
492                final T[] annotation = element.getAnnotationsByType(annotationClass);
493                if (annotation.length > 0) {
494                    annotations.addAll(Arrays.asList(annotation));
495                }
496            }
497            return annotations.toArray((T[]) EMPTY_ANNOTATION_ARRAY);
498        }
499
500        @Override
501        public <T extends Annotation> @Nullable T getDeclaredAnnotation(final Class<T> annotationClass) {
502            @Nullable T ret = null;
503            for (AnnotatedElement element : this.elements) {
504                ret = element.getDeclaredAnnotation(annotationClass);
505                if (ret != null) {
506                    break;
507                }
508            }
509            return ret;
510        }
511
512        @SuppressWarnings("unchecked")
513        @Override
514        public <T extends Annotation> T[] getDeclaredAnnotationsByType(final Class<T> annotationClass) {
515            final List<T> annotations = new ArrayList<>();
516            for (AnnotatedElement element : this.elements) {
517                final T[] annotation = element.getDeclaredAnnotationsByType(annotationClass);
518                if (annotation.length > 0) {
519                    annotations.addAll(Arrays.asList(annotation));
520                }
521            }
522            return annotations.toArray((T[]) EMPTY_ANNOTATION_ARRAY);
523        }
524
525        @Override
526        public Annotation[] getDeclaredAnnotations() {
527            final List<Annotation> annotations = new ArrayList<>();
528            for (AnnotatedElement element : this.elements) {
529                final Annotation[] annotation = element.getDeclaredAnnotations();
530                if (annotation.length > 0) {
531                    annotations.addAll(Arrays.asList(annotation));
532                }
533            }
534            return annotations.toArray(EMPTY_ANNOTATION_ARRAY);
535        }
536    }
