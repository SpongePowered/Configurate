335    private DocumentBuilder newDocumentBuilder() throws ConfigurateException {
336        final DocumentBuilderFactory builderFactory = DocumentBuilderFactory.newInstance();
337        if (this.schema != null) {
338            builderFactory.setSchema(this.schema);
339        }
340        if (!this.resolvesExternalContent) {
341            // Settings based on https://cheatsheetseries.owasp.org/cheatsheets/XML_External_Entity_Prevention_Cheat_Sheet.html
342            try {
343                builderFactory.setFeature(FEATURE_EXTERNAL_GENERAL_ENTITIES, false);
344                builderFactory.setFeature(FEATURE_EXTERNAL_PARAMETER_ENTITIES, false);
345                builderFactory.setFeature(FEATURE_LOAD_EXTERNAL_DTD, false);
346            } catch (final ParserConfigurationException e) {
347                throw new ConfigurateException(e);
348            }
349            builderFactory.setXIncludeAware(false);
350            builderFactory.setExpandEntityReferences(false);
351        }
352
353        try {
354            return builderFactory.newDocumentBuilder();
355        } catch (final ParserConfigurationException e) {
356            throw new ConfigurateException(e);
357        }
358    }

360    private Transformer newTransformer() throws ConfigurateException {
361        final TransformerFactory transformerFactory = TransformerFactory.newInstance();
362        if (!this.resolvesExternalContent) {
363            transformerFactory.setAttribute(XMLConstants.ACCESS_EXTERNAL_DTD, "");
364            transformerFactory.setAttribute(XMLConstants.ACCESS_EXTERNAL_STYLESHEET, "");
365        }
366        try {
367            final Transformer transformer = transformerFactory.newTransformer();
368
369            // we write the header ourselves.
370            transformer.setOutputProperty(OutputKeys.OMIT_XML_DECLARATION, "yes");
371
372            if (this.indent > 0) {
373                transformer.setOutputProperty(OutputKeys.INDENT, "yes");
374                transformer.setOutputProperty(INDENT_PROPERTY, Integer.toString(this.indent));
375            }
376            return transformer;
377        } catch (final TransformerConfigurationException e) {
378            throw new ConfigurateException(e);
379        }
380    }

382    @Override
383    public @NonNull AttributedConfigurationNode load(@NonNull ConfigurationOptions options) throws ParsingException {
384        if (source == null) {
385            throw new ParsingException(-1, -1, "", "No source present to read from!", null);
386        }
387        try (BufferedReader reader = source.call()) {
388            final DocumentBuilder documentBuilder = newDocumentBuilder();
389
390            final Document document;
391            try {
392                document = documentBuilder.parse(new InputSource(reader));
393            } catch (final SAXParseException ex) {
394                throw new ParsingException(ex.getLineNumber(), ex.getColumnNumber(), "", ex.getMessage(), ex.getCause());
395            } catch (final SAXException e) {
396                throw new ParsingException(-1, -1, null, null, e);
397            }
398
399            final NodeList children = document.getChildNodes();
400            for (int i = 0; i < children.getLength(); ++i) {
401                final Node child = children.item(i);
402                if (child.getNodeType() == Node.COMMENT_NODE) {
403                    options = options.header(unwrapHeader(child.getTextContent().trim()));
404                } else if (child.getNodeType() == Node.ELEMENT_NODE) {
405                    final AttributedConfigurationNode node = createNode(options);
406                    readElement(child, node);
407                    return node;
408                }
409            }
410            // empty document, fall through
411        } catch (final FileNotFoundException | NoSuchFileException e) {
412            // Squash -- there's nothing to read
413        } catch (final ParsingException ex) {
414            throw ex;
415        } catch (final Exception e) {
416            throw new ParsingException(-1, -1, "", null, e);
417        }
418        return createNode(options);
419    }

421    /**
422     * Given a single comment node's comment, clear any prefix lines.
423     *
424     * @param headerContent the content of a header
425     * @return a formatted header, with lines separated by {@link #CONFIGURATE_LINE_SEPARATOR}
426     */
427    private String unwrapHeader(final String headerContent) {
428        if (headerContent.isEmpty()) {
429            return headerContent;
430        }
431        // TODO: 4.0 may have changed behaviour here when moving away from Guava
432        return CONFIGURATE_LINE_PATTERN.splitAsStream(headerContent)
433                .map(line -> {
434                    final String trimmedLine = line.trim();
435                    if (trimmedLine.startsWith(HEADER_PREFIX)) {
436                        line = line.substring(line.indexOf(HEADER_PREFIX) + 1);
437                    }
438
439                    if (line.length() > 0 && line.charAt(0) == ' ') {
440                        line = line.substring(1);
441                    }
442                    return line;
443                }).filter(line -> !line.isEmpty())
444                .collect(Collectors.joining(CONFIGURATE_LINE_SEPARATOR));
445    }

452    private enum NodeType {
453        MAP, LIST
454    }
455
456    private void readElement(final Node from, final AttributedConfigurationNode to) {
457        @Nullable NodeType type = null;
458
459        // copy the name of the tag
460        to.tagName(from.getNodeName());
461
462        final String potentialComment = (String) from.getUserData(USER_DATA_COMMENT);
463        if (potentialComment != null) {
464            to.comment(potentialComment);
465        }
466
467        // copy attributes
468        if (from.hasAttributes()) {
469            final NamedNodeMap attributes = from.getAttributes();
470            for (int i = 0; i < attributes.getLength(); i++) {
471                final Node attribute = attributes.item(i);
472                final String key = attribute.getNodeName();
473                final String value = attribute.getNodeValue();
474
475                // read the type of the node
476                if (key.equals(ATTRIBUTE_TYPE)) {
477                    if (value.equals("map")) {
478                        type = NodeType.MAP;
479                    } else if (value.equals("list")) {
480                        type = NodeType.LIST;
481                    }
482
483                    // don't add internal configurate attributes to the node
484                    continue;
485                }
486
487                to.addAttribute(key, value);
488            }
489        }
490
491        // read out the child nodes into a multimap
492        final Map<String, Collection<Node>> children = new LinkedHashMap<>();
493        if (from.hasChildNodes()) {
494            final StringBuilder comment = new StringBuilder();
495            final NodeList childNodes = from.getChildNodes();
496            for (int i = 0; i < childNodes.getLength(); i++) {
497                final Node child = childNodes.item(i);
498                if (child.getNodeType() == Node.ELEMENT_NODE) {
499                    children.computeIfAbsent(child.getNodeName(), $ -> new ArrayList<>()).add(child);
500                    if (comment.length() > 0) {
501                        child.setUserData(USER_DATA_COMMENT, comment.toString(), null);
502                        comment.setLength(0);
503                    }
504                } else if (child.getNodeType() == Node.COMMENT_NODE) {
505                    if (comment.length() > 0) {
506                        comment.append('\n');
507                    }
508
509                    comment.append(child.getTextContent().trim());
510                }
511            }
512        }
513
514        // if there are no child nodes present, assume it's a scalar value
515        if (children.isEmpty()) {
516            to.raw(parseValue(from.getTextContent()));
517            return;
518        }
519
520        // if type is null, we need to infer what type the element is
521        if (type == null) {
522            // if there are no duplicate keys, we can infer that it is a map
523            // otherwise, assume it's a list
524            type = NodeType.MAP;
525            for (Collection<Node> child : children.values()) {
526                if (child.size() > 1) {
527                    type = NodeType.LIST;
528                    break;
529                }
530            }
531        }
532
533        if (type == NodeType.MAP) {
534            to.raw(Collections.emptyMap());
535        } else {
536            to.raw(Collections.emptyList());
537        }
538
539        // read out the elements
540        for (Map.Entry<String, Collection<Node>> entry : children.entrySet()) {
541            AttributedConfigurationNode child;
542            if (type == NodeType.MAP) {
543                child = to.node(entry.getKey());
544                readElement(entry.getValue().iterator().next(), child);
545            } else {
546                for (Node element : entry.getValue()) {
547                    child = to.appendListNode();
548                    readElement(element, child);
549                }
550            }
551        }
552    }

556        if (this.includeXmlDeclaration) {
557            writer.write("<?xml version=\"1.0\" encoding=\"UTF-8\"?>");
558            writer.write(SYSTEM_LINE_SEPARATOR);
559        }

562    @Override
563    protected void saveInternal(final ConfigurationNode node, final Writer writer) throws ConfigurateException {
564        final DocumentBuilder documentBuilder = newDocumentBuilder();
565        final Document document = documentBuilder.newDocument();
566
567        final @Nullable Node comment = createCommentNode(document, node);
568        if (comment != null) {
569            document.appendChild(comment);
570        }
571
572        document.appendChild(writeNode(document, node, null));
573
574        final Transformer transformer = newTransformer();
575        final DOMSource source = new DOMSource(document);
576        try {
577            transformer.transform(source, new StreamResult(writer));
578        } catch (final TransformerException e) {
579            throw new ConfigurateException(node, e);
580        }
581    }

583    private void appendCommentIfNecessary(final Element parent, final ConfigurationNode node) {
584        final @Nullable Node possibleComment = createCommentNode(parent.getOwnerDocument(), node);
585        if (possibleComment != null) {
586            parent.appendChild(possibleComment);
587        }
588    }

590    private @Nullable Node createCommentNode(final Document doc, final ConfigurationNode node) {
591        if (node instanceof CommentedConfigurationNodeIntermediary<?>) {
592            final @Nullable String comment = ((CommentedConfigurationNodeIntermediary<?>) node).comment();
593            if (comment != null) {
594                return doc.createComment(" " + comment.trim() + " ");
595            }
596        }
597        return null;
598    }
599
600    private Element writeNode(final Document document, final ConfigurationNode node, final @Nullable String forcedTag) {
601        String tag = this.defaultTagName;
602        Map<String, String> attributes = Collections.emptyMap();
603
604        if (node instanceof AttributedConfigurationNode) {
605            final AttributedConfigurationNode attributedNode = (AttributedConfigurationNode) node;
606            tag = attributedNode.tagName();
607            attributes = attributedNode.attributes();
608        }
609
610        final Element element = document.createElement(forcedTag == null ? tag : forcedTag);
611        for (final Map.Entry<String, String> attribute : attributes.entrySet()) {
612            element.setAttribute(attribute.getKey(), attribute.getValue());
613        }
614
615        if (node.isMap()) {
616            for (final Map.Entry<Object, ? extends ConfigurationNode> child : node.childrenMap().entrySet()) {
617                appendCommentIfNecessary(element, child.getValue());
618                element.appendChild(writeNode(document, child.getValue(), child.getKey().toString()));
619            }
620        } else if (node.isList()) {
621            if (this.writeExplicitType) {
622                element.setAttribute(ATTRIBUTE_TYPE, "list");
623            }
624            for (final ConfigurationNode child : node.childrenList()) {
625                appendCommentIfNecessary(element, child);
626                element.appendChild(writeNode(document, child, null));
627            }
628        } else {
629            element.appendChild(document.createTextNode(Objects.toString(node.rawScalar())));
630        }
631
632        return element;
633    }

641    private static Object parseValue(final String value) {
642        if (value.equals("true") || value.equals("false")) {
643            return Boolean.parseBoolean(value);
644        }
645
646        try {
647            final double doubleValue = Double.parseDouble(value);
648            if (isInteger(doubleValue)) {
649                final long longValue = Long.parseLong(value); // prevent losing precision
650                final int intValue = (int) longValue;
651                if (longValue == intValue) {
652                    return intValue;
653                } else {
654                    return longValue;
655                }
656            }
657            return doubleValue;
658        } catch (final NumberFormatException e) {
659            return value;
660        }
661    }
662
663    private static boolean isInteger(final double value) {
664        return !Double.isNaN(value) && Double.isFinite(value) && value == Math.rint(value);
665    }
