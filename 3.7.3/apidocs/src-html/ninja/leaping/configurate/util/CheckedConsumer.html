017package ninja.leaping.configurate.util;
018
019import java.util.function.Consumer;
020
021/**
022 * A functional interface similar to Consumer, except allowing contained methods to throw exceptions
023 *
024 * @param <V> The value accepted
025 * @param <E> The exception type thrown
026 */
027@FunctionalInterface
028public interface CheckedConsumer<V, E extends Throwable> {
029    void accept(V value) throws E;
030
031    /**
032     * Create an instance from an ordinary consumer
033     *
034     * @param consumer The consumer to convert
035     * @param <V> The type returned by the consumer
036     * @return A function that executes the provided consumer
037     */
038    static <V> CheckedConsumer<V, RuntimeException> fromConsumer(Consumer<V> consumer) {
039        return consumer::accept;
040    }
041}
