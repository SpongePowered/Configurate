292    private DocumentBuilder newDocumentBuilder() throws ConfigurateException {
293        final DocumentBuilderFactory builderFactory = DocumentBuilderFactory.newInstance();
294        if (this.schema != null) {
295            builderFactory.setSchema(this.schema);
296        }
297        if (!this.resolvesExternalContent) {
298            // Settings based on https://cheatsheetseries.owasp.org/cheatsheets/XML_External_Entity_Prevention_Cheat_Sheet.html
299            try {
300                builderFactory.setFeature(FEATURE_EXTERNAL_GENERAL_ENTITIES, false);
301                builderFactory.setFeature(FEATURE_EXTERNAL_PARAMETER_ENTITIES, false);
302                builderFactory.setFeature(FEATURE_LOAD_EXTERNAL_DTD, false);
303            } catch (final ParserConfigurationException e) {
304                throw new ConfigurateException(e);
305            }
306            builderFactory.setXIncludeAware(false);
307            builderFactory.setExpandEntityReferences(false);
308        }
309
310        try {
311            return builderFactory.newDocumentBuilder();
312        } catch (final ParserConfigurationException e) {
313            throw new ConfigurateException(e);
314        }
315    }

317    private Transformer newTransformer() throws ConfigurateException {
318        final TransformerFactory transformerFactory = TransformerFactory.newInstance();
319        if (!this.resolvesExternalContent) {
320            transformerFactory.setAttribute(XMLConstants.ACCESS_EXTERNAL_DTD, "");
321            transformerFactory.setAttribute(XMLConstants.ACCESS_EXTERNAL_STYLESHEET, "");
322        }
323        try {
324            final Transformer transformer = transformerFactory.newTransformer();
325
326            // we write the header ourselves.
327            transformer.setOutputProperty(OutputKeys.OMIT_XML_DECLARATION, "yes");
328
329            if (this.indent > 0) {
330                transformer.setOutputProperty(OutputKeys.INDENT, "yes");
331                transformer.setOutputProperty(INDENT_PROPERTY, Integer.toString(this.indent));
332            }
333            return transformer;
334        } catch (final TransformerConfigurationException e) {
335            throw new ConfigurateException(e);
336        }
337    }

339    @Override
340    public @NonNull AttributedConfigurationNode load(@NonNull ConfigurationOptions options) throws ParsingException {
341        if (source == null) {
342            throw new ParsingException(-1, -1, "", "No source present to read from!", null);
343        }
344        try (BufferedReader reader = source.call()) {
345            final DocumentBuilder documentBuilder = newDocumentBuilder();
346
347            final Document document;
348            try {
349                document = documentBuilder.parse(new InputSource(reader));
350            } catch (final SAXParseException ex) {
351                throw new ParsingException(ex.getLineNumber(), ex.getColumnNumber(), "", ex.getMessage(), ex.getCause());
352            } catch (final SAXException e) {
353                throw new ParsingException(-1, -1, null, null, e);
354            }
355
356            final NodeList children = document.getChildNodes();
357            for (int i = 0; i < children.getLength(); ++i) {
358                final Node child = children.item(i);
359                if (child.getNodeType() == Node.COMMENT_NODE) {
360                    options = options.header(unwrapHeader(child.getTextContent().trim()));
361                } else if (child.getNodeType() == Node.ELEMENT_NODE) {
362                    final AttributedConfigurationNode node = createNode(options);
363                    readElement(child, node);
364                    return node;
365                }
366            }
367            // empty document, fall through
368        } catch (final FileNotFoundException | NoSuchFileException e) {
369            // Squash -- there's nothing to read
370        } catch (final ParsingException ex) {
371            throw ex;
372        } catch (final Exception e) {
373            throw new ParsingException(-1, -1, "", null, e);
374        }
375        return createNode(options);
376    }

378    /**
379     * Given a single comment node's comment, clear any prefix lines.
380     *
381     * @param headerContent the content of a header
382     * @return a formatted header, with lines separated by {@link #CONFIGURATE_LINE_SEPARATOR}
383     */
384    private String unwrapHeader(final String headerContent) {
385        if (headerContent.isEmpty()) {
386            return headerContent;
387        }
388        // TODO: 4.0 may have changed behaviour here when moving away from Guava
389        return CONFIGURATE_LINE_PATTERN.splitAsStream(headerContent)
390                .map(line -> {
391                    final String trimmedLine = line.trim();
392                    if (trimmedLine.startsWith(HEADER_PREFIX)) {
393                        line = line.substring(line.indexOf(HEADER_PREFIX) + 1);
394                    }
395
396                    if (line.startsWith(" ")) {
397                        line = line.substring(1);
398                    }
399                    return line;
400                }).filter(line -> !line.isEmpty())
401                .collect(Collectors.joining(CONFIGURATE_LINE_SEPARATOR));
402    }

409    private enum NodeType {
410        MAP, LIST
411    }
412
413    private void readElement(final Node from, final AttributedConfigurationNode to) {
414        @Nullable NodeType type = null;
415
416        // copy the name of the tag
417        to.tagName(from.getNodeName());
418
419        final String potentialComment = (String) from.getUserData(USER_DATA_COMMENT);
420        if (potentialComment != null) {
421            to.comment(potentialComment);
422        }
423
424        // copy attributes
425        if (from.hasAttributes()) {
426            final NamedNodeMap attributes = from.getAttributes();
427            for (int i = 0; i < attributes.getLength(); i++) {
428                final Node attribute = attributes.item(i);
429                final String key = attribute.getNodeName();
430                final String value = attribute.getNodeValue();
431
432                // read the type of the node
433                if (key.equals(ATTRIBUTE_TYPE)) {
434                    if (value.equals("map")) {
435                        type = NodeType.MAP;
436                    } else if (value.equals("list")) {
437                        type = NodeType.LIST;
438                    }
439
440                    // don't add internal configurate attributes to the node
441                    continue;
442                }
443
444                to.addAttribute(key, value);
445            }
446        }
447
448        // read out the child nodes into a multimap
449        final Map<String, Collection<Node>> children = new LinkedHashMap<>();
450        if (from.hasChildNodes()) {
451            final StringBuilder comment = new StringBuilder();
452            final NodeList childNodes = from.getChildNodes();
453            for (int i = 0; i < childNodes.getLength(); i++) {
454                final Node child = childNodes.item(i);
455                if (child.getNodeType() == Node.ELEMENT_NODE) {
456                    children.computeIfAbsent(child.getNodeName(), $ -> new ArrayList<>()).add(child);
457                    if (comment.length() > 0) {
458                        child.setUserData(USER_DATA_COMMENT, comment.toString(), null);
459                        comment.setLength(0);
460                    }
461                } else if (child.getNodeType() == Node.COMMENT_NODE) {
462                    if (comment.length() > 0) {
463                        comment.append('\n');
464                    }
465
466                    comment.append(child.getTextContent().trim());
467                }
468            }
469        }
470
471        // if there are no child nodes present, assume it's a scalar value
472        if (children.isEmpty()) {
473            to.raw(parseValue(from.getTextContent()));
474            return;
475        }
476
477        // if type is null, we need to infer what type the element is
478        if (type == null) {
479            // if there are no duplicate keys, we can infer that it is a map
480            // otherwise, assume it's a list
481            type = NodeType.MAP;
482            for (Collection<Node> child : children.values()) {
483                if (child.size() > 1) {
484                    type = NodeType.LIST;
485                    break;
486                }
487            }
488        }
489
490        if (type == NodeType.MAP) {
491            to.raw(Collections.emptyMap());
492        } else {
493            to.raw(Collections.emptyList());
494        }
495
496        // read out the elements
497        for (Map.Entry<String, Collection<Node>> entry : children.entrySet()) {
498            AttributedConfigurationNode child;
499            if (type == NodeType.MAP) {
500                child = to.node(entry.getKey());
501                readElement(entry.getValue().iterator().next(), child);
502            } else {
503                for (Node element : entry.getValue()) {
504                    child = to.appendListNode();
505                    readElement(element, child);
506                }
507            }
508        }
509    }

511    @Override
512    protected void writeHeaderInternal(final Writer writer) throws IOException {
513        if (this.includeXmlDeclaration) {
514            writer.write("<?xml version=\"1.0\" encoding=\"UTF-8\"?>");
515            writer.write(SYSTEM_LINE_SEPARATOR);
516        }
517    }

519    @Override
520    protected void saveInternal(final ConfigurationNode node, final Writer writer) throws ConfigurateException {
521        final DocumentBuilder documentBuilder = newDocumentBuilder();
522        final Document document = documentBuilder.newDocument();
523
524        final @Nullable Node comment = createCommentNode(document, node);
525        if (comment != null) {
526            document.appendChild(comment);
527        }
528
529        document.appendChild(writeNode(document, node, null));
530
531        final Transformer transformer = newTransformer();
532        final DOMSource source = new DOMSource(document);
533        try {
534            transformer.transform(source, new StreamResult(writer));
535        } catch (final TransformerException e) {
536            throw new ConfigurateException(node, e);
537        }
538    }

540    private void appendCommentIfNecessary(final Element parent, final ConfigurationNode node) {
541        final @Nullable Node possibleComment = createCommentNode(parent.getOwnerDocument(), node);
542        if (possibleComment != null) {
543            parent.appendChild(possibleComment);
544        }
545    }
546
547    private @Nullable Node createCommentNode(final Document doc, final ConfigurationNode node) {
548        if (node instanceof CommentedConfigurationNodeIntermediary<?>) {
549            final @Nullable String comment = ((CommentedConfigurationNodeIntermediary<?>) node).comment();
550            if (comment != null) {
551                return doc.createComment(" " + comment.trim() + " ");
552            }
553        }
554        return null;
555    }
556
557    private Element writeNode(final Document document, final ConfigurationNode node, final @Nullable String forcedTag) {
558        String tag = this.defaultTagName;
559        Map<String, String> attributes = Collections.emptyMap();
560
561        if (node instanceof AttributedConfigurationNode) {
562            final AttributedConfigurationNode attributedNode = (AttributedConfigurationNode) node;
563            tag = attributedNode.tagName();
564            attributes = attributedNode.attributes();
565        }
566
567        final Element element = document.createElement(forcedTag == null ? tag : forcedTag);
568        for (final Map.Entry<String, String> attribute : attributes.entrySet()) {
569            element.setAttribute(attribute.getKey(), attribute.getValue());
570        }
571
572        if (node.isMap()) {
573            for (final Map.Entry<Object, ? extends ConfigurationNode> child : node.childrenMap().entrySet()) {
574                appendCommentIfNecessary(element, child.getValue());
575                element.appendChild(writeNode(document, child.getValue(), child.getKey().toString()));
576            }
577        } else if (node.isList()) {
578            if (this.writeExplicitType) {
579                element.setAttribute(ATTRIBUTE_TYPE, "list");
580            }
581            for (final ConfigurationNode child : node.childrenList()) {
582                appendCommentIfNecessary(element, child);
583                element.appendChild(writeNode(document, child, null));
584            }
585        } else {
586            element.appendChild(document.createTextNode(Objects.toString(node.rawScalar())));
587        }
588
589        return element;
590    }

594        options = options.nativeTypes(NATIVE_TYPES);

598    private static Object parseValue(final String value) {
599        if (value.equals("true") || value.equals("false")) {
600            return Boolean.parseBoolean(value);
601        }
602
603        try {
604            final double doubleValue = Double.parseDouble(value);
605            if (isInteger(doubleValue)) {
606                final long longValue = Long.parseLong(value); // prevent losing precision
607                final int intValue = (int) longValue;
608                if (longValue == intValue) {
609                    return intValue;
610                } else {
611                    return longValue;
612                }
613            }
614            return doubleValue;
615        } catch (final NumberFormatException e) {
616            return value;
617        }
618    }

620    private static boolean isInteger(final double value) {
621        return !Double.isNaN(value) && Double.isFinite(value) && value == Math.rint(value);
622    }
