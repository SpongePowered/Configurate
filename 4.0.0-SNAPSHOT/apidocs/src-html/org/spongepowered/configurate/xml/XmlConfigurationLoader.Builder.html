226    private XmlConfigurationLoader(Builder builder) {

235    private DocumentBuilder newDocumentBuilder() {
236        DocumentBuilderFactory builderFactory = DocumentBuilderFactory.newInstance();
237        if (schema != null) {
238            builderFactory.setSchema(schema);
239        }
240
241        try {
242            return builderFactory.newDocumentBuilder();
243        } catch (ParserConfigurationException e) {
244            throw new RuntimeException(e);
245        }
246    }
247
248    private Transformer newTransformer() {
249        TransformerFactory transformerFactory = TransformerFactory.newInstance();
250        try {
251            Transformer transformer = transformerFactory.newTransformer();
252
253            // we write the header ourselves.
254            transformer.setOutputProperty(OutputKeys.OMIT_XML_DECLARATION, "yes");
255
256            if (indent > 0) {
257                transformer.setOutputProperty(OutputKeys.INDENT, "yes");
258                transformer.setOutputProperty(INDENT_PROPERTY, Integer.toString(indent));
259            }
260            return transformer;
261        } catch (TransformerConfigurationException e) {
262            throw new RuntimeException(e);
263        }
264    }

266    @Override
267    public @NonNull AttributedConfigurationNode load(@NonNull ConfigurationOptions options) throws IOException {
268        if (source == null) {
269            throw new IOException("No source present to read from!");
270        }
271        try (BufferedReader reader = source.call()) {
272            DocumentBuilder documentBuilder = newDocumentBuilder();
273
274            Document document;
275            try {
276                document = documentBuilder.parse(new InputSource(reader));
277            } catch (SAXException e) {
278                throw new IOException(e);
279            }
280
281            NodeList children = document.getChildNodes();
282            for (int i = 0; i < children.getLength(); ++i) {
283                Node child = children.item(i);
284                System.out.println(child);
285                if (child.getNodeType() == Node.COMMENT_NODE) {
286                    options = options.withHeader(unwrapHeader(child.getTextContent().trim()));
287                } else if (child.getNodeType() == Node.ELEMENT_NODE) {
288                    AttributedConfigurationNode node = createEmptyNode(options);
289                    readElement(child, node);
290                    return node;
291                }
292            }
293            // empty document, fall through
294        } catch (FileNotFoundException | NoSuchFileException e) {
295            // Squash -- there's nothing to read
296        } catch (IOException e) {
297            throw e;
298        } catch (Exception e) {
299            throw new IOException(e);
300        }
301        return createEmptyNode(options);
302    }

304    /**
305     * Given a single comment node's comment, clear any prefix lines
306     * @param headerContent The content of a header
307     * @return A formatted header, with lines separated by {@link #CONFIGURATE_LINE_SEPARATOR}
308     */
309    private String unwrapHeader(String headerContent) {
310        if (headerContent.isEmpty()) {
311            return headerContent;
312        }
313        StringBuilder result = new StringBuilder();
314        for (Iterator<String> it = LINE_SPLITTER.split(headerContent).iterator(); it.hasNext();) {
315            String line = it.next();
316            final String trimmedLine = line.trim();
317            if (trimmedLine.startsWith(HEADER_PREFIX)) {
318                line = line.substring(line.indexOf(HEADER_PREFIX) + 1);
319            }
320
321            if (line.startsWith(" ")) {
322                line = line.substring(1);
323            }
324
325            if (it.hasNext() || !line.isEmpty()) {
326                result.append(line);
327            }
328
329            if (it.hasNext()) {
330                result.append(CONFIGURATE_LINE_SEPARATOR);
331            }
332        }
333        return result.toString();
334    }

341    private enum NodeType {
342        MAP, LIST
343    }
344
345    private void readElement(Node from, AttributedConfigurationNode to) {
346        @Nullable NodeType type = null;
347
348        // copy the name of the tag
349        to.setTagName(from.getNodeName());
350
351        String potentialComment = (String) from.getUserData(USER_DATA_COMMENT);
352        if (potentialComment != null) {
353            to.setComment(potentialComment);
354        }
355
356        // copy attributes
357        if (from.hasAttributes()) {
358            NamedNodeMap attributes = from.getAttributes();
359            for (int i = 0; i < attributes.getLength(); i++) {
360                Node attribute = attributes.item(i);
361                String key = attribute.getNodeName();
362                String value = attribute.getNodeValue();
363
364                // read the type of the node
365                if (key.equals(ATTRIBUTE_TYPE)) {
366                    if (value.equals("map")) {
367                        type = NodeType.MAP;
368                    } else if (value.equals("list")) {
369                        type = NodeType.LIST;
370                    }
371
372                    // don't add internal configurate attributes to the node
373                    continue;
374                }
375
376                to.addAttribute(key, value);
377            }
378        }
379
380        // read out the child nodes into a multimap
381        Multimap<String, Node> children = MultimapBuilder.linkedHashKeys().arrayListValues().build();
382        if (from.hasChildNodes()) {
383            StringBuilder comment = new StringBuilder();
384            NodeList childNodes = from.getChildNodes();
385            for (int i = 0; i < childNodes.getLength(); i++) {
386                Node child = childNodes.item(i);
387                if (child.getNodeType() == Node.ELEMENT_NODE) {
388                    children.put(child.getNodeName(), child);
389                    if (comment.length() > 0) {
390                        child.setUserData(USER_DATA_COMMENT, comment.toString(), null);
391                        comment.setLength(0);
392                    }
393                } else if (child.getNodeType() == Node.COMMENT_NODE) {
394                    if (comment.length() > 0) {
395                        comment.append('\n');
396                    }
397
398                    comment.append(child.getTextContent().trim());
399                }
400            }
401        }
402
403        // if there are no child nodes present, assume it's a scalar value
404        if (children.isEmpty()) {
405            to.setValue(parseValue(from.getTextContent()));
406            return;
407        }
408
409        // if type is null, we need to infer what type the element is
410        if (type == null) {
411            // if there are no duplicate keys, we can infer that it is a map
412            // otherwise, assume it's a list
413            if (children.keys().size() == children.keySet().size()) {
414                type = NodeType.MAP;
415            } else {
416                type = NodeType.LIST;
417            }
418        }
419
420        if (type == NodeType.MAP) {
421            to.setValue(ImmutableMap.of());
422        } else {
423            to.setValue(ImmutableList.of());
424        }
425
426        // read out the elements
427        for (Map.Entry<String, Node> entry : children.entries()) {
428            AttributedConfigurationNode child;
429            if (type == NodeType.MAP) {
430                child = to.getNode(entry.getKey());
431            } else {
432                child = to.appendListNode();
433            }
434
435            readElement(entry.getValue(), child);
436        }
437    }

439    @Override
440    protected void writeHeaderInternal(Writer writer) throws IOException {
441        if (includeXmlDeclaration) {
442            writer.write("<?xml version=\"1.0\" encoding=\"UTF-8\"?>");
443            writer.write(SYSTEM_LINE_SEPARATOR);
444        }

447    @Override
448    protected void saveInternal(ConfigurationNode node, Writer writer) throws IOException {
449        DocumentBuilder documentBuilder = newDocumentBuilder();
450        Document document = documentBuilder.newDocument();
451
452        @Nullable Node comment = createCommentNode(document, node);
453        if (comment != null) {
454            document.appendChild(comment);
455        }
456
457        document.appendChild(writeNode(document, node, null));
458
459        Transformer transformer = newTransformer();
460        DOMSource source = new DOMSource(document);
461        try {
462            transformer.transform(source, new StreamResult(writer));
463        } catch (TransformerException e) {
464            throw new IOException(e);
465        }
466    }

468    private void appendCommentIfNecessary(Element parent, ConfigurationNode node) {
469        @Nullable Node possibleComment = createCommentNode(parent.getOwnerDocument(), node);
470        if (possibleComment != null) {
471            parent.appendChild(possibleComment);
472        }
473    }
474
475    private @Nullable Node createCommentNode(Document doc, ConfigurationNode node) {
476        if (node instanceof CommentedConfigurationNodeIntermediary<?>) {
477            @Nullable String comment = ((CommentedConfigurationNodeIntermediary<?>) node).getComment().orElse(null);
478            if (comment != null) {
479                return doc.createComment(" " + comment.trim() + " ");
480            }
481        }
482        return null;
483    }
484
485    private Element writeNode(Document document, ConfigurationNode node, @Nullable String forcedTag) {
486        String tag = defaultTagName;
487        Map<String, String> attributes = ImmutableMap.of();
488
489        if (node instanceof AttributedConfigurationNode) {
490            AttributedConfigurationNode attributedNode = ((AttributedConfigurationNode) node);
491            tag = attributedNode.getTagName();
492            attributes = attributedNode.getAttributes();
493        }
494
495        Element element = document.createElement(forcedTag == null ? tag : forcedTag);
496        for (Map.Entry<String, String> attribute : attributes.entrySet()) {
497            element.setAttribute(attribute.getKey(), attribute.getValue());
498        }
499
500        if (node.isMap()) {
501            for (Map.Entry<Object, ? extends ConfigurationNode> child : node.getChildrenMap().entrySet()) {
502                appendCommentIfNecessary(element, child.getValue());
503                element.appendChild(writeNode(document, child.getValue(), child.getKey().toString()));
504            }
505        } else if (node.isList()) {
506            if (writeExplicitType) {
507                element.setAttribute(ATTRIBUTE_TYPE, "list");
508            }
509            for (ConfigurationNode child : node.getChildrenList()) {
510                appendCommentIfNecessary(element, child);
511                element.appendChild(writeNode(document, child, null));
512            }
513        } else {
514            element.appendChild(document.createTextNode(Objects.toString(node.getValue())));
515        }
516
517        return element;
518    }

520    @Override
521    public AttributedConfigurationNode createEmptyNode(ConfigurationOptions options) {
522        options = options.withAcceptedTypes(ImmutableSet.of(Double.class, Long.class,
523                Integer.class, Boolean.class, String.class, Number.class));
524        return AttributedConfigurationNode.root("root", options);
525    }

527    private static Object parseValue(String value) {
528        if (value.equals("true") || value.equals("false")) {
529            return Boolean.parseBoolean(value);
530        }
531
532        try {
533            double doubleValue = Double.parseDouble(value);
534            if (DoubleMath.isMathematicalInteger(doubleValue)) {
535                long longValue = (long) doubleValue;
536                int intValue = (int) longValue;
537                if (longValue == intValue) {
538                    return intValue;
539                } else {
540                    return longValue;
541                }
542            }
543            return doubleValue;
544        } catch (NumberFormatException e) {
545            return value;
546        }
547    }
