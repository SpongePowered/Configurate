017package org.spongepowered.configurate;
018
019import org.checkerframework.checker.nullness.qual.Nullable;
020
021/**
022 * Intermediate interface for different types of commented configuration nodes.
023 *
024 * @param <N> self type
025 */
026public interface CommentedConfigurationNodeIntermediary<N extends CommentedConfigurationNodeIntermediary<N>> extends ScopedConfigurationNode<N> {
027
028    /**
029     * Gets the current value for the comment.
030     *
031     * <p>If the comment contains multiple lines, the lines will be split
032     * by \n</p>
033     *
034     * @return the configuration's current comment
035     */
036    @Nullable String comment();
037
038    /**
039     * Sets the comment for this configuration node.
040     *
041     * @param comment the comment to set. Line breaks should be represented as
042     *                LFs (\n)
043     * @return this node
044     */
045    N comment(@Nullable String comment);
046
047    /**
048     * Set a comment on this node if it does not presently have a comment.
049     *
050     * <p>The provided comment must not be null, because setting a null comment
051     * would be a no-op</p>
052     *
053     * @param comment the comment to set. Line breaks should be represented as
054     *                LFs (\n)
055     * @return this node
056     */
057    N commentIfAbsent(String comment);
058
059}
