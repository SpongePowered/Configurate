020import com.google.common.primitives.Primitives;
021import org.checkerframework.checker.nullness.qual.NonNull;
022import org.checkerframework.checker.nullness.qual.Nullable;
023import org.spongepowered.configurate.loader.AbstractConfigurationLoader;
024import org.spongepowered.configurate.loader.ConfigurationLoader;
025import org.spongepowered.configurate.objectmapping.DefaultObjectMapperFactory;
026import org.spongepowered.configurate.objectmapping.ObjectMapperFactory;
027import org.spongepowered.configurate.serialize.TypeSerializerCollection;

031import java.util.Objects;

035import static java.util.Objects.requireNonNull;

037/**
038 * This object is a holder for general configuration options.
039 *
040 * <p>This is meant to hold options that are used in configuring how the configuration data
041 * structures are handled, rather than the serialization configuration that is located in
042 * {@link ConfigurationLoader}s.</p>
043 *
044 * <p>This class is immutable.</p>
045 */
046public final class ConfigurationOptions {
047    private static final ConfigurationOptions DEFAULTS = new ConfigurationOptions(MapFactories.insertionOrdered(), null,
048        TypeSerializerCollection.defaults(), null, DefaultObjectMapperFactory.getInstance(), false);
049    @NonNull private final MapFactory mapFactory;
050    @Nullable private final String header;
051    @NonNull private final TypeSerializerCollection serializers;
052    @Nullable private final ImmutableSet<Class<?>> acceptedTypes;
053    @NonNull private final ObjectMapperFactory objectMapperFactory;
054    private final boolean shouldCopyDefaults;
055
056    private ConfigurationOptions(@NonNull MapFactory mapFactory, @Nullable String header, @NonNull TypeSerializerCollection serializers, @Nullable Set<Class<?>> acceptedTypes, @NonNull ObjectMapperFactory objectMapperFactory, boolean shouldCopyDefaults) {
057        this.mapFactory = mapFactory;
058        this.header = header;
059        this.serializers = serializers;
060        this.acceptedTypes = acceptedTypes == null ? null : ImmutableSet.copyOf(acceptedTypes);
061        this.objectMapperFactory = objectMapperFactory;
062        this.shouldCopyDefaults = shouldCopyDefaults;
063    }
064
065    /**
066     * Get the default set of options. This may be overridden by your chosen configuration loader,
067     * so when building configurations it is recommended to access
068     * {@code AbstractConfigurationLoader.Builder#getDefaultOptions()} instead.
069     *
070     * @return the default options
071     */
072    @NonNull
073    public static ConfigurationOptions defaults() {
074        return DEFAULTS;
075    }
076
077    /**
078     * Gets the {@link MapFactory} specified in these options.
079     *
080     * @return The map factory
081     */
082    @NonNull
083    public MapFactory getMapFactory() {
084        return mapFactory;
085    }
086
087    /**
088     * Creates a new {@link ConfigurationOptions} instance, with the specified {@link MapFactory}
089     * set, and all other settings copied from this instance.
090     *
091     * @param mapFactory The new factory to use to create a map
092     * @return The new options object
093     */
094    @NonNull
095    public ConfigurationOptions withMapFactory(@NonNull MapFactory mapFactory) {
096        requireNonNull(mapFactory, "mapFactory");
097        if (this.mapFactory == mapFactory) {
098            return this;
099        }
100        return new ConfigurationOptions(mapFactory, header, serializers, acceptedTypes, objectMapperFactory, shouldCopyDefaults);
101    }
102
103    /**
104     * Gets the header specified in these options.
105     *
106     * @return The current header. Lines are split by \n,
107     */
108    @Nullable
109    public String getHeader() {
110        return this.header;
111    }
112
113    /**
114     * Creates a new {@link ConfigurationOptions} instance, with the specified header
115     * set, and all other settings copied from this instance.
116     *
117     * @param header The new header to use
118     * @return The new options object
119     */
120    @NonNull
121    public ConfigurationOptions withHeader(@Nullable String header) {
122        if (Objects.equals(this.header, header)) {
123            return this;
124        }
125        return new ConfigurationOptions(mapFactory, header, serializers, acceptedTypes, objectMapperFactory, shouldCopyDefaults);
126    }
127
128    /**
129     * Gets the {@link TypeSerializerCollection} specified in these options.
130     *
131     * @return The type serializers
132     */
133    @NonNull
134    public TypeSerializerCollection getSerializers() {
135        return this.serializers;
136    }
137
138    /**
139     * Creates a new {@link ConfigurationOptions} instance, with the specified {@link TypeSerializerCollection}
140     * set, and all other settings copied from this instance.
141     *
142     * @param serializers The serializers to use
143     * @return The new options object
144     */
145    @NonNull
146    public ConfigurationOptions withSerializers(@NonNull TypeSerializerCollection serializers) {
147        requireNonNull(serializers, "serializers");
148        if (this.serializers == serializers) {
149            return this;
150        }
151        return new ConfigurationOptions(mapFactory, header, serializers, acceptedTypes, objectMapperFactory, shouldCopyDefaults);
152    }
153
154    /**
155     * Creates a new {@link ConfigurationOptions} instance, with a new
156     * {@link TypeSerializerCollection} created as a child of this options' current collection.
157     * The provided function will be called with the builder for this new collection to allow
158     * registering more type serializers.
159     *
160     * @param serializerBuilder accepts a builder for the collection that will be used in the
161     *                          returned options object.
162     * @return The new options object
163     */
164    public @NonNull ConfigurationOptions withSerializers(@NonNull Consumer<TypeSerializerCollection.Builder> serializerBuilder) {
165        requireNonNull(serializerBuilder, "serializerBuilder");
166        final TypeSerializerCollection.Builder builder = this.serializers.childBuilder();
167        serializerBuilder.accept(builder);
168        return new ConfigurationOptions(mapFactory, header, builder.build(), acceptedTypes, objectMapperFactory, shouldCopyDefaults);
169    }
170
171    /**
172     * Gets the {@link ObjectMapperFactory} specified in these options.
173     *
174     * @return The factory used to construct ObjectMapper instances
175     */
176    @NonNull
177    public ObjectMapperFactory getObjectMapperFactory() {
178        return this.objectMapperFactory;
179    }
180
181    /**
182     * Creates a new {@link ConfigurationOptions} instance, with the specified {@link ObjectMapperFactory}
183     * set, and all other settings copied from this instance.
184     *
185     * @param objectMapperFactory The factory to use to produce object mapper instances. Must not be null
186     * @return updated options object
187     */
188    @NonNull
189    public ConfigurationOptions withObjectMapperFactory(@NonNull ObjectMapperFactory objectMapperFactory) {
190        requireNonNull(objectMapperFactory, "factory");
191        if (this.objectMapperFactory == objectMapperFactory) {
192            return this;
193        }
194        return new ConfigurationOptions(mapFactory, header, serializers, acceptedTypes, objectMapperFactory, shouldCopyDefaults);
195    }
196
197    /**
198     * Gets whether objects of the provided type are accepted as values for nodes with this as
199     * their options object.
200     *
201     * @param type The type to check
202     * @return Whether the type is accepted
203     */
204    public boolean acceptsType(@NonNull Class<?> type) {
205        requireNonNull(type, "type");
206
207        if (this.acceptedTypes == null) {
208            return true;
209        }
210        if (this.acceptedTypes.contains(type)) {
211            return true;
212        }
213
214        if (type.isPrimitive() && this.acceptedTypes.contains(Primitives.wrap(type))) {
215            return true;
216        }
217
218        if (Primitives.isWrapperType(type) && this.acceptedTypes.contains(Primitives.unwrap(type))) {
219            return true;
220        }
221
222        for (Class<?> clazz : this.acceptedTypes) {
223            if (clazz.isAssignableFrom(type)) {
224                return true;
225            }
226        }
227
228        return false;
229    }
230
231    /**
232     * Creates a new {@link ConfigurationOptions} instance, with the specified accepted types
233     * set, and all other settings copied from this instance.
234     *
235     * <p>'Accepted types' are types which are accepted as native values for the configuration.</p>
236     *
237     * <p>Null indicates that all types are accepted.</p>
238     *
239     * @param acceptedTypes The types that will be accepted to a call to {@link ConfigurationNode#setValue(Object)}
240     * @return updated options object
241     */
242    @NonNull
243    public ConfigurationOptions withAcceptedTypes(@Nullable Set<Class<?>> acceptedTypes) {
244        if (Objects.equals(this.acceptedTypes, acceptedTypes)) {
245            return this;
246        }
247        return new ConfigurationOptions(mapFactory, header, serializers, acceptedTypes, objectMapperFactory, shouldCopyDefaults);
248    }
249
250    /**
251     * Gets whether or not default parameters provided to {@link ConfigurationNode} getter methods
252     * should be set to the node when used.
253     *
254     * @return Whether defaults should be copied into value
255     */
256    public boolean shouldCopyDefaults() {
257        return shouldCopyDefaults;
258    }
259
260    /**
261     * Creates a new {@link ConfigurationOptions} instance, with the specified 'copy defaults' setting
262     * set, and all other settings copied from this instance.
263     *
264     * @see #shouldCopyDefaults() for information on what this method does
265     * @param shouldCopyDefaults whether to copy defaults
266     * @return updated options object
267     */
268    @NonNull
269    public ConfigurationOptions withShouldCopyDefaults(boolean shouldCopyDefaults) {
270        if (this.shouldCopyDefaults == shouldCopyDefaults) {
271            return this;
272        }
273        return new ConfigurationOptions(mapFactory, header, serializers, acceptedTypes, objectMapperFactory, shouldCopyDefaults);
274    }
275
276    @Override
277    public boolean equals(Object o) {
278        if (this == o) return true;
279        if (!(o instanceof ConfigurationOptions)) return false;
280        ConfigurationOptions that = (ConfigurationOptions) o;
281        return Objects.equals(shouldCopyDefaults, that.shouldCopyDefaults) &&
282                Objects.equals(mapFactory, that.mapFactory) &&
283                Objects.equals(header, that.header) &&
284                Objects.equals(serializers, that.serializers) &&
285                Objects.equals(acceptedTypes, that.acceptedTypes) &&
286                Objects.equals(objectMapperFactory, that.objectMapperFactory);
287    }
288
289    @Override
290    public int hashCode() {
291        return Objects.hash(mapFactory, header, serializers, acceptedTypes, objectMapperFactory, shouldCopyDefaults);
292    }
293
294    @Override
295    public String toString() {
296        return "ConfigurationOptions{" +
297                "mapFactory=" + mapFactory +
298                ", header='" + header + '\'' +
299                ", serializers=" + serializers +
300                ", acceptedTypes=" + acceptedTypes +
301                ", objectMapperFactory=" + objectMapperFactory +
302                ", shouldCopyDefaults=" + shouldCopyDefaults +
303                '}';
304    }
