017package org.spongepowered.configurate.reference;
018
019import org.checkerframework.checker.nullness.qual.Nullable;
020import org.spongepowered.configurate.ConfigurationNode;
021import org.spongepowered.configurate.reactive.Publisher;
022
023import java.util.function.Function;
024
025/**
026 * A pointer to a node within a configuration tree.
027 *
028 * <p>This value will update automatically with changes to the underlying
029 * configuration file. Subscribers will be provided the current value upon
030 * subscription, followed by any changes.
031 *
032 * @param <T> the type of value to return
033 * @param <N> the type of node
034 */
035public interface ValueReference<T, N extends ConfigurationNode> extends Publisher<T> {
036
037    /**
038     * Get the current value at this node.
039     *
040     * <p>Any deserialization failures will be submitted to the owning
041     * {@link ConfigurationReference}'s error callback
042     *
043     * @return the deserialized value, or null if deserialization fails.
044     */
045    @Nullable T get();
046
047    /**
048     * Set the new value of this node. The configuration won't be saved.
049     *
050     * <p>Any serialization errors will be provided to the error callback of
051     * the owning {@link ConfigurationReference}
052     *
053     * @param value the value
054     * @return true if successful, false if serialization fails
055     */
056    boolean set(@Nullable T value);
057
058    /**
059     * Set the new value of this node and save the underlying configuration.
060     *
061     * <p>Any serialization errors will be provided to the error callback of the
062     * owning {@link ConfigurationReference}
063     *
064     * @param value the value
065     * @return true if successful, false if serialization fails
066     */
067    boolean setAndSave(@Nullable T value); // @cs-: NoGetSetPrefix (not a property accessor)
068
069    /**
070     * Set the new value of this node and save the underlying configuration
071     * asynchronously on the executor of the owning {@link ConfigurationReference}.
072     *
073     * <p>Any serialization errors will be submitted to subscribers of the
074     * returned {@link Publisher}
075     *
076     * @param value the value
077     * @return true if successful, false if serialization fails
078     */
079    Publisher<Boolean> setAndSaveAsync(@Nullable T value); // @cs-: NoGetSetPrefix (not a property accessor)
080
081    /**
082     * Update this value and the underlying node, without saving.
083     *
084     * <p>Any serialization errors will be provided to the error callback of
085     * the owning {@link ConfigurationReference}
086     *
087     * @param action to transform this node's value
088     * @return whether this update was successful
089     */
090    boolean update(Function<@Nullable T, ? extends T> action);
091
092    /**
093     * Update, performing the action and save on the executor of the owning
094     * {@link ConfigurationReference}. Any errors that occur while saving will
095     * be passed along to any subscribers.
096     *
097     * <p>The updated value will only be exposed if the changes are successful.
098     *
099     * @param action to transform this node's value
100     * @return whether this update was successful
101     */
102    Publisher<Boolean> updateAsync(Function<@Nullable T, ? extends T> action);
103
104    /**
105     * Get the node this value reference points to.
106     *
107     * @return the node
108     */
109    N node();
110
111}
