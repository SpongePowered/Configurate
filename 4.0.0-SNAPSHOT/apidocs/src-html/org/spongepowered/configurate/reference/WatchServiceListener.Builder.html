046/**
047 * A wrapper around NIO's {@link WatchService} that uses the provided watch key
048 * to poll for changes, and calls listeners once an event occurs.
049 *
050 * <p>Some deduplication is performed because Windows can be fairly spammy with
051 * its events, so one callback may receive multiple events at one time.
052 *
053 * <p>Callback functions are {@link Subscriber Subscribers} that take the
054 * {@link WatchEvent} as their parameter.
055 *
056 * <p>Listening to a directory provides updates on the directory's immediate
057 * children, but does not listen recursively.
058 */
059public final class WatchServiceListener implements AutoCloseable {
060
061    @SuppressWarnings("rawtypes") // IntelliJ says it's unnecessary, but the compiler shows warnings
062    private static final WatchEvent.Kind<?>[] DEFAULT_WATCH_EVENTS = new WatchEvent.Kind[]{StandardWatchEventKinds.OVERFLOW,
063        StandardWatchEventKinds.ENTRY_CREATE, StandardWatchEventKinds.ENTRY_DELETE, StandardWatchEventKinds.ENTRY_MODIFY};
064    private static final int PARALLEL_THRESHOLD = 100;
065    private static final ThreadFactory DEFAULT_THREAD_FACTORY = new PrefixedNameThreadFactory("Configurate-WatchService", true);
066
067    private final WatchService watchService;
068    private volatile boolean open = true;
069    private final Thread executor;
070    final Executor taskExecutor;
071    private final ConcurrentHashMap<Path, DirectoryListenerRegistration> activeListeners = new ConcurrentHashMap<>();
072    private static final ThreadLocal<IOException> exceptionHolder = new ThreadLocal<>();
073
074    /**
075     * Returns a new builder for a WatchServiceListener to create a
076     * customized listener.
077     *
078     * @return a new builder
079     */
080    public static Builder builder() {
081        return new Builder();
082    }
083
084    /**
085     * Create a new {@link WatchServiceListener} using a new cached thread pool
086     * executor and the default filesystem.
087     *
088     * @return a new instance with default values
089     * @throws IOException if a watch service cannot be created
090     * @see #builder() for customization
091     */
092    public static WatchServiceListener create() throws IOException {
093        return new WatchServiceListener(DEFAULT_THREAD_FACTORY, FileSystems.getDefault(), ForkJoinPool.commonPool());
094    }
095
096    private WatchServiceListener(final ThreadFactory factory, final FileSystem fileSystem, final Executor taskExecutor) throws IOException {
097        this.watchService = fileSystem.newWatchService();
098        this.executor = factory.newThread(() -> {
099            while (this.open) {
100                final WatchKey key;
101                try {
102                    key = this.watchService.take();
103                } catch (final InterruptedException e) {
104                    this.open = false;
105                    Thread.currentThread().interrupt();
106                    break;
107                } catch (final ClosedWatchServiceException e) {
108                    break;
109                }
110                final Path watched = (Path) key.watchable();
111                final DirectoryListenerRegistration registration = this.activeListeners.get(watched);
112                if (registration != null) {
113                    final Set<Object> seenContexts = new HashSet<>();
114                    for (WatchEvent<?> event : key.pollEvents()) {
115                        if (!key.isValid()) {
116                            break;
117                        }
118
119                        if (!seenContexts.add(event.context())) {
120                            continue;
121                        }
122
123                        // Process listeners
124                        registration.submit(event);
125                        if (registration.closeIfEmpty()) {
126                            key.cancel();
127                            break;
128                        }
129                    }
130
131                    // If the watch key is no longer valid, send all listeners a close event
132                    if (!key.reset()) {
133                        final DirectoryListenerRegistration oldListeners = this.activeListeners.remove(watched);
134                        oldListeners.onClose();
135                    }
136                }
137                try {
138                    Thread.sleep(20);
139                } catch (final InterruptedException e) {
140                    Thread.currentThread().interrupt();
141                    break;
142                }
143            }
144        });
145        this.taskExecutor = taskExecutor;
146        this.executor.start();
147    }
148
149    /**
150     * Gets or creates a registration holder for a specific directory. This
151     * handles registering with the watch service if necessary.
152     *
153     * @param directory the directory to listen to
154     * @return a registration, created new if necessary.
155     * @throws ConfigurateException if produced while registering the path with
156     *          our WatchService
157     */
158    private DirectoryListenerRegistration registration(final Path directory) throws ConfigurateException {
159        final @Nullable DirectoryListenerRegistration reg = this.activeListeners.computeIfAbsent(directory, dir -> {
160            try {
161                return new DirectoryListenerRegistration(dir.register(this.watchService, DEFAULT_WATCH_EVENTS), this.taskExecutor);
162            } catch (final IOException ex) {
163                exceptionHolder.set(ex);
164                return null;
165            }
166        });
167
168        if (reg == null) {
169            throw new ConfigurateException("While adding listener for " + directory, exceptionHolder.get());
170        }
171        return reg;
172    }
173
174    /**
175     * Listen for changes to a specific file or directory.
176     *
177     * @param file the path of the file or directory to listen for changes on.
178     * @param callback a subscriber that will be notified when changes occur.
179     * @return a {@link Disposable} that can be used to cancel this subscription
180     * @throws ConfigurateException if a filesystem error occurs.
181     * @throws IllegalArgumentException if the provided path is a directory.
182     */
183    public Disposable listenToFile(Path file, final Subscriber<WatchEvent<?>> callback) throws ConfigurateException, IllegalArgumentException {
184        file = file.toAbsolutePath();
185        if (Files.isDirectory(file)) {
186            throw new IllegalArgumentException("Path " + file + " must be a file");
187        }
188
189        final Path fileName = file.getFileName();
190        return registration(file.getParent()).subscribe(fileName, callback);
191    }
192
193    /**
194     * Listen to a directory. Callbacks will receive events both for the
195     * directory and for its contents.
196     *
197     * @param directory the directory to listen to
198     * @param callback a subscriber that will be notified when changes occur.
199     * @return a {@link Disposable} that can be used to cancel this subscription
200     * @throws ConfigurateException when an error occurs registering with the
201     *                              underlying watch service.
202     * @throws IllegalArgumentException if the provided path is not a directory
203     */
204    public Disposable listenToDirectory(Path directory, final Subscriber<WatchEvent<?>> callback)
205            throws ConfigurateException, IllegalArgumentException {
206        directory = directory.toAbsolutePath();
207        if (!(Files.isDirectory(directory) || !Files.exists(directory))) {
208            throw new IllegalArgumentException("Path " + directory + " must be a directory");
209        }
210
211        return registration(directory).subscribe(callback);
212    }
213
214    /**
215     * Create a new {@link ConfigurationReference} subscribed to FS updates.
216     *
217     * @param loaderFunc function that will create a new loader
218     * @param path path to to for changes
219     * @param <N> node type
220     * @return new reference
221     * @throws ConfigurateException if unable to complete an initial load of
222     *      the configuration.
223     */
224    public <N extends ScopedConfigurationNode<N>> ConfigurationReference<N>
225        listenToConfiguration(final Function<Path, ConfigurationLoader<? extends N>> loaderFunc, final Path path) throws ConfigurateException {
226        return ConfigurationReference.watching(loaderFunc, path, this);
227    }
228
229    @Override
230    public void close() throws IOException {
231        this.open = false;
232        this.watchService.close();
233        this.activeListeners.forEachValue(PARALLEL_THRESHOLD, DirectoryListenerRegistration::onClose);
234        this.activeListeners.clear();
235        try {
236            this.executor.interrupt();
237            this.executor.join();
238        } catch (final InterruptedException e) {
239            throw new IOException("Failed to await termination of executor thread!");
240        }
241    }
242
243    /**
244     * Set the parameters needed to create a {@link WatchServiceListener}. All params are optional and defaults will be
245     * used if no values are specified.
246     */
247    public static final class Builder {
248
249        private @Nullable ThreadFactory threadFactory;
250        private @Nullable FileSystem fileSystem;
251        private @Nullable Executor taskExecutor;
252
253        private Builder() { }
254
255        /**
256         * Set the thread factory that will be used to create the polling thread
257         * for the returned watch service.
258         *
259         * @param factory the thread factory to use to create the deamon thread
260         * @return this builder
261         */
262        public Builder threadFactory(final ThreadFactory factory) {
263            this.threadFactory = requireNonNull(factory, "factory");
264            return this;
265        }
266
267        /**
268         * Set the executor that will be used to execute tasks queued based on
269         * received events. By default, the
270         * {@link ForkJoinPool#commonPool() common pool} is used.
271         *
272         * @param executor the executor to use
273         * @return this builder
274         */
275        public Builder taskExecutor(final Executor executor) {
276            this.taskExecutor = requireNonNull(executor, "executor");
277            return this;
278        }
279
280        /**
281         * Set the filesystem expected to be used for paths. A separate
282         * {@link WatchServiceListener} should be created to listen to events on
283         * each different file system.
284         *
285         * @param system the file system to use.
286         * @return this builder
287         */
288        public Builder fileSystem(final FileSystem system) {
289            this.fileSystem = system;
290            return this;
291        }
292
293        /**
294         * Create a new listener, using default values for any unset parameters.
295         *
296         * @return a newly created executor
297         * @throws IOException if thrown by {@link WatchServiceListener}'s constructor
298         */
299        public WatchServiceListener build() throws IOException {
300            if (this.threadFactory == null) {
301                this.threadFactory = DEFAULT_THREAD_FACTORY;
302            }
303
304            if (this.fileSystem == null) {
305                this.fileSystem = FileSystems.getDefault();
306            }
307
308            if (this.taskExecutor == null) {
309                this.taskExecutor = ForkJoinPool.commonPool();
310            }
311
312            return new WatchServiceListener(this.threadFactory, this.fileSystem, this.taskExecutor);
313        }
314
315    }
