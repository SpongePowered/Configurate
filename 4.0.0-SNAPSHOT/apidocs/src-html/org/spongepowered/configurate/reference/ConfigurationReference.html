030import java.nio.file.Path;

035/**
036 * An updating reference to a base configuration node.
037 *
038 * @param <N> the type of node to work with
039 */
040public interface ConfigurationReference<N extends ConfigurationNode> extends AutoCloseable {
041
042    /**
043     * Create a new configuration reference that will only update when loaded.
044     *
045     * @param loader the loader to load and save from
046     * @param <N> the type of node
047     * @return the newly created reference, with an initial load performed
048     * @throws ConfigurateException if the configuration contained fails to load
049     */
050    static <N extends ScopedConfigurationNode<N>> ConfigurationReference<N>
051            fixed(ConfigurationLoader<? extends N> loader) throws ConfigurateException {
052        final ConfigurationReference<N> ret = new ManualConfigurationReference<>(loader, ForkJoinPool.commonPool());
053        ret.load();
054        return ret;
055    }
056
057    /**
058     * Create a new configuration reference that will automatically update when
059     * triggered by the provided {@link WatchServiceListener}.
060     *
061     * @param loaderCreator a function that can create a {@link ConfigurationLoader}
062     * @param file the file to load this configuration from
063     * @param listener the watch service listener that will receive events
064     * @param <T> the node type
065     * @return the created reference
066     * @throws ConfigurateException if the underlying loader fails to load
067     *         a configuration
068     * @see WatchServiceListener#listenToConfiguration(Function, Path)
069     */
070    static <T extends ScopedConfigurationNode<T>> ConfigurationReference<T>
071            watching(Function<Path, ConfigurationLoader<? extends T>> loaderCreator, Path file, WatchServiceListener listener)
072            throws ConfigurateException {
073        final WatchingConfigurationReference<T> ret = new WatchingConfigurationReference<>(loaderCreator.apply(file), listener.taskExecutor);
074        ret.load();
075        ret.disposable(listener.listenToFile(file, ret));
076
077        return ret;
078    }
079
080    /**
081     * Reload a configuration using the provided loader.
082     *
083     * <p>If the load fails, this reference will continue pointing to old
084     * configuration values.
085     *
086     * @throws ConfigurateException when an error occurs
087     */
088    void load() throws ConfigurateException;
089
090    /**
091     * Save this configuration using the provided loader.
092     *
093     * @throws ConfigurateException when an error occurs in the underlying IO
094     */
095    void save() throws ConfigurateException;
096
097    /**
098     * Update the configuration node pointed to by this reference, and save it
099     * using the reference's loader.
100     *
101     * <p>Even if the loader fails to save this new node, the node pointed to by
102     * this reference will be updated.
103     *
104     * @param newNode the new node to save
105     * @throws ConfigurateException when an error occurs within the loader
106     */
107    void save(ConfigurationNode newNode) throws ConfigurateException;
108
109    /**
110     * Save this configuration using the provided loader. Any errors will be
111     * submitted to subscribers of the returned publisher.
112     *
113     * @return publisher providing an event when the save is complete
114     */
115    Publisher<N> saveAsync();
116
117    /**
118     * Update this configuration using the provided function, returning a
119     * {@link Publisher} which will complete with the result of the operation.
120     * The update function will be called asynchronously, and will be saved
121     * to this reference's loader when complete.
122     *
123     * @param updater update function
124     * @return publisher providing an event when the update is complete
125     */
126    Publisher<N> updateAsync(Function<N, ? extends N> updater);
127
128    /**
129     * Get the base node this reference refers to.
130     *
131     * @return the node
132     */
133    N node();
134
135    /**
136     * Get the loader this reference uses to load and save its node.
137     *
138     * @return the loader
139     */
140    ConfigurationLoader<? extends N> loader();
141
142    /**
143     * Get the node at the given path, relative to the root node.
144     *
145     * @param path the path, a series of path elements
146     * @return a child node
147     * @see ConfigurationNode#node(Object...)
148     */
149    N get(Object... path);
150
151    /**
152     * Get the node at the given path, relative to the root node.
153     *
154     * @param path the path, a series of path elements
155     * @return a child node
156     * @see ConfigurationNode#node(Iterable)
157     */
158    N get(Iterable<?> path);
159
160    /**
161     * Update the value of the node at the given path, using the root node as
162     * a base.
163     *
164     * @param path the path to get the child at
165     * @param value the value to set the child node to
166     */
167    default void set(Object[] path, @Nullable Object value) throws SerializationException {
168        node().node(path).set(value);
169    }
170
171    /**
172     * Set the value of the node at {@code path} to the given value.
173     *
174     * <p>This uses the appropriate
175     * {@link org.spongepowered.configurate.serialize.TypeSerializer} to
176     * serialize the data if it's not directly supported by the
177     * provided configuration.</p>
178     *
179     * @param path the path to set the value at
180     * @param type the type of data to serialize
181     * @param value the value to set
182     * @param <T> the type parameter for the value
183     * @throws IllegalArgumentException if a raw type is provided
184     * @throws SerializationException if thrown by the serialization mechanism
185     */
186    default <T> void set(Object[] path, Class<T> type, @Nullable T value) throws SerializationException {
187        node().node(path).set(type, value);
188    }
189
190    /**
191     * Set the value of the node at {@code path} to the given value.
192     *
193     * <p>This uses the appropriate
194     * {@link org.spongepowered.configurate.serialize.TypeSerializer} to
195     * serialize the data if it's not directly supported by the
196     * provided configuration.</p>
197     *
198     * @param path the path to set the value at
199     * @param type the type of data to serialize
200     * @param value the value to set
201     * @param <T> the type parameter for the value
202     * @throws SerializationException if thrown by the serialization mechanism
203     */
204    default <T> void set(Object[] path, TypeToken<T> type, @Nullable T value) throws SerializationException {
205        node().node(path).set(type, value);
206    }
207
208    /**
209     * Update the value of the node at the given path, using the root node as
210     * a base.
211     *
212     * @param path the path to get the child at
213     * @param value the value to set the child node to
214     */
215    default void set(NodePath path, @Nullable Object value) throws SerializationException {
216        node().node(path).set(value);
217    }
218
219    /**
220     * Set the value of the node at {@code path} to the given value.
221     *
222     * <p>This uses the appropriate
223     * {@link org.spongepowered.configurate.serialize.TypeSerializer} to
224     * serialize the data if it's not directly supported by the
225     * provided configuration.</p>
226     *
227     * @param path the path to set the value at
228     * @param type the type of data to serialize
229     * @param value the value to set
230     * @param <T> the type parameter for the value
231     * @throws SerializationException if thrown by the serialization mechanism
232     */
233    default <T> void set(NodePath path, Class<T> type, @Nullable T value) throws SerializationException {
234        node().node(path).set(type, value);
235    }
236
237    /**
238     * Set the value of the node at {@code path} to the given value.
239     *
240     * <p>This uses the appropriate
241     * {@link org.spongepowered.configurate.serialize.TypeSerializer} to
242     * serialize the data if it's not directly supported by the
243     * provided configuration.</p>
244     *
245     * @param path the path to set the value at
246     * @param type the type of data to serialize
247     * @param value the value to set
248     * @param <T> the type parameter for the value
249     * @throws SerializationException if thrown by the serialization mechanism
250     */
251    default <T> void set(NodePath path, TypeToken<T> type, @Nullable T value) throws SerializationException {
252        node().node(path).set(type, value);
253    }
254
255    /**
256     * Create a reference to the node at the provided path. The value will be
257     * deserialized according to the provided TypeToken.
258     *
259     * <p>The returned reference will update with reloads of and changes to the
260     * value of the provided configuration. Any serialization errors encountered
261     * will be submitted to the {@link #errors()} stream.
262     *
263     * @param type the value's type
264     * @param path the path from the root node to the node containing the value
265     * @param <T>  the value type
266     * @return a deserializing reference to the node at the given path
267     * @throws SerializationException if a type serializer could not be found
268     *         for the provided type
269     */
270    default <T> ValueReference<T, N> referenceTo(TypeToken<T> type, Object... path) throws SerializationException {
271        return referenceTo(type, NodePath.of(path));
272    }
273
274    /**
275     * Create a reference to the node at the provided path. The value will be
276     * deserialized according to type of the provided {@link Class}.
277     *
278     * <p>The returned reference will update with reloads of and changes to the
279     * value of the provided configuration. Any serialization errors encountered
280     * will be submitted to the {@link #errors()} stream.
281     *
282     * @param type the value's type
283     * @param path the path from the root node to the node containing the value
284     * @param <T>  the value type
285     * @return a deserializing reference to the node at the given path
286     * @throws SerializationException if a type serializer could not be found
287     *         for the provided type
288     */
289    default <T> ValueReference<T, N> referenceTo(Class<T> type, Object... path) throws SerializationException {
290        return referenceTo(type, NodePath.of(path));
291    }
292
293    /**
294     * Create a reference to the node at the provided path. The value will be
295     * deserialized according to the provided {@link TypeToken}.
296     *
297     * <p>The returned reference will update with reloads of and changes to the
298     * value of the provided configuration. Any serialization errors encountered
299     * will be submitted to the {@link #errors()} stream.
300     *
301     * @param type the value's type
302     * @param path the path from the root node to the node containing the value
303     * @param <T> the value type
304     * @return a deserializing reference to the node at the given path
305     * @throws SerializationException if a type serializer could not be found
306     *          for the provided type
307     */
308    default <T> ValueReference<T, N> referenceTo(TypeToken<T> type, NodePath path) throws SerializationException {
309        return referenceTo(type, path, null);
310    }
311
312    /**
313     * Create a reference to the node at the provided path. The value will be
314     * deserialized according to type of the provided {@link Class}.
315     *
316     * <p>The returned reference will update with reloads of and changes to the
317     * value of the provided configuration. Any serialization errors encountered
318     * will be submitted to the {@link #errors()} stream.
319     *
320     * @param type the value's type
321     * @param path the path from the root node to the node containing the value
322     * @param <T> the value type
323     * @return a deserializing reference to the node at the given path
324     * @throws SerializationException if a type serializer could not be found
325     *          for the provided type
326     */
327    default <T> ValueReference<T, N> referenceTo(Class<T> type, NodePath path) throws SerializationException {
328        return referenceTo(type, path, null);
329    }
330
331    /**
332     * Create a reference to the node at the provided path. The value will be
333     * deserialized according to the provided {@link TypeToken}.
334     *
335     * <p>The returned reference will update with reloads of and changes to the
336     * value of the provided configuration. Any serialization errors encountered
337     * will be submitted to the {@link #errors()} stream.
338     *
339     * @param type the value's type.
340     * @param path the path from the root node to the node containing the value
341     * @param defaultValue the value to use when there is no data present
342     *                     in the targeted node.
343     * @param <T> the value type
344     * @return a deserializing reference to the node at the given path
345     * @throws SerializationException if a type serializer could not be found
346     *         for the provided type
347     */
348    <T> ValueReference<T, N> referenceTo(TypeToken<T> type, NodePath path, @Nullable T defaultValue) throws SerializationException;
349
350    /**
351     * Create a reference to the node at the provided path. The value will be
352     * deserialized according to type of the provided {@link Class}.
353     *
354     * <p>The returned reference will update with reloads of and changes to the
355     * value of the provided configuration. Any serialization errors encountered
356     * will be submitted to the {@link #errors()} stream.
357     *
358     * @param type value's type
359     * @param path path from the root node to the node containing the value
360     * @param defaultValue value to use when there is no data present
361     *                     in the targeted node.
362     * @param <T> value type
363     * @return a deserializing reference to the node at the given path
364     * @throws SerializationException if a type serializer could not be found
365     *          for the provided type
366     */
367    <T> ValueReference<T, N> referenceTo(Class<T> type, NodePath path, @Nullable T defaultValue) throws SerializationException;
368
369    /**
370     * Access the {@link Publisher} that will broadcast update events, providing the newly created node. The returned
371     * publisher will be transaction-aware, i.e. any {@link TransactionalSubscriber} attached will progress through
372     * their phases appropriately
373     *
374     * @return the publisher
375     */
376    Publisher<N> updates();
377
378    /**
379     * A stream that will receive errors that occur while loading or saving to
380     * this reference.
381     *
382     * @return the publisher
383     */
384    Publisher<Map.Entry<ErrorPhase, Throwable>> errors();
385
386    /**
387     * {@inheritDoc}
388     */
389    @Override
390    void close();
391
392    /**
393     * Representing the phase where an error occurred.
394     */
395    enum ErrorPhase {
396        LOADING, SAVING, UNKNOWN, VALUE;
397
398    }
