017package org.spongepowered.configurate.util;
018
019import static java.util.Objects.requireNonNull;
020
021import java.util.Comparator;
022import java.util.LinkedHashMap;
023import java.util.Map;
024import java.util.concurrent.ConcurrentHashMap;
025import java.util.concurrent.ConcurrentMap;
026import java.util.concurrent.ConcurrentSkipListMap;
027
028/**
029 * Default implementations of {@link MapFactory}.
030 */
031public final class MapFactories {
032
033    private MapFactories() {}
034
035    /**
036     * Returns a {@link MapFactory} which creates maps without an order.
037     *
038     * @return a map factory which produces unordered maps
039     */
040    public static MapFactory unordered() {
041        return DefaultFactory.UNORDERED;
042    }
043
044    /**
045     * Returns a {@link MapFactory} which creates maps which are sorted using the given comparator.
046     *
047     * @param comparator the comparator used to sort the map keys
048     * @return a map factory which produces sorted maps
049     */
050    public static MapFactory sorted(final Comparator<Object> comparator) {
051        return new SortedMapFactory(requireNonNull(comparator, "comparator"));
052    }
053
054    /**
055     * Returns a {@link MapFactory} which creates maps which are naturally sorted.
056     *
057     * @return a map factory which produces naturally sorted maps
058     * @see Comparator#naturalOrder()
059     */
060    public static MapFactory sortedNatural() {
061        return DefaultFactory.SORTED_NATURAL;
062    }
063
064    /**
065     * Returns a {@link MapFactory} which creates maps which are sorted by insertion order.
066     *
067     * @return a map factory which produces maps sorted by insertion order
068     */
069    public static MapFactory insertionOrdered() {
070        return DefaultFactory.INSERTION_ORDERED;
071    }
072
073    private enum DefaultFactory implements MapFactory {
074        UNORDERED {
075            @Override
076            public <K, V> ConcurrentMap<K, V> create() {
077                return new ConcurrentHashMap<>();
078            }
079        },
080        SORTED_NATURAL {
081            @Override
082            public <K, V> ConcurrentMap<K, V> create() {
083                return new ConcurrentSkipListMap<>();
084            }
085        },
086        INSERTION_ORDERED {
087            @Override
088            public <K, V> Map<K, V> create() {
089                return new LinkedHashMap<>();
090            }
091        }
092    }
093
094    private static final class SortedMapFactory implements MapFactory {
095        private final Comparator<Object> comparator;
096
097        private SortedMapFactory(final Comparator<Object> comparator) {
098            this.comparator = comparator;
099        }
100
101        @Override
102        public <K, V> ConcurrentMap<K, V> create() {
103            return new ConcurrentSkipListMap<>(this.comparator);
104        }
105
106        @Override
107        public boolean equals(final Object obj) {
108            return obj instanceof SortedMapFactory && this.comparator.equals(((SortedMapFactory) obj).comparator);
109        }
110
111        @Override
112        public int hashCode() {
113            return this.comparator.hashCode();
114        }
115
116        @Override
117        public String toString() {
118            return "SortedMapFactory{comparator=" + this.comparator + '}';
119        }
120    }
121
122}
