017package org.spongepowered.configurate.util;
018
019import static java.util.Objects.requireNonNull;
020
021import org.checkerframework.checker.nullness.qual.Nullable;
022
023import java.util.Collections;
024import java.util.HashMap;
025import java.util.Map;
026import java.util.WeakHashMap;
027
028/**
029 * Utility class to cache more flexible enum lookup.
030 *
031 * <p>While normally case and punctuation have to match exactly, this method
032 * performs lookup that:</p>
033 *
034 * <ul>
035 *     <li>is case-insensitive</li>
036 *     <li>ignores underscores</li>
037 *     <li>caches mappings</li>
038 * </ul>
039 *
040 * <p>If the enum has two fields that are equal except for case and underscores,
041 * an exact match will return the appropriate value, and any fuzzy matches will
042 * map to the first value in the enum that is applicable.</p>
043 */
044public final class EnumLookup {
045
046    private EnumLookup() {
047
048    }
049
050    private static final Map<Class<? extends Enum<?>>, Map<String, Enum<?>>> ENUM_FIELD_CACHE = new WeakHashMap<>();
051
052    private static String processKey(final String key) {
053        // stick a flower at the front so processed keys are different from literal keys
054        return "ðŸŒ¸" + key.toLowerCase().replace("_", "");
055    }
056
057    /**
058     * Perform a fuzzy lookup of {@code key} in enum {@code clazz}.
059     *
060     * @param clazz class to find key in
061     * @param key key to find
062     * @param <T> type of value
063     * @return optionally the enum
064     */
065    @SuppressWarnings("unchecked")
066    public static <T extends Enum<T>> @Nullable T lookupEnum(final Class<T> clazz, final String key) {
067        final Map<String, Enum<?>> vals = ENUM_FIELD_CACHE.computeIfAbsent(requireNonNull(clazz), c2 -> {
068            final Map<String, Enum<?>> ret = new HashMap<>();
069            for (Enum<?> field : c2.getEnumConstants()) {
070                ret.put(field.name(), field);
071                ret.putIfAbsent(processKey(field.name()), field);
072            }
073            return Collections.unmodifiableMap(ret);
074        });
075
076
077        final Enum<?> possibleRet = vals.get(requireNonNull(key, "key"));
078        if (possibleRet != null) {
079            return (T) possibleRet;
080        }
081        return (T) vals.get(processKey(key));
082    }
083
084}
