017package org.spongepowered.configurate.util;
018
019import com.google.common.cache.CacheBuilder;
020import com.google.common.cache.CacheLoader;
021import com.google.common.cache.LoadingCache;
022import com.google.common.collect.ImmutableMap;
023
024import java.util.HashMap;
025import java.util.Map;
026import java.util.Optional;
027import java.util.concurrent.ExecutionException;
028
029import static java.util.Objects.requireNonNull;
030
031/**
032 * Utility class to cache more flexible enum lookup.
033 *
034 * <p>While normally case and punctuation have to match exactly, this method performs lookup that:</p>
035 *
036 * <ul>
037 *     <li>is case-insensitive</li>
038 *     <li>ignores underscores</li>
039 *     <li>caches mappings</li>
040 * </ul>
041 *
042 * <p>If the enum has two fields that are equal except for case and underscores, an exact match
043 * will return the appropriate value, and any fuzzy matches will map to the first value in the enum
044 * that is applicable.</p>
045 */
046public final class EnumLookup {
047    private EnumLookup() {}
048
049    private static final LoadingCache<Class<? extends Enum<?>>, Map<String, Enum<?>>> ENUM_FIELD_CACHE = CacheBuilder
050            .newBuilder()
051            .weakKeys()
052            .maximumSize(512)
053            .build(new CacheLoader<Class<? extends Enum<?>>, Map<String, Enum<?>>>() {
054                @Override
055                public Map<String, Enum<?>> load(Class<? extends Enum<?>> key) {
056                    Map<String, Enum<?>> ret = new HashMap<>();
057                    for (Enum<?> field : key.getEnumConstants()) {
058                        ret.put(field.name(), field);
059                        ret.putIfAbsent(processKey(field.name()), field);
060                    }
061                    return ImmutableMap.copyOf(ret);
062                }
063            });
064
065    private static String processKey(String key) {
066        // stick a flower at the front so processed keys are different from literal keys
067        return "ðŸŒ¸" + key.toLowerCase().replace("_", "");
068    }
069
070    @SuppressWarnings("unchecked")
071    public static <T extends Enum<T>> Optional<T> lookupEnum(Class<T> clazz, String key) {
072        try {
073            Map<String, Enum<?>> vals = ENUM_FIELD_CACHE.get(requireNonNull(clazz, "clazz"));
074            Enum<?> possibleRet = vals.get(requireNonNull(key, "key"));
075            if (possibleRet != null) {
076                return Optional.of((T) possibleRet);
077            }
078            return Optional.ofNullable((T) vals.get(processKey(key)));
079        } catch (ExecutionException e) {
080            return Optional.empty();
081        }
082    }
083}
