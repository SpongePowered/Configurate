017package org.spongepowered.configurate.util;
018
019import static java.util.Objects.requireNonNull;
020
021import org.checkerframework.checker.nullness.qual.NonNull;
022
023import java.util.function.Function;
024
025/**
026 * A function with one input and one output which may throw a checked exception.
027 *
028 * @param <I> the input parameter type
029 * @param <O> the output parameter type
030 * @param <E> the type thrown
031 */
032@FunctionalInterface
033public interface CheckedFunction<I, O, E extends Exception> {
034
035    /**
036     * Perform the action.
037     *
038     * @param one parameter
039     * @return return value
040     * @throws E thrown when defined by types accepting this function
041     */
042    O apply(I one) throws E;
043
044    /**
045     * Convert a JDK {@link Function} into its checked variant.
046     *
047     * @param func the function
048     * @param <I> parameter type
049     * @param <O> return type
050     * @return the function as a checked function
051     */
052    static <I, O> CheckedFunction<I, O, RuntimeException> from(Function<I, @NonNull O> func) {
053        return requireNonNull(func, "func")::apply;
054    }
055
056}
