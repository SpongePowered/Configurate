017package org.spongepowered.configurate.util;
018
019import static java.util.Objects.requireNonNull;
020
021/**
022 * Extra string utilities.
023 */
024public final class Strings {
025
026    private Strings() {}
027
028    /**
029     * Create a new string with the contents of the provided string repeated
030     * {@code times} times.
031     *
032     * <p>Available in {@link String} itself as of JDK 11.
033     *
034     * @param content text to repeat
035     * @param times amount to repeat
036     * @return repeated string
037     */
038    public static String repeat(final String content, final int times) {
039        requireNonNull(content, "content");
040
041        switch (times) {
042            case 0: return "";
043            case 1: return content;
044            default:
045                final StringBuilder ret = new StringBuilder(content.length() * times);
046                for (int i = 0; i < times; ++i) {
047                    ret.append(content);
048                }
049                return ret.toString();
050        }
051    }
052
053}
