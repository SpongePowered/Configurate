017package org.spongepowered.configurate.util;
018
019import java.util.regex.Matcher;
020import java.util.regex.Pattern;
021
022/**
023 * Standard naming schemes.
024 */
025public enum NamingSchemes implements NamingScheme {
026
027    /**
028     * Passes through names unchanged.
029     */
030    PASSTHROUGH {
031        @Override
032        public String coerce(final String input) {
033            return input;
034        }
035    },
036
037    /**
038     * Reformats names to {@code camelCase} style.
039     */
040    CAMEL_CASE {
041        @Override
042        @SuppressWarnings("JdkObsolete") // StringBuilder methods not present until JDK 9
043        public String coerce(final String input) {
044            final Matcher match = DASH_UNDERSCORE.matcher(input);
045            if (!match.find()) {
046                return input;
047            }
048            final StringBuffer ret = new StringBuffer(input.length());
049            do {
050                match.appendReplacement(ret, "");
051                ret.appendCodePoint(input.codePointAt(match.start()));
052                ret.appendCodePoint(Character.toUpperCase(input.codePointBefore(match.end())));
053            } while (match.find());
054            match.appendTail(ret);
055            return ret.toString();
056        }
057    },
058
059    /**
060     * Reformats names to {@code snake_case} format.
061     */
062    SNAKE_CASE {
063        @Override
064        public String coerce(final String input) {
065            return NamingSchemes.enforceLowerCaseSeparatorChar(input, UNDERSCORE, DASH);
066        }
067    },
068
069    /**
070     * Reformats names to {@code lower-case-dashed} format.
071     */
072    LOWER_CASE_DASHED {
073        @Override
074        public String coerce(final String input) {
075            return NamingSchemes.enforceLowerCaseSeparatorChar(input, DASH, UNDERSCORE);
076        }
077    };
078
079    private static final Pattern DASH_UNDERSCORE = Pattern.compile(".[-_].");
080    private static final char UNDERSCORE = '_';
081    private static final char DASH = '-';
082
083    // Common logic for snake_case and dash-separated
084    private static String enforceLowerCaseSeparatorChar(final String input, final char preferredDelimiter, final char convertDelimiter) {
085        final StringBuilder build = new StringBuilder(input);
086        for (int i = 0; i < build.length(); i++) {
087            final int ch = build.codePointAt(i);
088            if (ch == convertDelimiter) {
089                if (i != 0 && i != build.length() - 1) { // only convert actual separators
090                    build.setCharAt(i, preferredDelimiter);
091                }
092            } else if (Character.isUpperCase(ch)) {
093                build.insert(i++, preferredDelimiter);
094                final int lower = Character.toLowerCase(ch);
095                if (Character.isBmpCodePoint(lower)) {
096                    build.setCharAt(i, (char) lower);
097                } else {
098                    build.setCharAt(i++, Character.highSurrogate(lower));
099                    build.setCharAt(i, Character.lowSurrogate(lower));
100                }
101            }
102        }
103        return build.toString();
104
105    }
106
107}
