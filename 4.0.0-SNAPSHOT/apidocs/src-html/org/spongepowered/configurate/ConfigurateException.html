017package org.spongepowered.configurate;
018
019import org.checkerframework.checker.nullness.qual.Nullable;
020import org.spongepowered.configurate.transformation.NodePath;
021
022import java.util.function.Supplier;
023
024/**
025 * Any sort of error thrown within Configurate.
026 *
027 * <p>Configurate's errors are designed to provide a view of as
028 * many errors as possible within one configuration tree, though the
029 * {@link Throwable#getSuppressed() suppressed exceptions}</p>
030 *
031 * @since 4.0.0
032 */
033public class ConfigurateException extends Exception {
034
035    private static final long serialVersionUID = 1635526451813128733L;
036
037    private @Nullable Supplier<NodePath> path;
038
039    /**
040     * Create a new unknown exception.
041     *
042     * @since 4.0.0
043     */
044    public ConfigurateException() {
045    }
046
047    /**
048     * Create a new exception at unknown path with provided
049     * informational message.
050     *
051     * @param message informational message
052     * @since 4.0.0
053     */
054    public ConfigurateException(final String message) {
055        super(message);
056    }
057
058    /**
059     * Create a new exception with a cause and unknown message.
060     *
061     * @param cause the cause of this exception
062     * @since 4.0.0
063     */
064    public ConfigurateException(final Throwable cause) {
065        super(cause);
066    }
067
068    /**
069     * Create a new exception with informational message and cause.
070     *
071     * @param message the informational message
072     * @param cause the cause of the exception
073     * @since 4.0.0
074     */
075    public ConfigurateException(final @Nullable String message, final @Nullable Throwable cause) {
076        super(message, cause);
077    }
078
079    /**
080     * Create a new exception pre-initialized with path and message.
081     *
082     * @param pos node where the error occurred
083     * @param message message describing the error
084     * @since 4.0.0
085     */
086    public ConfigurateException(final ConfigurationNode pos, final String message) {
087        super(message);
088        this.path = pos::path;
089    }
090
091    /**
092     * Create a new exception pre-initialized with path and cause.
093     *
094     * @param pos node where the error occurred
095     * @param cause direct cause of this exception
096     * @since 4.0.0
097     */
098    public ConfigurateException(final ConfigurationNode pos, final Throwable cause) {
099        super(cause);
100        this.path = pos::path;
101    }
102
103    /**
104     * Create a new exception pre-initialized with path, message, and cause.
105     *
106     * @param pos node where the error occurred
107     * @param message message describing the error
108     * @param cause direct cause of this exception
109     * @since 4.0.0
110     */
111    public ConfigurateException(final ConfigurationNode pos, final @Nullable String message, final @Nullable Throwable cause) {
112        super(message, cause);
113        this.path = pos::path;
114    }
115
116    /**
117     * Create a new exception pre-initialized with path, message, and cause.
118     *
119     * @param path path to the node where the error occurred
120     * @param message message describing the error
121     * @param cause direct cause of this exception
122     * @since 4.0.0
123     */
124    public ConfigurateException(final NodePath path, final @Nullable String message, final @Nullable Throwable cause) {
125        super(message, cause);
126        this.path = () -> path;
127    }
128
129    /**
130     * Get the path associated with this failure.
131     *
132     * @return the path
133     * @since 4.0.0
134     */
135    public NodePath path() {
136        final @Nullable Supplier<NodePath> path = this.path;
137        return path == null ? NodePath.path() : path.get();
138    }
139
140    /**
141     * Initialize path if none has been set.
142     *
143     * @param path new path
144     * @since 4.0.0
145     */
146    public void initPath(final Supplier<NodePath> path) {
147        if (this.path == null) {
148            this.path = path;
149        }
150    }
151
152    /**
153     * Get the exception's message without any extra formatting.
154     *
155     * @return the raw message
156     * @since 4.0.0
157     */
158    public @Nullable String rawMessage() {
159        return super.getMessage();
160    }
161
162    /**
163     * Get a description of the location of this error, with path included.
164     *
165     * @return message
166     * @since 4.0.0
167     */
168    @Override
169    public @Nullable String getMessage() {
170        return this.path().toString() + ": " + super.getMessage();
171    }
172
173}
