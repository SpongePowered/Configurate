017package org.spongepowered.configurate;
018
019import org.checkerframework.checker.nullness.qual.Nullable;
020import org.spongepowered.configurate.transformation.NodePath;
021
022import java.util.function.Supplier;
023
024/**
025 * Any sort of error thrown within Configurate.
026 *
027 * <p>Configurate's errors are designed to provide a view of as
028 * many errors as possible within one configuration tree, though the
029 * {@link Throwable#getSuppressed() suppressed exceptions}</p>
030 */
031public class ConfigurateException extends Exception {
032
033    private static final long serialVersionUID = 1635526451813128733L;
034
035    private @Nullable Supplier<NodePath> path;
036
037    /**
038     * Create a new unknown exception.
039     */
040    public ConfigurateException() {
041        this.path = null;
042    }
043
044    /**
045     * Create a new exception at unknown path with provided
046     * informational message.
047     *
048     * @param message informational message
049     */
050    public ConfigurateException(final String message) {
051        super(message);
052        this.path = null;
053    }
054
055    /**
056     * Create a new exception with a cause and unknown message.
057     *
058     * @param cause the cause of this exception
059     */
060    public ConfigurateException(final Throwable cause) {
061        super(cause);
062        this.path = null;
063    }
064
065    /**
066     * Create a new exception with informational message and cause.
067     *
068     * @param message the informational message
069     * @param cause the cause of the exception
070     */
071    public ConfigurateException(final @Nullable String message, final @Nullable Throwable cause) {
072        super(message, cause);
073        this.path = null;
074    }
075
076    /**
077     * Create a new exception pre-initialized with path and message.
078     *
079     * @param pos node where the error occurred
080     * @param message message describing the error
081     */
082    public ConfigurateException(final ConfigurationNode pos, final String message) {
083        super(message);
084        this.path = pos::path;
085    }
086
087    /**
088     * Create a new exception pre-initialized with path and cause.
089     *
090     * @param pos node where the error occurred
091     * @param cause direct cause of this exception
092     */
093    public ConfigurateException(final ConfigurationNode pos, final Throwable cause) {
094        super(cause);
095        this.path = pos::path;
096    }
097
098    /**
099     * Create a new exception pre-initialized with path, message, and cause.
100     *
101     * @param pos node where the error occurred
102     * @param message message describing the error
103     * @param cause direct cause of this exception
104     */
105    public ConfigurateException(final ConfigurationNode pos, final @Nullable String message, final @Nullable Throwable cause) {
106        super(message, cause);
107        this.path = pos::path;
108    }
109
110    /**
111     * Create a new exception pre-initialized with path, message, and cause.
112     *
113     * @param path path to the node where the error occurred
114     * @param message message describing the error
115     * @param cause direct cause of this exception
116     */
117    public ConfigurateException(final NodePath path, final @Nullable String message, final @Nullable Throwable cause) {
118        super(message, cause);
119        this.path = () -> path;
120    }
121
122    /**
123     * Get the path associated with this failure.
124     *
125     * @return the path
126     */
127    public NodePath path() {
128        final @Nullable Supplier<NodePath> path = this.path;
129        return path == null ? NodePath.path() : path.get();
130    }
131
132    /**
133     * Initialize path if none has been set.
134     *
135     * @param path new path
136     */
137    public void initPath(final Supplier<NodePath> path) {
138        if (this.path == null) {
139            this.path = path;
140        }
141    }
142
143    /**
144     * Get the exception's message without any extra formatting.
145     *
146     * @return the raw message
147     */
148    public @Nullable String rawMessage() {
149        return super.getMessage();
150    }
151
152    /**
153     * Get a description of the location of this error, with path included.
154     *
155     * @return message
156     */
157    @Override
158    public @Nullable String getMessage() {
159        return this.path().toString() + ": " + super.getMessage();
160    }
161
162}
