017package org.spongepowered.configurate;
018
019import com.google.auto.value.AutoValue;
020import io.leangen.geantyref.TypeToken;
021import org.checkerframework.checker.nullness.qual.Nullable;
022
023/**
024 * A flag for configuration loaders describing how a node should be serialized.
025 *
026 * <p>A loader may not accept every representation hint available, but any
027 * understood hints should be exposed as constant fields on the loader class.
028 * Any unknown hints will be ignored.
029 *
030 * @param <V> the value type
031 */
032@AutoValue
033public abstract class RepresentationHint<V> {
034
035    /**
036     * Create a new basic representation hint.
037     *
038     * <p>The created hint will be inheritable and have no default
039     * value set.</p>
040     *
041     * @param identifier hint identifier
042     * @param valueType type of value the hint will hold
043     * @param <V> value type
044     * @return a new hint
045     */
046    public static <V> RepresentationHint<V> of(final String identifier, final Class<V> valueType) {
047        return RepresentationHint.<V>builder().identifier(identifier).valueType(valueType).build();
048    }
049
050    /**
051     * Create a new basic representation hint.
052     *
053     * <p>The created hint will be inheritable and have no default
054     * value set.</p>
055     *
056     * @param identifier hint identifier
057     * @param valueType type of value the hint will hold
058     * @param <V> value type
059     * @return a new hint
060     */
061    public static <V> RepresentationHint<V> of(final String identifier, final TypeToken<V> valueType) {
062        return RepresentationHint.<V>builder().identifier(identifier).valueType(valueType).build();
063    }
064
065    /**
066     * Create a builder for a new hint.
067     *
068     * @param <V> value type
069     * @return a new builder
070     */
071    public static <V> Builder<V> builder() {
072        return new AutoValue_RepresentationHint.Builder<>();
073    }
074
075    RepresentationHint() { }
076
077    /**
078     * An identifier used to represent this hint in error messages.
079     *
080     * @return the identifier
081     */
082    public abstract String identifier();
083
084    /**
085     * The type that values of this type have to have.
086     *
087     * @return value type
088     */
089    public abstract TypeToken<V> valueType();
090
091    /**
092     * If a value for a representation hint cannot be found by quering a node
093     * or any of this parents, the default value will be returned.
094     *
095     * @return default type
096     */
097    public abstract @Nullable V defaultValue();
098
099    /**
100     * Get whether or not this hint can draw its value from parent nodes.
101     *
102     * @return if inheritable
103     */
104    public abstract boolean inheritable();
105
106    /**
107     * A builder for {@link RepresentationHint}s.
108     *
109     * @param <V> value type
110     */
111    @AutoValue.Builder
112    public abstract static class Builder<V> {
113
114        Builder() {
115            this.inheritable(true);
116        }
117
118        /**
119         * Set the identifier to refer to this hint.
120         *
121         * @param identifier hint identifier
122         * @return this builder
123         */
124        public abstract Builder<V> identifier(String identifier);
125
126        /**
127         * Set the type used for this node's value.
128         *
129         * <p>Raw types are forbidden.</p>
130         *
131         * @param valueType the value type
132         * @return this builder
133         */
134        public final Builder<V> valueType(final Class<V> valueType) {
135            return valueType(TypeToken.get(valueType));
136        }
137
138        /**
139         * Set the type used for this node's value.
140         *
141         * <p>Raw types are forbidden.</p>
142         *
143         * @param valueType the value type
144         * @return this builder
145         */
146        public abstract Builder<V> valueType(TypeToken<V> valueType);
147
148        /**
149         * Set the default value when this hint is not present in the hierarchy.
150         *
151         * <p>This defaults to {@code null}.</p>
152         *
153         * @param defaultValue value to return on gets
154         * @return this builder
155         */
156        public abstract Builder<V> defaultValue(@Nullable V defaultValue);
157
158        /**
159         * Set whether or not the hint can be inherited.
160         *
161         * <p>Defaults to {@code true}.</p>
162         *
163         * @param inheritable if inheritable
164         * @return this builder
165         * @see #inheritable()
166         */
167        public abstract Builder<V> inheritable(boolean inheritable);
168
169        /**
170         * Create a new hint from the provided options.
171         *
172         * <p>The {@code identifier} and {@code valueType} must have been set to
173         * build a complete hint.</p>
174         *
175         * @return a new representation hint
176         */
177        public abstract RepresentationHint<V> build();
178
179    }
180
181}
