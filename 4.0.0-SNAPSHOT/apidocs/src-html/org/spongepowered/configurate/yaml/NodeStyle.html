017package org.spongepowered.configurate.yaml;
018
019import org.checkerframework.checker.nullness.qual.Nullable;
020import org.yaml.snakeyaml.DumperOptions;
021
022/**
023 * Representation of collections and mappings in a YAML document.
024 */
025public enum NodeStyle {
026
027    /**
028     * Always use <a href="https://yaml.org/spec/1.1/#id903421">block style</a>.
029     */
030    BLOCK(DumperOptions.FlowStyle.BLOCK),
031
032    /**
033     * Always use <a href="https://yaml.org/spec/1.1/#id902924">flow style</a>.
034     */
035    FLOW(DumperOptions.FlowStyle.FLOW)
036    ;
037
038    private final DumperOptions.FlowStyle snake;
039
040    NodeStyle(final DumperOptions.FlowStyle snake) {
041        this.snake = snake;
042    }
043
044    static DumperOptions.FlowStyle asSnakeYaml(final @Nullable NodeStyle style) {
045        return style == null ? DumperOptions.FlowStyle.AUTO : style.snake;
046    }
047
048}
