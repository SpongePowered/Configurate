017package org.spongepowered.configurate.yaml;
018
019import com.google.common.collect.ImmutableSet;
020import org.checkerframework.checker.nullness.qual.NonNull;
021import org.spongepowered.configurate.BasicConfigurationNode;
022import org.spongepowered.configurate.ConfigurationNode;
023import org.spongepowered.configurate.ConfigurationOptions;
024import org.spongepowered.configurate.loader.AbstractConfigurationLoader;
025import org.spongepowered.configurate.loader.CommentHandler;
026import org.spongepowered.configurate.loader.CommentHandlers;
027import org.yaml.snakeyaml.DumperOptions;
028import org.yaml.snakeyaml.DumperOptions.FlowStyle;
029import org.yaml.snakeyaml.Yaml;
030
031import java.io.BufferedReader;
032import java.io.Writer;
033import java.math.BigInteger;
034import java.sql.Timestamp;
035import java.util.Date;
036import java.util.List;
037import java.util.Map;
038import java.util.Set;
039
040/**
041 * A loader for YAML-formatted configurations, using the SnakeYAML library for parsing and generation.
042 *
043 */
044public class YamlConfigurationLoader extends AbstractConfigurationLoader<BasicConfigurationNode> {
045
046    /**
047     * Creates a new {@link YamlConfigurationLoader} builder.
048     *
049     * @return A new builder
050     */
051    @NonNull
052    public static Builder builder() {
053        return new Builder();
054    }
055
056    /**
057     * Builds a {@link YamlConfigurationLoader}.
058     */
059    public static class Builder extends AbstractConfigurationLoader.Builder<Builder> {
060        private final DumperOptions options = new DumperOptions();
061
062        protected Builder() {
063            setIndent(4);
064            // From the YAML 1.1 Global tags
065            // https://yaml.org/type/
066            // using SnakeYaml representation: https://bitbucket.org/asomov/snakeyaml/wiki/Documentation#markdown-header-yaml-tags-and-java-types
067            setDefaultOptions(getDefaultOptions().withAcceptedTypes(ImmutableSet.of(Boolean.class, Integer.class, Long.class, BigInteger.class, Double.class,
068                    byte[].class, String.class, Date.class, java.sql.Date.class, Timestamp.class,
069                    Set.class, List.class, Map.class)));
070        }
071
072        /**
073         * Sets the level of indentation the resultant loader should use.
074         *
075         * @param indent The indent level
076         * @return This builder (for chaining)
077         */
078        @NonNull
079        public Builder setIndent(int indent) {
080            options.setIndent(indent);
081            return this;
082        }
083
084        /**
085         * Gets the level of indentation to be used by the resultant loader.
086         *
087         * @return The indent level
088         */
089        public int getIndent() {
090            return options.getIndent();
091        }
092
093        /**
094         * Sets the flow style the resultant loader should use.
095         *
096         * Flow: the compact, json-like representation.<br>
097         * Example: <code>
098         *     {value: [list, of, elements], another: value}
099         * </code>
100         *
101         * Block: expanded, traditional YAML<br>
102         * Example: <code>
103         *     value:
104         *     - list
105         *     - of
106         *     - elements
107         *     another: value
108         * </code>
109         *
110         * @param style The flow style to use
111         * @return This builder (for chaining)
112         */
113        @NonNull
114        public Builder setFlowStyle(@NonNull FlowStyle style) {
115            options.setDefaultFlowStyle(style);
116            return this;
117        }
118
119        /**
120         * Gets the flow style to be used by the resultant loader.
121         *
122         * @return The flow style
123         */
124        @NonNull
125        public FlowStyle getFlowSyle() {
126            return options.getDefaultFlowStyle();
127        }
128
129        @NonNull
130        @Override
131        public YamlConfigurationLoader build() {
132            return new YamlConfigurationLoader(this);
133        }
134    }
135
136    private final ThreadLocal<Yaml> yaml;
137
138    private YamlConfigurationLoader(Builder builder) {
139        super(builder, new CommentHandler[] {CommentHandlers.HASH});
140        final DumperOptions opts = builder.options;
141        this.yaml = ThreadLocal.withInitial(() -> new Yaml(opts));
142    }
143
144    @Override
145    protected void loadInternal(BasicConfigurationNode node, BufferedReader reader) {
146        node.setValue(yaml.get().load(reader));
147    }
148
149    @Override
150    protected void saveInternal(ConfigurationNode node, Writer writer) {
151        yaml.get().dump(node.getValue(), writer);
152    }
153
154    @NonNull
155    @Override
156    public BasicConfigurationNode createEmptyNode(@NonNull ConfigurationOptions options) {
157        return BasicConfigurationNode.root(options);
158    }
159}
