017package org.spongepowered.configurate;
018
019import io.leangen.geantyref.TypeToken;
020import org.checkerframework.checker.nullness.qual.Nullable;
021import org.spongepowered.configurate.serialize.SerializationException;
022import org.spongepowered.configurate.serialize.TypeSerializer;
023import org.spongepowered.configurate.util.CheckedConsumer;
024
025import java.lang.reflect.Type;
026import java.util.List;
027import java.util.Map;
028import java.util.stream.Collector;
029
030/**
031 * Intermediate node type to reduce need for casting.
032 *
033 * <p>Any methods that return {@link ConfigurationNode} in
034 * {@link ConfigurationNode} should be overridden to return the {@link N}
035 * self-type instead.</p>
036 *
037 * @param <N> self type
038 */
039public interface ScopedConfigurationNode<N extends ScopedConfigurationNode<N>> extends ConfigurationNode {
040
041    /**
042     * Get a correctly typed instance of this node.
043     *
044     * @return the node type
045     */
046    N self();
047
048    /**
049     * {@inheritDoc}
050     */
051    @Override
052    N appendListNode();
053
054    /**
055     * {@inheritDoc}
056     */
057    @Override
058    N copy();
059
060    /**
061     * {@inheritDoc}
062     */
063    @Override
064    N node(Object... path);
065
066    /**
067     * {@inheritDoc}
068     */
069    @Override
070    N node(Iterable<?> path);
071
072    /**
073     * {@inheritDoc}
074     */
075    @Override
076    @Nullable N parent();
077
078    /**
079     * {@inheritDoc}
080     */
081    @Override
082    N from(ConfigurationNode other);
083
084    /**
085     * {@inheritDoc}
086     */
087    @Override
088    N mergeFrom(ConfigurationNode other);
089
090    /**
091     * {@inheritDoc}
092     */
093    @Override
094    N set(@Nullable Object value) throws SerializationException;
095
096    /**
097     * {@inheritDoc}
098     */
099    @Override
100    @SuppressWarnings({"unchecked", "rawtypes"}) // for TypeSerializer.serialize
101    default N set(Type type, @Nullable Object value) throws SerializationException {
102        if (value == null) {
103            return set(null);
104        }
105
106        final @Nullable TypeSerializer<?> serial = options().serializers().get(type);
107        if (serial != null) {
108            ((TypeSerializer) serial).serialize(type, value, self());
109        } else if (options().acceptsType(value.getClass())) {
110            raw(value); // Just write if no applicable serializer exists?
111        } else {
112            throw new SerializationException("No serializer available for type " + type);
113        }
114        return self();
115    }
116
117    @Override
118    default <V> N set(Class<V> type, @Nullable V value) throws SerializationException {
119        return set((Type) type, value);
120    }
121
122    @Override
123    default <V> N set(TypeToken<V> type, @Nullable V value) throws SerializationException {
124        return set(type.getType(), value);
125    }
126
127    /**
128     * {@inheritDoc}
129     */
130    @Override
131    N raw(@Nullable Object value);
132
133    /**
134     * {@inheritDoc}
135     */
136    @Override
137    List<N> childrenList();
138
139    /**
140     * {@inheritDoc}
141     */
142    @Override
143    Map<Object, N> childrenMap();
144
145    /**
146     * {@inheritDoc}
147     */
148    @SuppressWarnings({"unchecked", "rawtypes"})
149    @Override
150    default <V> Collector<Map.Entry<?, V>, N, N> toMapCollector(final TypeToken<V> valueType) {
151        return (Collector) ConfigurationNode.super.toMapCollector(valueType);
152    }
153
154    /**
155     * {@inheritDoc}
156     */
157    @SuppressWarnings({"unchecked", "rawtypes"})
158    @Override
159    default <V> Collector<Map.Entry<?, V>, N, N> toMapCollector(final Class<V> valueType) {
160        return (Collector) ConfigurationNode.super.toMapCollector(valueType);
161    }
162
163    /**
164     * {@inheritDoc}
165     */
166    @SuppressWarnings({"unchecked", "rawtypes"})
167    @Override
168    default <V> Collector<V, N, N> toListCollector(final TypeToken<V> valueType) {
169        return (Collector) ConfigurationNode.super.toListCollector(valueType);
170    }
171
172    /**
173     * {@inheritDoc}
174     */
175    @SuppressWarnings({"unchecked", "rawtypes"})
176    @Override
177    default <V> Collector<V, N, N> toListCollector(final Class<V> valueType) {
178        return (Collector) ConfigurationNode.super.toListCollector(valueType);
179    }
180
181    /**
182     * Execute an action on this node. This allows performing multiple
183     * operations on a single node without having to clutter up the surrounding
184     * scope.
185     *
186     * @param <E> thrown type
187     * @param action the action to perform on this node
188     * @return this node
189     */
190    default <E extends Exception> N act(CheckedConsumer<? super N, E> action) throws E {
191        action.accept(self());
192        return self();
193    }
194
195    @Override
196    <V> N hint(RepresentationHint<V> hint, @Nullable V value);
197
198}
