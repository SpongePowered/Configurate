017package org.spongepowered.configurate;
018
019import com.google.common.reflect.TypeToken;
020import org.checkerframework.checker.nullness.qual.NonNull;
021import org.checkerframework.checker.nullness.qual.Nullable;
022import org.spongepowered.configurate.objectmapping.ObjectMappingException;
023import org.spongepowered.configurate.serialize.TypeSerializer;
024
025import java.util.List;
026import java.util.Map;
027import java.util.function.Consumer;
028
029public interface ScopedConfigurationNode<N extends ScopedConfigurationNode<N>> extends ConfigurationNode {
030
031    /**
032     * Get a correctly typed instance of this node
033     * @return The node type
034     */
035    N self();
036
037    /**
038     * {@inheritDoc}
039     */
040    @Override
041    @NonNull N appendListNode();
042
043    /**
044     * {@inheritDoc}
045     */
046    @Override
047    @NonNull N copy();
048
049    /**
050     * {@inheritDoc}
051     */
052    @Override
053    @NonNull N getNode(@NonNull Object... path);
054
055    /**
056     * {@inheritDoc}
057     */
058    @Override
059    @Nullable N getParent();
060
061    /**
062     * {@inheritDoc}
063     */
064    @Override
065    @NonNull N mergeValuesFrom(@NonNull ConfigurationNode other);
066
067    /**
068     * {@inheritDoc}
069     */
070    @Override
071    @NonNull N setValue(@Nullable Object value);
072
073    /**
074     * {@inheritDoc}
075     */
076    @Override
077    @NonNull
078    default <V> N setValue(@NonNull TypeToken<V> type, @Nullable V value) throws ObjectMappingException {
079        if (value == null) {
080            return setValue(null);
081        }
082
083        TypeSerializer<V> serial = getOptions().getSerializers().get(type);
084        if (serial != null) {
085            serial.serialize(type, value, self());
086        } else if (getOptions().acceptsType(value.getClass())) {
087            setValue(value); // Just write if no applicable serializer exists?
088        } else {
089            throw new ObjectMappingException("No serializer available for type " + type);
090        }
091        return self();
092    }
093
094    /**
095     * {@inheritDoc}
096     */
097    @Override
098    @NonNull List<N> getChildrenList();
099
100    /**
101     * {@inheritDoc}
102     */
103    @Override
104    @NonNull Map<Object, N> getChildrenMap();
105
106    /**
107     * {@inheritDoc}
108     */
109    @Override
110    @NonNull N getNode(@NonNull Iterable<Object> path);
111
112    /**
113     * Execute an action on this node. This allows performing multiple operations
114     * on a single node without having to clutter up the surrounding scope.
115     *
116     * @param action The action to perform on this node
117     * @return this
118     */
119    default N act(Consumer<? super N> action) {
120        action.accept(self());
121        return self();
122    }
123
124    /**
125     * Visit this node hierarchy as described in {@link ConfigurationVisitor}
126     *
127     * @param visitor The visitor
128     * @param <S> The state type
129     * @param <T> The terminal type
130     * @param <E> exception type that may be thrown
131     * @throws E when throw by visitor implementation
132     * @return The returned terminal from the visitor
133     */
134    default <S, T, E extends Exception> T visit(ConfigurationVisitor<N, S, T, E> visitor) throws E {
135        return visit(visitor, visitor.newState());
136    }
137
138    /**
139     * Visit this node hierarchy as described in {@link ConfigurationVisitor}
140     *
141     * @param visitor The visitor
142     * @param state The state to start with
143     * @param <T> The terminal type
144     * @param <S> The state type
145     * @param <E> exception type that may be thrown
146     * @throws E when throw by visitor implementation
147     * @return The returned terminal from the visitor
148     */
149    <S, T, E extends Exception> T visit(ConfigurationVisitor<? super N, S, T, E> visitor, S state) throws E;
150
151    /**
152     * Visit this node hierarchy as described in {@link ConfigurationVisitor}
153     * This overload will remove the need for exception handling for visitors that do not have any checked exceptions.
154     *
155     * @param visitor The visitor
156     * @param <S> The state type
157     * @param <T> The terminal type
158     * @return The returned terminal from the visitor
159     */
160    default <S, T> T visit(ConfigurationVisitor.Safe<N, S, T> visitor) {
161        return visit(visitor, visitor.newState());
162    }
163
164    /**
165     * Visit this node hierarchy as described in {@link ConfigurationVisitor}
166     * This overload will remove the need for exception handling for visitors that do not have any checked exceptions.
167     *
168     * @param visitor The visitor
169     * @param state The state to start with
170     * @param <T> The terminal type
171     * @param <S> The state type
172     * @return The returned terminal from the visitor
173     */
174    <S, T> T visit(ConfigurationVisitor.Safe<? super N, S, T> visitor, S state);
175}
