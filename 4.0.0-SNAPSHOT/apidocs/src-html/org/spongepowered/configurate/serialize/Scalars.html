017package org.spongepowered.configurate.serialize;
018
019import java.net.URL;
020import java.util.UUID;
021import java.util.regex.Pattern;
022
023/**
024 * Scalar value serializers available
025 *
026 * Each of these serializers can be used through a configuration node, or directly with a value.
027 */
028public interface Scalars {
029    ScalarSerializer<Boolean> BOOLEAN = new BooleanSerializer();
030    ScalarSerializer<String> STRING = new StringSerializer();
031    ScalarSerializer<Character> CHAR = new CharSerializer();
032    ScalarSerializer<Enum<?>> ENUM = new EnumValueSerializer();
033    ScalarSerializer<Pattern> PATTERN = new PatternSerializer();
034    ScalarSerializer<java.net.URI> URI = new URISerializer();
035    ScalarSerializer<URL> URL = new URLSerializer();
036    ScalarSerializer<UUID> UUID = new UUIDSerializer();
037    ScalarSerializer<Byte> BYTE = NumericSerializers.BYTE;
038    ScalarSerializer<Short> SHORT = NumericSerializers.SHORT;
039    ScalarSerializer<Integer> INTEGER = NumericSerializers.INTEGER;
040    ScalarSerializer<Long> LONG = NumericSerializers.LONG;
041    ScalarSerializer<Float> FLOAT = NumericSerializers.FLOAT;
042    ScalarSerializer<Double> DOUBLE = NumericSerializers.DOUBLE;
043}
