017package org.spongepowered.configurate.serialize;
018
019import java.net.URL;
020import java.util.UUID;
021import java.util.regex.Pattern;
022
023/**
024 * Scalar value serializers available
025 *
026 * <p>Each of these serializers can be used through a configuration node, or
027 * directly with a value.
028 */
029public final class Scalars {
030
031    private Scalars() {}
032
033    /**
034     * Serializer for {@code boolean} values.
035     *
036     * <p>Case-insensitive true values are: {@code true}, {@code t},
037     * {@code yes}, {@code y}, and {@code 1}.</p>
038     *
039     * <p>Case-insensitive false values are: {@code false}, {@code f},
040     * {@code no}, {@code n}, and {@code 0}</p>
041     */
042    public static final ScalarSerializer<Boolean> BOOLEAN = new BooleanSerializer();
043
044    /**
045     * Serializer for {@link String} values.
046     *
047     * <p>Values that are not already strings are converted
048     * using {@link Object#toString()}.</p>
049     */
050    public static final ScalarSerializer<String> STRING = new StringSerializer();
051
052    /**
053     * Serializer for {@code char} values.
054     *
055     * <p>A character can be converted from a 1-long {@link String}, or
056     * a number.</p>
057     */
058    public static final ScalarSerializer<Character> CHAR = new CharSerializer();
059
060    /**
061     * Serializer for values in {@code enum} classes.
062     *
063     * <p>Value lookup is case-insensitive and ignores underscores.</p>
064     */
065    public static final ScalarSerializer<Enum<?>> ENUM = new EnumValueSerializer();
066
067    /**
068     * Serializer for {@link Pattern} values.
069     *
070     * <p>Patterns will be compiled with default options.</p>
071     */
072    public static final ScalarSerializer<Pattern> PATTERN = new PatternSerializer();
073
074    /**
075     * Serializer for {@link java.net.URI} values.
076     */
077    public static final ScalarSerializer<java.net.URI> URI = new UriSerializer();
078
079    /**
080     * Serializer for {@link URL} values.
081     */
082    public static final ScalarSerializer<URL> URL = new UrlSerializer();
083
084    /**
085     * Serializer for {@link UUID} values.
086     *
087     * <p>UUIDs will be accept in RFC format, and RFC format without
088     * dashes (Mojang style).</p>
089     */
090    public static final ScalarSerializer<UUID> UUID = new UuidSerializer();
091
092    /**
093     * Serializer for {@link Byte} values.
094     */
095    public static final ScalarSerializer<Byte> BYTE = NumericSerializers.BYTE;
096
097    /**
098     * Serializer for {@link Short} values.
099     */
100    public static final ScalarSerializer<Short> SHORT = NumericSerializers.SHORT;
101
102    /**
103     * Serializer for {@link Integer} values.
104     */
105    public static final ScalarSerializer<Integer> INTEGER = NumericSerializers.INTEGER;
106
107    /**
108     * Serializer for {@link Long} values.
109     */
110    public static final ScalarSerializer<Long> LONG = NumericSerializers.LONG;
111
112    /**
113     * Serializer for {@link Float} values.
114     */
115    public static final ScalarSerializer<Float> FLOAT = NumericSerializers.FLOAT;
116
117    /**
118     * Serializer for {@link Double} values.
119     */
120    public static final ScalarSerializer<Double> DOUBLE = NumericSerializers.DOUBLE;
121
122}
