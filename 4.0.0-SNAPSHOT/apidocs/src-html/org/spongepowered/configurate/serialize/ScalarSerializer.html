017package org.spongepowered.configurate.serialize;
018
019import io.leangen.geantyref.GenericTypeReflector;
020import io.leangen.geantyref.TypeToken;
021import org.checkerframework.checker.nullness.qual.Nullable;
022import org.spongepowered.configurate.ConfigurationNode;
023
024import java.lang.reflect.Type;
025import java.util.List;
026import java.util.function.Predicate;
027
028/**
029 * Serialize a value that can be represented as a scalar value within a node.
030 * Implementations must be able to serialize when one of the accepted types is
031 * a {@link String}, and may support any other types as desired.
032 *
033 * <p>When serializing to a node, null values will be passed through directly.
034 * If the type serialized by this serializer is one of the native types of the
035 * backing node, it will be written directly to the node without
036 * any transformation.
037 *
038 * <p>Any serialized value must be deserializable by the same serializer.
039 *
040 * @param <T> the object type to serialize
041 */
042public abstract class ScalarSerializer<T> implements TypeSerializer<T> {
043
044    private final TypeToken<T> type;
045
046    @SuppressWarnings("unchecked")
047    protected ScalarSerializer(final TypeToken<T> type) {
048        final Type boxed = GenericTypeReflector.box(type.getType());
049        this.type = boxed == type.getType() ? type : (TypeToken<T>) TypeToken.get(boxed);
050    }
051
052    protected ScalarSerializer(final Class<T> type) {
053        if (type.getTypeParameters().length > 0) {
054            throw new IllegalArgumentException("Provided type " + type + " has type parameters but was not provided as a TypeToken!");
055        }
056        this.type = TypeToken.get(type);
057    }
058
059    @SuppressWarnings("unchecked")
060    ScalarSerializer(final Type type) {
061        this.type = (TypeToken<T>) TypeToken.get(type);
062    }
063
064    /**
065     * Get the general type token applicable for this serializer. This token may
066     * be parameterized.
067     *
068     * @return the type token for this serializer
069     */
070    public final TypeToken<T> type() {
071        return this.type;
072    }
073
074    @Override
075    public final T deserialize(Type type, final ConfigurationNode node) throws SerializationException {
076        ConfigurationNode deserializeFrom = node;
077        if (node.isList()) {
078            final List<? extends ConfigurationNode> children = node.childrenList();
079            if (children.size() == 1) {
080                deserializeFrom = children.get(0);
081            }
082        }
083
084        if (deserializeFrom.isList() || deserializeFrom.isMap()) {
085            throw new SerializationException(type, "Value must be provided as a scalar!");
086        }
087
088        final @Nullable Object value = deserializeFrom.rawScalar();
089        if (value == null) {
090            throw new SerializationException(type, "No scalar value present");
091        }
092
093        type = GenericTypeReflector.box(type); // every primitive type should be boxed (cuz generics!)
094        final @Nullable T possible = cast(value);
095        if (possible != null) {
096            return possible;
097        }
098
099        return deserialize(type, value);
100    }
101
102    /**
103     * Attempt to deserialize the provided object using an unspecialized type.
104     * This may fail on more complicated deserialization processes such as with
105     * enum types.
106     *
107     * @param value the object to deserialize.
108     * @return the deserialized object, if possible
109     * @throws SerializationException if unable to coerce the value to the
110     *                                requested type.
111     */
112    public final T deserialize(final Object value) throws SerializationException {
113        final @Nullable T possible = cast(value);
114        if (possible != null) {
115            return possible;
116        }
117
118        return this.deserialize(this.type().getType(), value);
119    }
120
121    /**
122     * Given an object of unknown type, attempt to convert it into the given
123     * type.
124     *
125     * @param type the specific type of the type's usage
126     * @param obj the object to convert
127     * @return a converted object
128     * @throws SerializationException if the object could not be converted for
129     *                                any reason
130     */
131    public abstract T deserialize(Type type, Object obj) throws SerializationException;
132
133    @Override
134    public final void serialize(final Type type, final @Nullable T obj, final ConfigurationNode node) {
135        if (obj == null) {
136            node.raw(null);
137            return;
138        }
139
140        if (node.options().acceptsType(obj.getClass())) {
141            node.raw(obj);
142            return;
143        }
144
145        node.raw(serialize(obj, node.options()::acceptsType));
146    }
147
148    /**
149     * Serialize the provided value to a supported type, testing against the
150     * provided predicate.
151     *
152     * @param item the value to serialize
153     * @param typeSupported a predicate to allow choosing which types are
154     *                      supported
155     * @return a serialized form of this object
156     */
157    protected abstract Object serialize(T item, Predicate<Class<?>> typeSupported);
158
159    @SuppressWarnings("unchecked")
160    private @Nullable T cast(final Object value) {
161        final Class<?> rawType = GenericTypeReflector.erase(this.type().getType());
162        if (rawType.isInstance(value)) {
163            return (T) value;
164        }
165        return null;
166    }
167
168    /**
169     * Attempt to deserialize the provided object, but rather than throwing an
170     * exception when a deserialization error occurs, return null instead.
171     *
172     * @param obj the object to try to deserialize
173     * @return an instance of the appropriate type, or null
174     * @see #deserialize(Object)
175     */
176    public final @Nullable T tryDeserialize(final @Nullable Object obj) {
177        if (obj == null) {
178            return null;
179        }
180
181        try {
182            return deserialize(obj);
183        } catch (final SerializationException ex) {
184            return null;
185        }
186    }
187
188    /**
189     * Serialize the item to a {@link String}, in a representation that can be
190     * interpreted by this serializer again.
191     *
192     * @param item the item to serialize
193     * @return the serialized form of the item
194     */
195    public final String serializeToString(final T item) {
196        if (item instanceof CharSequence) {
197            return item.toString();
198        }
199        // Otherwise, use the serializer
200        return (String) serialize(item, clazz -> clazz.isAssignableFrom(String.class));
201    }
202
203}
