082    private final @Nullable TypeSerializerCollection parent;
083    private final List<RegisteredSerializer> serializers;
084    private final Map<Type, TypeSerializer<?>> typeMatches = new ConcurrentHashMap<>();
085
086    private TypeSerializerCollection(final @Nullable TypeSerializerCollection parent, final List<RegisteredSerializer> serializers) {
087        this.parent = parent;
088        this.serializers = UnmodifiableCollections.copyOf(serializers);
089    }
090
091    /**
092     * Resolve a type serializer.
093     *
094     * <p>First, all registered serializers from this collection are queried
095     * then if a parent collection is set, that collection is queried.
096     *
097     * @param token the type a serializer is required for
098     * @param <T> the type to serialize
099     * @return a serializer if any is present, or null if no applicable
100     *          serializer is found
101     */
102    @SuppressWarnings("unchecked")
103    public <T> @Nullable TypeSerializer<T> get(final TypeToken<T> token) {
104        requireNonNull(token, "type");
105        return (TypeSerializer<T>) get(token.getType());
106    }
107
108    /**
109     * Resolve a type serializer.
110     *
111     * <p>First, all registered serializers from this collection are queried
112     * then if a parent collection is set, that collection is queried.
113     *
114     * <p>This method will fail when provided a raw parameterized type</p>
115     *
116     * @param token the type a serializer is required for
117     * @param <T> the type to serialize
118     * @return a serializer if any is present, or null if no applicable
119     *          serializer is found
120     */
121    @SuppressWarnings("unchecked")
122    public <T> @Nullable TypeSerializer<T> get(final Class<T> token) {
123        requireNonNull(token, "type");
124        requireCompleteParameters(token);
125
126        return (TypeSerializer<T>) get((Type) token);
127    }
128
129    /**
130     * Resolve a type serializer.
131     *
132     * <p>First, all registered serializers from this collection are queried
133     * then if a parent collection is set, that collection is queried.
134     *
135     * @param type the type a serializer is required for
136     * @return a serializer if any is present, or null if no applicable
137     *          serializer is found
138     */
139    public @Nullable TypeSerializer<?> get(Type type) {
140        type = GenericTypeReflector.toCanonicalBoxed(annotate(requireNonNull(type, "type"))).getType();
141        @Nullable TypeSerializer<?> serial = this.typeMatches.computeIfAbsent(type, param -> {
142            for (RegisteredSerializer ent : this.serializers) {
143                if (ent.predicate.test(param)) {
144                    return ent.serializer;
145                }
146            }
147            return null;
148        });
149
150        if (serial == null && this.parent != null) {
151            serial = this.parent.get(type);
152        }
153        return serial;
154    }

166    @Override
167    public String toString() {
168        return "TypeSerializerCollection{"
169                + "parent=" + this.parent
170                + ", serializers=" + this.serializers
171                + '}';
172    }
173
174    @Override public boolean equals(final Object other) {
175        if (this == other) {
176            return true;
177        }
178        if (!(other instanceof TypeSerializerCollection)) {
179            return false;
180        }
181        final TypeSerializerCollection that = (TypeSerializerCollection) other;
182        return Objects.equals(this.parent, that.parent)
183                && this.serializers.equals(that.serializers);
184    }
185
186    @Override
187    public int hashCode() {
188        return Objects.hash(this.parent, this.serializers);
189    }

215    /**
216     * A builder to construct new serializer collections.
217     */
218    public static class Builder {
219        private final @Nullable TypeSerializerCollection parent;
220        private final List<RegisteredSerializer> serializers = new ArrayList<>();
221
222        Builder(final @Nullable TypeSerializerCollection parent) {
223            this.parent = parent;
224        }
225
226        /**
227         * Register a type serializer for a given type.
228         *
229         * <p>Serializers registered will match all subclasses of the provided
230         * type, as well as unwrapped primitive equivalents of the type.
231         *
232         * @param type the type to accept
233         * @param serializer the serializer that will be serialized with
234         * @param <T> the type to generify around
235         * @return this builder
236         */
237        public <T> Builder register(final TypeToken<T> type, final TypeSerializer<? super T> serializer) {
238            return register0(type.getType(), serializer);
239        }
240
241        /**
242         * Register a type serializer for a given type.
243         *
244         * <p>Serializers registered will match all subclasses of the provided
245         * type, as well as unboxed primitive equivalents of the type.
246         *
247         * @param type the type to accept
248         * @param serializer the serializer that will be serialized with
249         * @param <T> the type to generify around
250         * @return this builder
251         */
252        public <T> Builder register(final Class<T> type, final TypeSerializer<? super T> serializer) {
253            return register0(type, serializer);
254        }
255
256        /**
257         * Register a type serializer matching against a given predicate.
258         *
259         * @param test the predicate to match types against
260         * @param serializer the serializer to serialize matching types with
261         * @param <T> the type parameter
262         * @return this builder
263         */
264        public <T> Builder register(final Predicate<Type> test, final TypeSerializer<? super T> serializer) {
265            requireNonNull(test, "test");
266            requireNonNull(serializer, "serializer");
267            this.serializers.add(new RegisteredSerializer(test, serializer));
268            return this;
269        }
270
271        /**
272         * Register a scalar serializer with its own attached type token.
273         *
274         * <p>Serializers registered will match all subclasses of the provided
275         * type, as well as unboxed primitive equivalents of the type.</p>
276         *
277         * @param serializer serializer to register
278         * @param <T> value type
279         * @return this builder
280         */
281        public <T> Builder register(final ScalarSerializer<T> serializer) {
282            requireNonNull(serializer, "serializer");
283            return register(serializer.type(), serializer);
284        }
285
286        private Builder register0(final Type type, final TypeSerializer<?> serializer) {
287            requireNonNull(type, "type");
288            requireNonNull(serializer, "serializer");
289            this.serializers.add(new RegisteredSerializer(test -> {
290                // Test direct type
291                if (GenericTypeReflector.isSuperType(type, test)) {
292                    return true;
293                }
294
295                // And upper bounds
296                if (test instanceof WildcardType) {
297                    final Type[] upperBounds = ((WildcardType) test).getUpperBounds();
298                    if (upperBounds.length == 1) {
299                        return isSuperType(type, upperBounds[0]);
300                    }
301                }
302                return false;
303            }, serializer));
304            return this;
305        }
306
307        /**
308         * Register an <em>exact</em> type serializer for a given type.
309         *
310         * <p>Serializers will only match exact object types. For example, a
311         * serializer registered for {@code List<String>} would not match when
312         * {@code ArrayList<String>} is queried.</p>
313         *
314         * @param type the type to accept
315         * @param serializer the serializer that will be serialized with
316         * @param <T> the type to generify around
317         * @return this builder
318         */
319        public <T> Builder registerExact(final TypeToken<T> type, final TypeSerializer<? super T> serializer) {
320            return registerExact0(type.getType(), serializer);
321        }
322
323        /**
324         * Register an <em>exact</em> type serializer for a given type.
325         *
326         * <p>Serializers will only match exact object types. For example, a
327         * serializer registered for {@code List<String>} would not match when
328         * {@code ArrayList<String>} is queried.</p>
329         *
330         * @param type the type to accept
331         * @param serializer the serializer that will be serialized with
332         * @param <T> the type to generify around
333         * @return this builder
334         */
335        public <T> Builder registerExact(final Class<T> type, final TypeSerializer<? super T> serializer) {
336            return registerExact0(type, serializer);
337        }
338
339        /**
340         * Register a scalar serializer with its own attached type token.
341         *
342         * <p>Serializers will only match exact object types. For example, a
343         * serializer registered for {@code List<String>} would not match when
344         * {@code ArrayList<String>} is queried.</p>
345         *
346         * @param serializer serializer to register
347         * @param <T> value type
348         * @return this builder
349         */
350        public <T> Builder registerExact(final ScalarSerializer<T> serializer) {
351            requireNonNull(serializer, "serializer");
352            return registerExact(serializer.type(), serializer);
353        }
354
355        private Builder registerExact0(final Type type, final TypeSerializer<?> serializer) {
356            requireNonNull(type, "type");
357            requireNonNull(serializer, "serializer");
358            this.serializers.add(new RegisteredSerializer(test -> test.equals(type), serializer));
359            return this;
360        }
361
362        /**
363         * Register all serializers from {@code other} into this collection.
364         *
365         * <p>Creating a child collection should be preferred, but when merging
366         * multiple sets of serializers together, directly adding other
367         * collections may be the best choice.</p>
368         *
369         * @param other source collection
370         * @return this builder
371         */
372        public Builder registerAll(final TypeSerializerCollection other) {
373            this.serializers.addAll(requireNonNull(other, "other").serializers);
374            return this;
375        }
376
377        /**
378         * Register a customized object mapper to handle
379         * {@link ConfigSerializable}-annotated objects.
380         *
381         * @param factory factory to retrieve object mappers from
382         * @return this builder
383         */
384        public Builder registerAnnotatedObjects(final ObjectMapper.Factory factory) {
385            return register(Builder::isAnnotatedTarget, factory.asTypeSerializer());
386        }
387
388        /**
389         * A predicate to restrict the type serializer created by
390         * {@link ObjectMapper.Factory#asTypeSerializer()} to annotated types.
391         *
392         * @return whether a type is annotated with {@link ConfigSerializable}
393         */
394        static boolean isAnnotatedTarget(final Type type) {
395            return GenericTypeReflector.annotate(type).isAnnotationPresent(ConfigSerializable.class);
396        }
397
398        /**
399         * Create a new type serializer collection.
400         *
401         * @return a newly created collection
402         */
403        public TypeSerializerCollection build() {
404            return new TypeSerializerCollection(this.parent, this.serializers);
405        }
406    }
407
408    private static final class RegisteredSerializer {
409
410        private final Predicate<Type> predicate;
411        private final TypeSerializer<?> serializer;
412
413        private RegisteredSerializer(final Predicate<Type> predicate, final TypeSerializer<?> serializer) {
414            this.predicate = predicate;
415            this.serializer = serializer;
416        }
417
418    }
