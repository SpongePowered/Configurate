017package org.spongepowered.configurate.serialize;
018
019import com.google.common.base.Preconditions;
020import com.google.common.collect.ImmutableList;
021import com.google.common.reflect.TypeToken;
022import org.checkerframework.checker.nullness.qual.Nullable;
023
024import java.util.List;
025import java.util.Map;
026import java.util.concurrent.ConcurrentHashMap;
027import java.util.function.Predicate;
028
029import static java.util.Objects.requireNonNull;
030
031/**
032 * A calculated collection of {@link TypeSerializer}s
033 */
034public class TypeSerializerCollection {
035    private static final TypeSerializerCollection DEFAULTS;
036
037    static {
038        DEFAULTS = TypeSerializerCollection.builder()
039                .register(Scalars.STRING)
040                .register(Scalars.BOOLEAN)
041                .register(MapSerializer.TYPE, new MapSerializer())
042                .register(ListSerializer.TYPE, new ListSerializer())
043                .register(Scalars.BYTE)
044                .register(Scalars.SHORT)
045                .register(Scalars.INTEGER)
046                .register(Scalars.LONG)
047                .register(Scalars.FLOAT)
048                .register(Scalars.DOUBLE)
049                .register(AnnotatedObjectSerializer.predicate(), new AnnotatedObjectSerializer())
050                .register(Scalars.ENUM)
051                .register(Scalars.CHAR)
052                .register(Scalars.URI)
053                .register(Scalars.URL)
054                .register(Scalars.UUID)
055                .register(Scalars.PATTERN)
056                .register(ArraySerializer.Objects.predicate(), new ArraySerializer.Objects())
057                .register(ArraySerializer.Booleans.TYPE, new ArraySerializer.Booleans())
058                .register(ArraySerializer.Bytes.TYPE, new ArraySerializer.Bytes())
059                .register(ArraySerializer.Chars.TYPE, new ArraySerializer.Chars())
060                .register(ArraySerializer.Shorts.TYPE, new ArraySerializer.Shorts())
061                .register(ArraySerializer.Ints.TYPE, new ArraySerializer.Ints())
062                .register(ArraySerializer.Longs.TYPE, new ArraySerializer.Longs())
063                .register(ArraySerializer.Floats.TYPE, new ArraySerializer.Floats())
064                .register(ArraySerializer.Doubles.TYPE, new ArraySerializer.Doubles())
065                .register(SetSerializer.TYPE, new SetSerializer())
066                .register(ConfigurationNodeSerializer.TYPE, new ConfigurationNodeSerializer())
067                .build();
068    }
069
070    private final @Nullable TypeSerializerCollection parent;
071    private final List<RegisteredSerializer> serializers;
072    private final Map<TypeToken<?>, TypeSerializer<?>> typeMatches = new ConcurrentHashMap<>();
073
074    private TypeSerializerCollection(@Nullable TypeSerializerCollection parent, List<RegisteredSerializer> serializers) {
075        this.parent = parent;
076        this.serializers = ImmutableList.copyOf(serializers);
077    }
078
079    /**
080     * Resolve a type serializer. First, all registered serializers from this collection are queried,
081     * then if a parent collection is set that collection is queried.
082     *
083     * @param type The type a serializer is required for
084     * @param <T> The type to serialize
085     * @return A serializer if any is present, or null if no applicable serializer is found
086     */
087    @SuppressWarnings({"unchecked", "rawtypes"})
088    public <T> @Nullable TypeSerializer<T> get(TypeToken<T> type) {
089        Preconditions.checkNotNull(type, "type");
090        type = type.wrap();
091
092        @Nullable TypeSerializer<?> serial = typeMatches.computeIfAbsent(type, token -> {
093            for (RegisteredSerializer ent : serializers) {
094                if (ent.predicate.test(token)) {
095                    return ent.serializer;
096                }
097            }
098            return null;
099        });
100
101        if (serial == null && parent != null) {
102            serial = parent.get(type);
103        }
104
105        return (TypeSerializer) serial;
106    }
107
108    /**
109     * Create a new builder to begin building a collection of type serializers that inherits from this collection
110     *
111     * @return The new builder
112     */
113    public Builder childBuilder() {
114        return new Builder(this);
115    }
116
117    /**
118     * Create a builder for a new type serializer collection without a parent set.
119     *
120     * If <em>any</em> of the standard serializers provided by Configurate are desired,
121     * either the default collection or a collection inheriting from the default collection should be applied.
122     *
123     * @return The builder
124     */
125    public static Builder builder() {
126        return new Builder(null);
127    }
128
129    /**
130     * Get a collection containing all of Configurate's built-in type serializers.
131     *
132     * @return The collection
133     */
134    public static TypeSerializerCollection defaults() {
135        return DEFAULTS;
136    }
137
138    public static class Builder {
139        private final @Nullable TypeSerializerCollection parent;
140        private final ImmutableList.Builder<RegisteredSerializer> serializers = ImmutableList.builder();
141
142        Builder(@Nullable TypeSerializerCollection parent) {
143            this.parent = parent;
144        }
145
146        /**
147         * Register a type serializer for a given type. Serializers registered will match all subclasses of the provided
148         * type, as well as unwrapped primitive equivalents of the type.
149         *
150         * @param type The type to accept
151         * @param serializer The serializer that will be serialized with
152         * @param <T> The type to generify around
153         * @return this
154         */
155        public <T> Builder register(TypeToken<T> type, TypeSerializer<? super T> serializer) {
156            Preconditions.checkNotNull(type, "type");
157            Preconditions.checkNotNull(serializer, "serializer");
158            serializers.add(new RegisteredSerializer(new SuperTypePredicate(type), serializer));
159            return this;
160        }
161
162        /**
163         * Register a type serializer matching against a given predicate.
164         *
165         * @param test The predicate to match types against
166         * @param serializer The serializer to serialize matching types with
167         * @param <T> The type parameter
168         * @return this
169         */
170        @SuppressWarnings({"unchecked", "rawtypes"})
171        public <T> Builder register(Predicate<TypeToken<T>> test, TypeSerializer<? super T> serializer) {
172            requireNonNull(test, "test");
173            requireNonNull(serializer, "serializer");
174            serializers.add(new RegisteredSerializer((Predicate) test, serializer));
175            return this;
176        }
177
178        public <T> Builder register(ScalarSerializer<T> serializer) {
179            requireNonNull(serializer, "serializer");
180            return register(serializer.type(), serializer);
181        }
182
183        /**
184         * Create a new type serializer collection
185         *
186         * @return The resulting collection
187         */
188        public TypeSerializerCollection build() {
189            return new TypeSerializerCollection(this.parent, this.serializers.build());
190        }
191    }
192
193    private static final class RegisteredSerializer {
194        private final Predicate<TypeToken<?>> predicate;
195        private final TypeSerializer<?> serializer;
196
197        private RegisteredSerializer(Predicate<TypeToken<?>> predicate, TypeSerializer<?> serializer) {
198            this.predicate = predicate;
199            this.serializer = serializer;
200        }
201    }
202}
