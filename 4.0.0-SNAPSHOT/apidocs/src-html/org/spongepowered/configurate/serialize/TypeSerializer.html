017package org.spongepowered.configurate.serialize;
018
019import com.google.common.reflect.TypeToken;
020import org.checkerframework.checker.nullness.qual.Nullable;
021import org.spongepowered.configurate.ScopedConfigurationNode;
022import org.spongepowered.configurate.objectmapping.ObjectMappingException;
023import org.spongepowered.configurate.util.CheckedFunction;
024
025import java.util.function.BiFunction;
026import java.util.function.Predicate;
027
028/**
029 * Represents an object which can serialize and deserialize objects of a given type.
030 *
031 * @param <T> The type
032 */
033public interface TypeSerializer<T> {
034
035    /**
036     * Given the provided functions, create a new serializer for a scalar value.
037     *
038     * The returned serializer must fulfill all the requirements of a {@link ScalarSerializer}
039     *
040     * @param type The type of value returned by the serializer
041     * @param serializer The serialization function, implementing {@link ScalarSerializer#serialize(Object, Predicate)}
042     * @param deserializer The deserialization function, implementing {@link ScalarSerializer#deserialize(TypeToken, Object)}
043     * @param <T> The type of value to deserialize
044     * @return A new and unregistered type serializer
045     */
046    static <T> ScalarSerializer<T> of(TypeToken<T> type, BiFunction<T, Predicate<Class<?>>, Object> serializer, CheckedFunction<Object, T, ObjectMappingException> deserializer) {
047        return new FunctionScalarSerializer<>(type, deserializer, serializer);
048    }
049
050    /**
051     * Given the provided functions, create a new serializer for a scalar value.
052     *
053     * The returned serializer must fulfill all the requirements of a {@link ScalarSerializer}
054     *
055     * @param type The type of value. Must not be a parameterized type
056     * @param serializer The serialization function, implementing {@link ScalarSerializer#serialize(Object, Predicate)}
057     * @param deserializer The deserialization function, implementing {@link ScalarSerializer#deserialize(TypeToken, Object)}
058     * @param <T> The type of value to deserialize
059     * @see #of(TypeToken, BiFunction, CheckedFunction) for the version of this function that takes a parameterized type
060     * @return A new and unregistered type serializer
061     */
062    static <T> ScalarSerializer<T> of(Class<T> type, BiFunction<T, Predicate<Class<?>>, Object> serializer, CheckedFunction<Object, T, ObjectMappingException> deserializer) {
063        if (type.getTypeParameters().length > 0) {
064            throw new IllegalArgumentException("Parameterized types must be specified using TypeTokens, not raw classes");
065        }
066
067        return new FunctionScalarSerializer<>(TypeToken.of(type), deserializer, serializer);
068    }
069
070    /**
071     * Deserialize an object (of the correct type) from the given configuration node.
072     *
073     * @param type The type of return value required
074     * @param node The node containing serialized data
075     * @param <Node> The type of node to deserialize from
076     * @return An object
077     * @throws ObjectMappingException If the presented data is invalid
078     */
079    <Node extends ScopedConfigurationNode<Node>> @Nullable T deserialize(TypeToken<?> type, Node node) throws ObjectMappingException;
080
081    /**
082     * Serialize an object to the given configuration node.
083     *
084     * @param type The type of the input object
085     * @param obj The object to be serialized
086     * @param node The node to write to
087     * @param <Node> The type of node to serialize to
088     * @throws ObjectMappingException If the object cannot be serialized
089     */
090    <Node extends ScopedConfigurationNode<Node>> void serialize(TypeToken<?> type, @Nullable T obj, Node node) throws ObjectMappingException;
091
092}
