017package org.spongepowered.configurate.serialize;
018
019import org.checkerframework.checker.nullness.qual.Nullable;
020import org.spongepowered.configurate.ConfigurateException;
021import org.spongepowered.configurate.ConfigurationNode;
022import org.spongepowered.configurate.transformation.NodePath;
023
024import java.lang.reflect.Type;
025
026/**
027 * Exception thrown on errors encountered while using type serializers.
028 */
029public class SerializationException extends ConfigurateException {
030
031    public static final long serialVersionUID = -910568375387191784L;
032    private transient @Nullable Type expectedType;
033
034    /**
035     * Create an exception with unknown message and cause.
036     */
037    public SerializationException() {
038        this.expectedType = null;
039    }
040
041    /**
042     * Create an exception without a cause.
043     *
044     * @param message message with information about the exception
045     */
046    public SerializationException(final String message) {
047        super(message);
048    }
049
050    /**
051     * Create an exception with a cause and no additional information.
052     *
053     * @param cause wrapped causing throwable
054     */
055    public SerializationException(final Throwable cause) {
056        super(cause);
057    }
058
059    /**
060     * Create an exception without a cause.
061     *
062     * @param expectedType declared type being processed
063     * @param message message with information about the exception
064     */
065    public SerializationException(final Type expectedType, final String message) {
066        super(message);
067        this.expectedType = expectedType;
068    }
069
070    /**
071     * Create an exception with a cause and no additional information.
072     *
073     * @param expectedType declared type being processed
074     * @param cause wrapped causing throwable
075     */
076    public SerializationException(final Type expectedType, final Throwable cause) {
077        super(cause);
078        this.expectedType = expectedType;
079    }
080
081    /**
082     * Create an exception with message and wrapped cause.
083     *
084     * @param expectedType declared type being processed
085     * @param message informational message
086     * @param cause cause to be wrapped
087     */
088    public SerializationException(final Type expectedType, final String message, final Throwable cause) {
089        super(message, cause);
090        this.expectedType = expectedType;
091    }
092
093    /**
094     * Create an exception with a message and unknown cause.
095     *
096     * @param node node being processed
097     * @param message informational message
098     * @param expectedType declared type being processed
099     */
100    public SerializationException(final ConfigurationNode node, final Type expectedType, final String message) {
101        this(node, expectedType, message, null);
102    }
103
104    /**
105     * Create an exception with wrapped cause.
106     *
107     * @param node node being processed
108     * @param expectedType declared type being processed
109     * @param cause cause to be wrapped
110     */
111    public SerializationException(final ConfigurationNode node, final Type expectedType, final Throwable cause) {
112        this(node, expectedType, null, cause);
113    }
114
115    /**
116     * Create an exception with message and wrapped cause.
117     *
118     * @param node node being processed
119     * @param expectedType declared type being processed
120     * @param message informational message
121     * @param cause cause to be wrapped
122     */
123    public SerializationException(final ConfigurationNode node, final Type expectedType,
124            final @Nullable String message, final @Nullable Throwable cause) {
125        super(node, message, cause);
126        this.expectedType = expectedType;
127    }
128
129    /**
130     * Create an exception with message and wrapped cause.
131     *
132     * @param path path to node being processed
133     * @param expectedType declared type being processed
134     * @param message informational message
135     */
136    public SerializationException(final NodePath path, final Type expectedType, final String message) {
137        super(path, message, null);
138
139        this.expectedType = expectedType;
140    }
141
142    /**
143     * Get the desired type associated with the failed object mapping operation.
144     *
145     * @return type
146     */
147    public @Nullable Type expectedType() {
148        return this.expectedType;
149    }
150
151    @Override
152    public @Nullable String getMessage() {
153        if (this.expectedType == null) {
154            return super.getMessage();
155        } else {
156            return path() + " of type " + this.expectedType.getTypeName() + ": " + rawMessage();
157        }
158    }
159
160    /**
161     * Initialize the expected type.
162     *
163     * <p>If a type has already been set, it will not be overridden.</p>
164     *
165     * @param type expected type
166     */
167    public void initType(final Type type) {
168        if (this.expectedType != null) {
169            this.expectedType = type;
170        }
171    }
172
173}
