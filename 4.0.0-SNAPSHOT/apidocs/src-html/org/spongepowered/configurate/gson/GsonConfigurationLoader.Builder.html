020import com.google.common.collect.ImmutableList;

023import com.google.gson.JsonParseException;
024import com.google.gson.stream.JsonReader;

028import org.checkerframework.checker.nullness.qual.Nullable;

036import java.io.BufferedReader;
037import java.io.IOException;
038import java.io.Writer;
039import java.util.List;
040import java.util.Map;

042/**
043 * A loader for JSON-formatted configurations, using the GSON library for parsing and generation.
044 */
045public class GsonConfigurationLoader extends AbstractConfigurationLoader<BasicConfigurationNode> {
046
047    /**
048     * Creates a new {@link GsonConfigurationLoader} builder.
049     *
050     * @return A new builder
051     */
052    @NonNull
053    public static Builder builder() {
054        return new Builder();
055    }
056
057    /**
058     * Builds a {@link GsonConfigurationLoader}.
059     */
060    public static class Builder extends AbstractConfigurationLoader.Builder<Builder> {
061        private boolean lenient = true;
062        private int indent = 2;
063
064        protected Builder() {
065        }
066
067        /**
068         * Sets the level of indentation the resultant loader should use.
069         *
070         * @param indent The indent level
071         * @return This builder (for chaining)
072         */
073        @NonNull
074        public Builder setIndent(int indent) {
075            this.indent = indent;
076            return this;
077        }
078
079        /**
080         * Gets the level of indentation to be used by the resultant loader.
081         *
082         * @return The indent level
083         */
084        public int getIndent() {
085            return this.indent;
086        }
087
088        /**
089         * Sets if the resultant loader should parse leniently.
090         *
091         * @see JsonReader#setLenient(boolean)
092         * @param lenient Whether the parser should parse leniently
093         * @return This builder (for chaining)
094         */
095        @NonNull
096        public Builder setLenient(boolean lenient) {
097            this.lenient = lenient;
098            return this;
099        }
100
101        /**
102         * Gets if the resultant loader should parse leniently.
103         *
104         * @return Whether the parser should parse leniently
105         */
106        public boolean isLenient() {
107            return this.lenient;
108        }
109
110        @NonNull
111        @Override
112        public GsonConfigurationLoader build() {
113            return new GsonConfigurationLoader(this);
114        }
115    }
116
117    private final boolean lenient;
118    private final String indent;
119
120    private GsonConfigurationLoader(Builder builder) {
121        super(builder, new CommentHandler[] {CommentHandlers.DOUBLE_SLASH, CommentHandlers.SLASH_BLOCK, CommentHandlers.HASH});
122        this.lenient = builder.isLenient();
123        this.indent = Strings.repeat(" ", builder.getIndent());
124    }
125
126    @Override
127    protected void loadInternal(BasicConfigurationNode node, BufferedReader reader) throws IOException {
128        reader.mark(1);
129        if (reader.read() == -1) {
130            return;
131        }
132        reader.reset();
133        try (JsonReader parser = new JsonReader(reader)) {
134            parser.setLenient(lenient);
135            parseValue(parser, node);
136        }
137    }
138
139    private void parseValue(JsonReader parser, BasicConfigurationNode node) throws IOException {
140        JsonToken token = parser.peek();
141        switch (token) {
142            case BEGIN_OBJECT:
143                parseObject(parser, node);
144                break;
145            case BEGIN_ARRAY:
146                parseArray(parser, node);
147                break;
148            case NUMBER:
149                node.setValue(readNumber(parser));
150                break;
151            case STRING:
152                node.setValue(parser.nextString());
153                break;
154            case BOOLEAN:
155                node.setValue(parser.nextBoolean());
156                break;
157            case NULL: // Ignored values
158                parser.nextNull();
159                node.setValue(null);
160                break;
161            case NAME:
162                break;
163            default:
164                throw new IOException("Unsupported token type: " + token);
165        }
166    }
167
168    private Number readNumber(JsonReader reader) throws IOException {
169        String number = reader.nextString();
170        if (number.contains(".")) {
171            return Double.parseDouble(number);
172        }
173        long nextLong = Long.parseLong(number);
174        int nextInt = (int) nextLong;
175        if (nextInt == nextLong) {
176            return nextInt;
177        }
178        return nextLong;
179    }
180
181    private void parseArray(JsonReader parser, BasicConfigurationNode node) throws IOException {
182        parser.beginArray();
183
184        boolean written = false;
185        @Nullable JsonToken token;
186        while ((token = parser.peek()) != null) {
187            switch (token) {
188                case END_ARRAY:
189                    parser.endArray();
190                    // ensure the type is preserved
191                    if (!written) {
192                        node.setValue(ImmutableList.of());
193                    }
194                    return;
195                default:
196                    parseValue(parser, node.appendListNode());
197                    written = true;
198            }
199        }
200        throw new JsonParseException("Reached end of stream with unclosed array at!");
201
202    }
203
204    private void parseObject(JsonReader parser, BasicConfigurationNode node) throws IOException {
205        parser.beginObject();
206
207        boolean written = false;
208        @Nullable JsonToken token;
209        while ((token = parser.peek()) != null) {
210            switch (token) {
211                case END_OBJECT:
212                case END_DOCUMENT:
213                    parser.endObject();
214                    // ensure the type is preserved
215                    if (!written) {
216                        node.setValue(ImmutableMap.of());
217                    }
218                    return;
219                case NAME:
220                    parseValue(parser, node.getNode(parser.nextName()));
221                    written = true;
222                    break;
223                default:
224                    throw new JsonParseException("Received improper object value " + token);
225            }
226        }
227        throw new JsonParseException("Reached end of stream with unclosed object!");
228    }
229
230    @Override
231    public void saveInternal(ConfigurationNode node, Writer writer) throws IOException {
232        if (!lenient && !node.isMap()) {
233            throw new IOException("Non-lenient json generators must have children of map type");
234        }
235        try (JsonWriter generator = new JsonWriter(writer)) {
236            generator.setIndent(indent);
237            generator.setLenient(lenient);
238            GsonVisitor.INSTANCE.visit(node, generator);
239            writer.write(SYSTEM_LINE_SEPARATOR); // Jackson doesn't add a newline at the end of files by default
240        }
241    }
242
243    @Override
244    public BasicConfigurationNode createEmptyNode(ConfigurationOptions options) {
245        options = options.withAcceptedTypes(ImmutableSet.of(Map.class, List.class, Double.class, Float.class,
246                Long.class, Integer.class, Boolean.class, String.class));
247        return BasicConfigurationNode.root(options);
248    }
