017package org.spongepowered.configurate.objectmapping.guice;
018
019import com.google.inject.ConfigurationException;
020import com.google.inject.Injector;
021import com.google.inject.Key;
022import com.google.inject.Provider;
023import org.spongepowered.configurate.objectmapping.FieldDiscoverer;
024import org.spongepowered.configurate.objectmapping.ObjectMapper;
025import org.spongepowered.configurate.serialize.SerializationException;
026
027import javax.inject.Inject;
028import javax.inject.Singleton;
029
030/**
031 * A factory for {@link ObjectMapper}s that will inherit the injector from
032 * wherever it is provided.
033 *
034 * <p>This class is intended to be constructed through Guice
035 * dependency injection.
036 */
037@Singleton
038public final class GuiceObjectMapperProvider {
039
040    private final ObjectMapper.Factory factory;
041
042    @Inject
043    GuiceObjectMapperProvider(final Injector baseInjector) {
044        this.factory = ObjectMapper.factoryBuilder()
045                .addDiscoverer(injectedObjectDiscoverer(baseInjector))
046                .build();
047    }
048
049    /**
050     * Get the default factory, with guice support added.
051     *
052     * @return the default guice factory
053     */
054    public ObjectMapper.Factory get() {
055        return this.factory;
056    }
057
058    /**
059     * Create a field discoverer that uses the provided injector.
060     *
061     * @param injector injector to create instances with
062     * @return new discoverer
063     */
064    public static FieldDiscoverer<?> injectedObjectDiscoverer(final Injector injector) {
065        return FieldDiscoverer.object(type -> {
066            try {
067                final Provider<?> prov = injector.getProvider(Key.get(type.getType()));
068                return prov::get;
069            } catch (final ConfigurationException ex) {
070                throw new SerializationException(type.getType(), "Cannot create instances", ex);
071            }
072        });
073    }
074
075    @Override
076    public String toString() {
077        return "GuiceObjectMapperFactory{}";
078    }
079
080}
