017package org.spongepowered.configurate.objectmapping.meta;
018
019import org.checkerframework.checker.regex.qual.Regex;
020
021import java.lang.annotation.Documented;
022import java.lang.annotation.ElementType;
023import java.lang.annotation.Retention;
024import java.lang.annotation.RetentionPolicy;
025import java.lang.annotation.Target;
026
027/**
028 * Constrains a field value to ensure it matches the provided expression.
029 *
030 * <p>This constraint will always pass with an empty field. See {@link Required}
031 * to enforce a non-null value.</p>
032 */
033@Documented
034@Retention(RetentionPolicy.RUNTIME)
035@Target(ElementType.FIELD)
036public @interface Matches {
037
038    /**
039     * Pattern to test string value against.
040     *
041     * @return pattern to test against
042     */
043    @Regex String value();
044
045    /**
046     * Message to throw in an exception when a match fails.
047     *
048     * <p>This message will be formatted as a MessageFormat with two
049     * parameters:</p>
050     * <ol start="0">
051     *     <li>the input string</li>
052     *     <li>the pattern being matched</li>
053     * </ol>
054     *
055     * @return message format.
056     */
057    String failureMessage() default "";
058
059}
