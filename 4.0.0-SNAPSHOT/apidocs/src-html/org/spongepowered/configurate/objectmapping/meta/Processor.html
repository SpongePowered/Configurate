017package org.spongepowered.configurate.objectmapping.meta;
018
019import org.spongepowered.configurate.CommentedConfigurationNodeIntermediary;
020import org.spongepowered.configurate.ConfigurationNode;
021
022import java.lang.annotation.Annotation;
023import java.lang.reflect.Type;
024import java.util.ResourceBundle;
025
026/**
027 * Performs a transformation on a value annotated with a specific type.
028 */
029@FunctionalInterface
030public interface Processor<V> {
031
032    /**
033     * Transform the output node on write.
034     *
035     * @param value source value
036     * @param destination destination node
037     */
038    void process(V value, ConfigurationNode destination);
039
040    /**
041     * Provider to, given an annotation instance and the type it's on,
042     * create a {@link Processor}.
043     *
044     * @param <A> annotation type
045     * @param <T> handled value type
046     */
047    @FunctionalInterface
048    interface Factory<A extends Annotation, T> {
049
050        /**
051         * Create a new processor given the annotation and data type.
052         *
053         * @param data annotation type on record field
054         * @param value declared field type
055         * @return new processor
056         */
057        Processor<T> make(A data, Type value);
058
059    }
060
061    /**
062     * Apply comments from {@link Comment} annotation on save.
063     *
064     * @return a new processor factory
065     */
066    static Processor.Factory<Comment, Object> comments() {
067        return (data, fieldType) -> (value, destination) -> {
068            if (destination instanceof CommentedConfigurationNodeIntermediary<?>) {
069                final CommentedConfigurationNodeIntermediary<?> commented = (CommentedConfigurationNodeIntermediary<?>) destination;
070                if (data.override()) {
071                    commented.comment(data.value());
072                } else {
073                    commented.commentIfAbsent(data.value());
074                }
075            }
076        };
077    }
078
079    /**
080     * Apply localized comments from {@link Comment} annotation on save.
081     *
082     * <p>The {@link Comment#value() comment's value} will be treated as a key
083     * into {@code source}, resolved to the system default locale. Missing keys
084     * will be written literally to node.</p>
085     *
086     * @param source source bundle for comments
087     * @return a new processor factory
088     */
089    static Processor.Factory<Comment, Object> localizedComments(final ResourceBundle source) {
090        return (data, fieldType) -> {
091            final String translated = Localization.key(source, data.value());
092            return (value, destination) -> {
093                if (destination instanceof CommentedConfigurationNodeIntermediary<?>) {
094                    final CommentedConfigurationNodeIntermediary<?> commented = (CommentedConfigurationNodeIntermediary<?>) destination;
095                    if (data.override()) {
096                        commented.comment(translated);
097                    } else {
098                        commented.commentIfAbsent(translated);
099                    }
100                }
101            };
102        };
103    }
104
105}
