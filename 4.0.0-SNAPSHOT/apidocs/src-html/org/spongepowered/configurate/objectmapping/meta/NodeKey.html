019import java.lang.annotation.Documented;
020import java.lang.annotation.ElementType;
021import java.lang.annotation.Retention;
022import java.lang.annotation.RetentionPolicy;
023import java.lang.annotation.Target;
024
025/**
026 * Marks a field that gets its value from the node's key.
027 *
028 * <p>Fields annotated as such should be treated as <em>read-only</em>.</p>
029 */
030@Retention(RetentionPolicy.RUNTIME)
031@Target(ElementType.FIELD)
032@Documented
033public @interface NodeKey {
034}
