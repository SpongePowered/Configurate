017package org.spongepowered.configurate.objectmapping;
018
019import static io.leangen.geantyref.GenericTypeReflector.box;
020import static io.leangen.geantyref.GenericTypeReflector.erase;
021
022import com.google.auto.value.AutoValue;
023import org.checkerframework.checker.nullness.qual.Nullable;
024import org.spongepowered.configurate.ConfigurationNode;
025import org.spongepowered.configurate.objectmapping.meta.Constraint;
026import org.spongepowered.configurate.objectmapping.meta.NodeResolver;
027import org.spongepowered.configurate.objectmapping.meta.Processor;
028import org.spongepowered.configurate.serialize.SerializationException;
029import org.spongepowered.configurate.serialize.TypeSerializer;
030import org.spongepowered.configurate.util.CheckedFunction;
031import org.spongepowered.configurate.util.UnmodifiableCollections;
032
033import java.lang.reflect.AnnotatedType;
034import java.util.List;
035import java.util.function.Supplier;
036
037/**
038 * Holder for field-specific information.
039 *
040 * @param <I> intermediate type
041 * @param <O> container type
042 */
043@AutoValue
044@SuppressWarnings("AutoValueImmutableFields") // we don't use guava collections
045public abstract class FieldData<I, O> {
046
047    /**
048     * Create a new data holder for a field.
049     *
050     * @param name field name
051     * @param resolvedFieldType type resolved for specific data
052     * @param constraints constraints to verify on this field
053     * @param processors processors to apply to fields backing this node
054     * @param <I> intermediate type
055     * @param <O> container type
056     * @return new field data
057     */
058    static <I, O> FieldData<I, O> of(final String name, final AnnotatedType resolvedFieldType,
059            final List<Constraint<?>> constraints, final List<Processor<?>> processors,
060            final Deserializer<I> deserializer, final CheckedFunction<O, @Nullable Object, Exception> serializer, final NodeResolver resolver) {
061        return new AutoValue_FieldData<>(name, resolvedFieldType,
062                UnmodifiableCollections.copyOf(constraints),
063                UnmodifiableCollections.copyOf(processors),
064                deserializer, serializer, resolver);
065    }
066
067    FieldData() {
068    }
069
070    /**
071     * The name of the field.
072     *
073     * @return field name
074     */
075    public abstract String name();
076
077    /**
078     * The calculated type of this field within the object type.
079     *
080     * <p>This value has had any possible type parameters resolved using
081     * information available in the context.</p>
082     *
083     * @return the resolved type
084     */
085    public abstract AnnotatedType resolvedType();
086
087    abstract List<Constraint<?>> constraints();
088
089    abstract List<Processor<?>> processors();
090
091    abstract Deserializer<I> deserializer();
092
093    abstract CheckedFunction<O, @Nullable Object, Exception> serializer();
094
095    abstract NodeResolver nodeResolver();
096
097    /**
098     * Test if an object would be valid for this field.
099     *
100     * @param instance instance to validate
101     * @return true if valid
102     */
103    public boolean isValid(final Object instance) {
104        try {
105            validate(instance);
106            return true;
107        } catch (final SerializationException ex) {
108            return false;
109        }
110    }
111
112    /**
113     * Try to ensure the provided value is acceptable.
114     *
115     * @param instance field value instance to validate
116     * @throws SerializationException if validation fails
117     */
118    @SuppressWarnings("unchecked")
119    public void validate(final @Nullable Object instance) throws SerializationException {
120        if (instance != null && !erase(box(resolvedType().getType())).isInstance(instance)) {
121            throw new SerializationException("Object " + instance + " is not of expected type " + resolvedType().getType());
122        }
123
124        for (Constraint<?> constraint : constraints()) {
125            ((Constraint<Object>) constraint).validate(instance);
126        }
127    }
128
129    TypeSerializer<?> serializerFrom(final ConfigurationNode node) throws SerializationException {
130        final @Nullable TypeSerializer<?> serial = node.options().serializers().get(resolvedType().getType());
131        if (serial == null) {
132            throw new SerializationException("No TypeSerializer found for field " + name() + " of type " + resolvedType().getType());
133        }
134        return serial;
135    }
136
137    /**
138     * Use this field's node resolvers to determine a target node.
139     *
140     * @param source parent node
141     * @return resolved node
142     */
143    public @Nullable ConfigurationNode resolveNode(final ConfigurationNode source) {
144        return this.nodeResolver().resolve(source);
145    }
146
147    /**
148     * A deserialization handler to appropriately place object data into fields.
149     *
150     * @param <I> intermediate data type
151     */
152    @FunctionalInterface
153    public interface Deserializer<I> {
154
155        /**
156         * Apply either a new value or implicit initializer to the
157         * {@code intermediate} object as appropriate.
158         *
159         * @param intermediate the intermediate container
160         * @param newValue new value to store
161         * @param implicitInitializer the implicit initializer
162         */
163        void accept(I intermediate, @Nullable Object newValue, Supplier<@Nullable Object> implicitInitializer);
164    }
165
166}
