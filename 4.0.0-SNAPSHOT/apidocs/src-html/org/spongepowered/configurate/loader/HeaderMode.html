017package org.spongepowered.configurate.loader;
018
019/**
020 * Modes which {@link ConfigurationLoader}s can use to manipulate headers
021 * when loading/saving.
022 */
023public enum HeaderMode {
024
025    /**
026     * Use the header loaded from an existing file, replacing any header set in
027     * the options.
028     */
029    PRESERVE,
030
031    /**
032     * Ignore any header present in input, and output a header if one has been
033     * set in the options.
034     */
035    PRESET,
036
037    /**
038     * Ignore any header present in input, and do not output any header.
039     */
040    NONE
041
042}
