031import java.io.BufferedWriter;
032import java.io.File;
033import java.io.FileNotFoundException;

035import java.io.InputStreamReader;
036import java.io.Writer;

048/**
049 * Base class for many stream-based configuration loaders. This class provides
050 * conversion from a variety of input sources to {@link BufferedReader}
051 * suppliers, providing a consistent API for loaders to read from and write to.
052 *
053 * <p>Either the source or sink may be null. If this is true, this loader may
054 * not support either loading or saving. In this case, implementing classes are
055 * expected to throw an {@link IOException} for the unsupported operation.</p>
056 *
057 * @param <N> the {@link ConfigurationNode} type produced by the loader
058 */
059public abstract class AbstractConfigurationLoader<N extends ScopedConfigurationNode<N>> implements ConfigurationLoader<N> {
060
061    /**
062     * The escape sequence used by Configurate to separate comment lines.
063     */
064    public static final String CONFIGURATE_LINE_SEPARATOR = "\n";
065
066    /**
067     * A pattern that will match line breaks in comments.
068     */
069    public static final Pattern CONFIGURATE_LINE_PATTERN = Pattern.compile(CONFIGURATE_LINE_SEPARATOR);
070
071    /**
072     * The line separator used by the system.
073     *
074     * @see System#lineSeparator()
075     */
076    protected static final String SYSTEM_LINE_SEPARATOR = System.lineSeparator();
077
078    /**
079     * The reader source for this loader.
080     *
081     * <p>Can be null (for loaders which don't support loading!)</p>
082     */
083    protected final @Nullable Callable<BufferedReader> source;
084
085    /**
086     * The writer sink for this loader.
087     *
088     * <p>Can be null (for loaders which don't support saving!)</p>
089     */
090    protected final @Nullable Callable<BufferedWriter> sink;
091
092    /**
093     * The comment handlers defined for this loader.
094     */
095    private final List<CommentHandler> commentHandlers;
096
097    /**
098     * The mode used to read/write configuration headers.
099     */
100    private final HeaderMode headerMode;
101
102    /**
103     * The default {@link ConfigurationOptions} used by this loader.
104     */
105    private final ConfigurationOptions defaultOptions;
106
107    protected AbstractConfigurationLoader(final Builder<?, ?> builder, final CommentHandler[] commentHandlers) {
108        this.source = builder.source();
109        this.sink = builder.sink();
110        this.headerMode = builder.headerMode();
111        this.commentHandlers = UnmodifiableCollections.toList(commentHandlers);
112        this.defaultOptions = builder.defaultOptions();
113    }
114
115    /**
116     * Gets the primary {@link CommentHandler} used by this loader.
117     *
118     * @return the default comment handler
119     */
120    public CommentHandler defaultCommentHandler() {
121        return this.commentHandlers.get(0);
122    }
123
124    @Override
125    public ConfigurationReference<N> loadToReference() throws ConfigurateException {
126        return ConfigurationReference.fixed(this);
127    }
128
129    @Override
130    public N load(ConfigurationOptions options) throws ParsingException {
131        if (this.source == null) {
132            throw new ParsingException(UNKNOWN_POS, UNKNOWN_POS, "", "No source present to read from!", null);
133        }
134        try (BufferedReader reader = this.source.call()) {
135            if (this.headerMode == HeaderMode.PRESERVE || this.headerMode == HeaderMode.NONE) {
136                final @Nullable String comment = CommentHandlers.extractComment(reader, this.commentHandlers);
137                if (comment != null && comment.length() > 0) {
138                    options = options.header(comment);
139                }
140            }
141            final N node = createNode(options);
142            loadInternal(node, reader);
143            return node;
144        } catch (final ParsingException ex) {
145            throw ex;
146        } catch (final FileNotFoundException | NoSuchFileException e) {
147            // Squash -- there's nothing to read
148            return createNode(options);
149        } catch (final IOException e) {
150            throw new ParsingException(UNKNOWN_POS, UNKNOWN_POS, options.header(), null, e);
151        } catch (final Exception e) {
152            throw new ParsingException(UNKNOWN_POS, UNKNOWN_POS, options.header(), "Unknown error occurred while loading", e);
153        }
154    }
155
156    protected abstract void loadInternal(N node, BufferedReader reader) throws ParsingException;
157
158    @Override
159    public void save(final ConfigurationNode node) throws ConfigurateException {
160        if (this.sink == null) {
161            throw new ConfigurateException(node, "No sink present to write to!");
162        }
163        try (Writer writer = this.sink.call()) {
164            writeHeaderInternal(writer);
165            if (this.headerMode != HeaderMode.NONE) {
166                final @Nullable String header = node.options().header();
167                if (header != null && !header.isEmpty()) {
168                    final Iterator<String> lines = defaultCommentHandler().toComment(CONFIGURATE_LINE_PATTERN.splitAsStream(header)).iterator();
169                    while (lines.hasNext()) {
170                        writer.write(lines.next());
171                        writer.write(SYSTEM_LINE_SEPARATOR);
172                    }
173                    writer.write(SYSTEM_LINE_SEPARATOR);
174                }
175            }
176            saveInternal(node, writer);
177        } catch (final ConfigurateException ex) {
178            throw ex;
179        } catch (final Exception ex) {
180            throw new ConfigurateException(node, ex);
181        }
182    }
183
184    protected void writeHeaderInternal(final Writer writer) throws IOException {}
185
186    protected abstract void saveInternal(ConfigurationNode node, Writer writer) throws ConfigurateException;
187
188    @Override
189    public ConfigurationOptions defaultOptions() {
190        return this.defaultOptions;
191    }
192
193    @Override
194    public final boolean canLoad() {
195        return this.source != null;
196    }
197
198    @Override
199    public final boolean canSave() {
200        return this.sink != null;
201    }
202
203    /**
204     * An abstract builder implementation for {@link AbstractConfigurationLoader}s.
205     *
206     * @param <T> the builder's own type (for chaining using generic types)
207     */
208    public abstract static class Builder<T extends Builder<T, L>, L extends AbstractConfigurationLoader<?>> {
209        protected HeaderMode headerMode = HeaderMode.PRESERVE;
210        protected @Nullable Callable<BufferedReader> source;
211        protected @Nullable Callable<BufferedWriter> sink;
212        protected ConfigurationOptions defaultOptions = ConfigurationOptions.defaults();
213
214        protected Builder() {}
215
216        @SuppressWarnings("unchecked")
217        private T self() {
218            return (T) this;
219        }
220
221        /**
222         * Sets the sink and source of the resultant loader to the given file.
223         *
224         * <p>The {@link #source() source} is defined using
225         * {@link Files#newBufferedReader(Path)} with UTF-8 encoding.</p>
226         *
227         * <p>The {@link #sink() sink} is defined using {@link AtomicFiles} with UTF-8
228         * encoding.</p>
229         *
230         * @param file the configuration file
231         * @return this builder (for chaining)
232         */
233        public T file(final File file) {
234            return path(requireNonNull(file, "file").toPath());
235        }
236
237        /**
238         * Sets the sink and source of the resultant loader to the given path.
239         *
240         * <p>The {@link #source() source} is defined using
241         * {@link Files#newBufferedReader(Path)} with UTF-8 encoding.</p>
242         *
243         * <p>The {@link #sink() sink} is defined using {@link AtomicFiles} with UTF-8
244         * encoding.</p>
245         *
246         * @param path the path of the configuration file
247         * @return this builder (for chaining)
248         */
249        public T path(final Path path) {
250            final Path absPath = requireNonNull(path, "path").toAbsolutePath();
251            this.source = () -> Files.newBufferedReader(absPath, StandardCharsets.UTF_8);
252            this.sink = AtomicFiles.atomicWriterFactory(absPath, StandardCharsets.UTF_8);
253            return self();
254        }
255
256        /**
257         * Sets the source of the resultant loader to the given URL.
258         *
259         * @param url the URL of the source
260         * @return this builder (for chaining)
261         */
262        public T url(final URL url) {
263            requireNonNull(url, "url");
264            this.source = () -> new BufferedReader(new InputStreamReader(url.openConnection().getInputStream(), StandardCharsets.UTF_8));
265            return self();
266        }
267
268        /**
269         * Sets the source of the resultant loader.
270         *
271         * <p>The "source" is used by the loader to load the configuration.</p>
272         *
273         * @param source the source
274         * @return this builder (for chaining)
275         */
276        public T source(final @Nullable Callable<BufferedReader> source) {
277            this.source = source;
278            return self();
279        }
280
281        /**
282         * Gets the source to be used by the resultant loader.
283         *
284         * @return the source
285         */
286        public @Nullable Callable<BufferedReader> source() {
287            return this.source;
288        }
289
290        /**
291         * Sets the sink of the resultant loader.
292         *
293         * <p>The "sink" is used by the loader to save the configuration.</p>
294         *
295         * @param sink the sink
296         * @return this builder (for chaining)
297         */
298        public T sink(final @Nullable Callable<BufferedWriter> sink) {
299            this.sink = sink;
300            return self();
301        }
302
303        /**
304         * Gets the sink to be used by the resultant loader.
305         *
306         * @return the sink
307         */
308        public @Nullable Callable<BufferedWriter> sink() {
309            return this.sink;
310        }
311
312        /**
313         * Sets the header mode of the resultant loader.
314         *
315         * @param mode the header mode
316         * @return this builder (for chaining)
317         */
318        public T headerMode(final HeaderMode mode) {
319            this.headerMode = requireNonNull(mode, "mode");
320            return self();
321        }
322
323        /**
324         * Gets the header mode to be used by the resultant loader.
325         *
326         * @return the header mode
327         */
328        public HeaderMode headerMode() {
329            return this.headerMode;
330        }
331
332        /**
333         * Sets the default configuration options to be used by the
334         * resultant loader.
335         *
336         * @param defaultOptions the options
337         * @return this builder (for chaining)
338         */
339        public T defaultOptions(final ConfigurationOptions defaultOptions) {
340            this.defaultOptions = requireNonNull(defaultOptions, "defaultOptions");
341            return self();
342        }
343
344        /**
345         * Sets the default configuration options to be used by the resultant
346         * loader by providing a function which takes the current default
347         * options and applies any desired changes.
348         *
349         * @param defaultOptions to transform the existing default options
350         * @return this builder (for chaining)
351         */
352        public T defaultOptions(final UnaryOperator<ConfigurationOptions> defaultOptions) {
353            this.defaultOptions = requireNonNull(defaultOptions.apply(this.defaultOptions), "defaultOptions (updated)");
354            return self();
355        }
356
357        /**
358         * Gets the default configuration options to be used by the resultant
359         * loader.
360         *
361         * @return the options
362         */
363        public ConfigurationOptions defaultOptions() {
364            return this.defaultOptions;
365        }
366
367        /**
368         * Builds the loader.
369         *
370         * @return a new loader
371         */
372        public abstract L build();
373
374    }
