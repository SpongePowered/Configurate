017package org.spongepowered.configurate.loader;
018
019import org.checkerframework.checker.nullness.qual.NonNull;
020import org.spongepowered.configurate.ConfigurationNode;
021import org.spongepowered.configurate.ConfigurationOptions;
022import org.spongepowered.configurate.reference.ConfigurationReference;
023
024import java.io.IOException;
025import java.nio.file.Path;
026import java.util.function.Function;
027
028/**
029 * Represents an object which can load and save {@link ConfigurationNode} objects in a specific
030 * configuration format.
031 *
032 * <p>An abstract implementation is provided by {@link AbstractConfigurationLoader}.</p>
033 *
034 * @param <N> The {@link ConfigurationNode} type produced by the loader
035 */
036public interface ConfigurationLoader<N extends ConfigurationNode> {
037
038    /**
039     * Gets the default {@link ConfigurationOptions} used by the loader.
040     *
041     * <p>New nodes will be created using the default options if a specific set is not defined.</p>
042     *
043     * @return The default options
044     */
045    @NonNull
046    ConfigurationOptions getDefaultOptions();
047
048    /**
049     * Attempts to load a {@link ConfigurationNode} using this loader, from the defined source.
050     *
051     * <p>The resultant node represents the root of the configuration being loaded.</p>
052     *
053     * <p>The {@link #getDefaultOptions() default options} will be used to construct the resultant
054     * configuration nodes.</p>
055     *
056     * @return The newly constructed node
057     * @throws IOException if any sort of error occurs with reading or parsing the configuration
058     */
059    @NonNull
060    default N load() throws IOException {
061        return load(getDefaultOptions());
062    }
063
064    /**
065     * Attempts to load data from the defined source into a {@link ConfigurationReference}.
066     * The returned reference will not reload automatically.
067     *
068     * @return The created reference
069     * @throws IOException when an error occurs within the loader
070     * @see org.spongepowered.configurate.reference.WatchServiceListener#listenToConfiguration(Function, Path) to create an auto-reloading configuration.
071     */
072    ConfigurationReference<N> loadToReference() throws IOException;
073
074    /**
075     * Attempts to load a {@link ConfigurationNode} using this loader, from the defined source.
076     *
077     * <p>The resultant node represents the root of the configuration being loaded.</p>
078     *
079     * @param options The options to load with
080     * @return The newly constructed node
081     * @throws IOException if any sort of error occurs with reading or parsing the configuration
082     */
083    @NonNull
084    N load(@NonNull ConfigurationOptions options) throws IOException;
085
086    /**
087     * Attempts to save a {@link ConfigurationNode} using this loader, to the defined sink.
088     *
089     * @param node The node to save
090     * @throws IOException if any sort of error occurs with writing or generating the configuration
091     */
092    void save(@NonNull ConfigurationNode node) throws IOException;
093
094    /**
095     * Return an empty node of the most appropriate type for this loader, using the default options.
096     *
097     * @return The appropriate node type
098     */
099    @NonNull
100    default N createEmptyNode() {
101        return createEmptyNode(getDefaultOptions());
102    }
103
104    /**
105     * Return an empty node of the most appropriate type for this loader
106     *
107     * @param options The options to use with this node. Must not be null (see {@link ConfigurationOptions#defaults()})
108     * @return The appropriate node type
109     */
110    @NonNull
111    N createEmptyNode(@NonNull ConfigurationOptions options);
112
113    /**
114     * Gets if this loader is capable of loading configurations.
115     *
116     * @return If this loader can load
117     */
118    default boolean canLoad() {
119        return true;
120    }
121
122    /**
123     * Gets if this loader is capable of saving configurations.
124     *
125     * @return If this loader can save
126     */
127    default boolean canSave() {
128        return true;
129    }
130}
