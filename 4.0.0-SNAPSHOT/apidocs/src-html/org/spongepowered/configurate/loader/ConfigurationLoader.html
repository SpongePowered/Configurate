017package org.spongepowered.configurate.loader;
018
019import org.spongepowered.configurate.ConfigurateException;
020import org.spongepowered.configurate.ConfigurationNode;
021import org.spongepowered.configurate.ConfigurationNodeFactory;
022import org.spongepowered.configurate.ConfigurationOptions;
023import org.spongepowered.configurate.reference.ConfigurationReference;
024import org.spongepowered.configurate.reference.WatchServiceListener;
025
026import java.nio.file.Path;
027import java.util.function.Function;
028
029/**
030 * Represents an object which can load and save {@link ConfigurationNode} objects in a specific
031 * configuration format.
032 *
033 * <p>An abstract implementation is provided by {@link AbstractConfigurationLoader}.</p>
034 *
035 * @param <N> the {@link ConfigurationNode} type produced by the loader
036 */
037public interface ConfigurationLoader<N extends ConfigurationNode> extends ConfigurationNodeFactory<N> {
038
039    /**
040     * Attempts to load a {@link ConfigurationNode} using this loader, from the
041     * defined source.
042     *
043     * <p>The resultant node represents the root of the configuration being
044     * loaded.</p>
045     *
046     * <p>The {@link #defaultOptions() default options} will be used to
047     * construct the resultant configuration nodes.</p>
048     *
049     * @return the newly constructed node
050     * @throws ConfigurateException if any sort of error occurs with reading or
051     *                              parsing the configuration
052     */
053    default N load() throws ConfigurateException {
054        return load(defaultOptions());
055    }
056
057    /**
058     * Attempts to load a {@link ConfigurationNode} using this loader, from the defined source.
059     *
060     * <p>The resultant node represents the root of the configuration being
061     * loaded.</p>
062     *
063     * @param options the options to load with
064     * @return the newly constructed node
065     * @throws ConfigurateException if any sort of error occurs with reading or
066     *                              parsing the configuration
067     */
068    N load(ConfigurationOptions options) throws ConfigurateException;
069
070    /**
071     * Attempts to load data from the defined source into a {@link ConfigurationReference}.
072     * The returned reference will not reload automatically.
073     *
074     * @return the created reference
075     * @throws ConfigurateException when an error occurs within the loader
076     * @see WatchServiceListener#listenToConfiguration(Function, Path) to
077     *      create an auto-reloading configuration.
078     */
079    ConfigurationReference<N> loadToReference() throws ConfigurateException;
080
081    /**
082     * Attempts to save a {@link ConfigurationNode} using this loader, to the defined sink.
083     *
084     * @param node the node to save
085     * @throws ConfigurateException if any sort of error occurs with writing or
086     *                     generating the configuration
087     */
088    void save(ConfigurationNode node) throws ConfigurateException;
089
090    /**
091     * Gets if this loader is capable of loading configurations.
092     *
093     * @return if this loader can load
094     */
095    default boolean canLoad() {
096        return true;
097    }
098
099    /**
100     * Gets if this loader is capable of saving configurations.
101     *
102     * @return if this loader can save
103     */
104    default boolean canSave() {
105        return true;
106    }
107
108}
