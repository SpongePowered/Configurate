017package org.spongepowered.configurate.loader;
018
019import com.google.errorprone.annotations.Immutable;
020import org.checkerframework.checker.nullness.qual.Nullable;
021
022import java.io.BufferedReader;
023import java.io.IOException;
024import java.nio.CharBuffer;
025import java.util.Iterator;
026import java.util.stream.Stream;
027
028/**
029 * Defines a number of default {@link CommentHandler}s.
030 */
031@Immutable
032public enum CommentHandlers implements CommentHandler {
033
034    /**
035     * {@link CommentHandler} for comments prefixed by the <code>#</code> character.
036     */
037    HASH(new AbstractPrefixHandler("#")),
038
039    /**
040     * {@link CommentHandler} for comments prefixed by a <code>//</code> escape.
041     */
042    DOUBLE_SLASH(new AbstractPrefixHandler("//")),
043
044    /**
045     * {@link CommentHandler} for comments delineated using <code>/*  *\</code>.
046     */
047    SLASH_BLOCK(new AbstractDelineatedHandler("/*", "*/", "*")),
048
049    /**
050     * {@link CommentHandler} for comments delineated using <code>&lt;!--  --&gt;</code>.
051     */
052    XML_STYLE(new AbstractDelineatedHandler("<!--", "-->", "~"));
053
054    /**
055     * Limit on the number of characters that may be read by a comment handler
056     * while still preserving the mark.
057     */
058    private static final int READAHEAD_LEN = 4096;
059
060    private final CommentHandler delegate;
061
062    CommentHandlers(final CommentHandler delegate) {
063        this.delegate = delegate;
064    }
065
066    @Override
067    public @Nullable String extractHeader(final BufferedReader reader) throws IOException {
068        return this.delegate.extractHeader(reader);
069    }
070
071    @Override
072    public Stream<String> toComment(final Stream<String> lines) {
073        return this.delegate.toComment(lines);
074    }
075
076    /**
077     * Uses provided comment handlers to extract a comment from the reader.
078     *
079     * @param reader to extract a comment from
080     * @param allowedHeaderTypes handlers to try
081     * @return extracted comment, or null if a comment could not be extracted
082     * @throws IOException if an IO error occurs
083     */
084    public static @Nullable String extractComment(final BufferedReader reader,
085            final Iterable<CommentHandler> allowedHeaderTypes) throws IOException {
086        reader.mark(READAHEAD_LEN);
087        for (CommentHandler handler : allowedHeaderTypes) {
088            final @Nullable String comment = handler.extractHeader(reader);
089            if (comment == null) {
090                reader.reset();
091            } else {
092                return comment;
093            }
094        }
095        return null;
096    }
097
098    @Immutable
099    private static final class AbstractDelineatedHandler implements CommentHandler {
100        private final String startSequence;
101        private final String endSequence;
102        private final String lineIndentSequence;
103
104        private AbstractDelineatedHandler(final String startSequence, final String endSequence, final String lineIndentSequence) {
105            this.startSequence = startSequence;
106            this.endSequence = endSequence;
107            this.lineIndentSequence = lineIndentSequence;
108        }
109
110        @Override
111        public @Nullable String extractHeader(final BufferedReader reader) throws IOException {
112            if (!beginsWithPrefix(this.startSequence, reader)) {
113                return null;
114            }
115
116            final StringBuilder build = new StringBuilder();
117            String line = reader.readLine();
118            if (line == null) {
119                return null;
120            }
121            if (handleSingleLine(build, line)) {
122                for (line = reader.readLine(); line != null; line = reader.readLine()) {
123                    if (!handleSingleLine(build, line)) {
124                        break;
125                    }
126                }
127            }
128            line = reader.readLine();
129            if (!(line == null || line.trim().isEmpty())) { // Require a blank line after a comment to make it a header
130                return null;
131            }
132
133            if (build.length() > 0) {
134                return build.toString();
135            } else {
136                return null;
137            }
138        }
139
140        private boolean handleSingleLine(final StringBuilder builder, String line) {
141            boolean moreLines = true;
142            if (line.trim().endsWith(this.endSequence)) {
143                line = line.substring(0, line.lastIndexOf(this.endSequence));
144                if (line.endsWith(" ")) {
145                    line = line.substring(0, line.length() - 1);
146                }
147
148                moreLines = false;
149                if (line.isEmpty()) {
150                    return false;
151                }
152            }
153            if (line.trim().startsWith(this.lineIndentSequence)) {
154                line = line.substring(line.indexOf(this.lineIndentSequence) + 1);
155            }
156
157            if (line.startsWith(" ")) {
158                line = line.substring(1);
159            }
160
161            if (builder.length() > 0) {
162                builder.append(AbstractConfigurationLoader.CONFIGURATE_LINE_SEPARATOR);
163            }
164            builder.append(line.replace("\r", "").replace("\n", "").replace("\r\n", ""));
165            return moreLines;
166        }
167
168        @Override
169        public Stream<String> toComment(final Stream<String> lines) {
170            final Stream.Builder<String> build = Stream.builder();
171            boolean first = true;
172            for (Iterator<String> it = lines.iterator(); it.hasNext();) {
173                final String next = it.next();
174                if (first) {
175                    if (!it.hasNext()) {
176                        build.add(this.startSequence + " " + next + " " + this.endSequence);
177                        return build.build();
178                    } else {
179                        build.add(this.startSequence);
180                    }
181                    first = false;
182                }
183                build.add(" " + this.lineIndentSequence + " " + next);
184            }
185            build.add(" " + this.endSequence);
186            return build.build();
187        }
188    }
189
190    @Immutable
191    private static final class AbstractPrefixHandler implements CommentHandler {
192        private final String commentPrefix;
193
194        AbstractPrefixHandler(final String commentPrefix) {
195            this.commentPrefix = commentPrefix;
196        }
197
198        @Override
199        public @Nullable String extractHeader(final BufferedReader reader) throws IOException {
200            if (!beginsWithPrefix(this.commentPrefix, reader)) {
201                return null;
202            }
203            boolean firstLine = true;
204
205            final StringBuilder build = new StringBuilder();
206            for (String line = reader.readLine(); line != null; line = reader.readLine()) {
207                if (firstLine) {
208                    if (line.startsWith(" ")) {
209                        line = line.substring(1);
210                    }
211                    build.append(line);
212                    firstLine = false;
213                } else if (line.trim().startsWith(this.commentPrefix)) {
214                    line = line.substring(line.indexOf(this.commentPrefix) + 1);
215                    if (line.startsWith(" ")) {
216                        line = line.substring(1);
217                    }
218                    if (build.length() > 0) {
219                        build.append(AbstractConfigurationLoader.CONFIGURATE_LINE_SEPARATOR);
220                    }
221                    build.append(line);
222                } else if (line.trim().isEmpty()) {
223                    break;
224                } else {
225                    return null;
226                }
227            }
228            // We've reached the end of the document?
229            return build.length() > 0 ? build.toString() : null;
230        }
231
232        @Override
233        public Stream<String> toComment(final Stream<String> lines) {
234            return lines
235                    .map(s -> {
236                        if (s.startsWith(" ")) {
237                            return this.commentPrefix + s;
238                        } else {
239                            return this.commentPrefix + " " + s;
240                        }
241                    });
242        }
243    }
244
245    /**
246     * Consumes the length of the comment prefix from the reader and returns
247     * whether or not the contents from the reader matches the expected prefix.
248     */
249    static boolean beginsWithPrefix(final String commentPrefix, final BufferedReader reader) throws IOException {
250        final CharBuffer buf = CharBuffer.allocate(commentPrefix.length());
251        if (reader.read(buf) != buf.limit()) {
252            return false;
253        }
254        buf.flip();
255        return commentPrefix.contentEquals(buf);
256    }
