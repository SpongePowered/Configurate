017package org.spongepowered.configurate.loader;
018
019import com.google.errorprone.annotations.Immutable;
020import org.checkerframework.checker.nullness.qual.Nullable;
021
022import java.io.BufferedReader;
023import java.io.IOException;
024import java.util.stream.Stream;
025
026/**
027 * Extracts comments from a buffered reader or collection of lines.
028 */
029@Immutable
030public interface CommentHandler {
031
032    /**
033     * Defines the handlers behaviour for reading comments.
034     *
035     * @param reader reader to get input from
036     * @return an extracted comment, if any
037     * @throws IOException if any IO error occurs in the process
038     */
039    @Nullable String extractHeader(BufferedReader reader) throws IOException;
040
041    /**
042     * Converts the specified lines into a comment.
043     *
044     * @param lines lines to make a comment
045     * @return transformed lines as a stream
046     */
047    Stream<String> toComment(Stream<String> lines);
048
049}
