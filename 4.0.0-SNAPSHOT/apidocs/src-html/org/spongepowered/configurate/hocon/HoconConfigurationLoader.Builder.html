055/**
056 * A loader for HOCON (Hodor)-formatted configurations, using the typesafe
057 * config library for parsing and generation.
058 */
059public final class HoconConfigurationLoader extends AbstractConfigurationLoader<CommentedConfigurationNode> {
060
061    private static final Set<Class<?>> NATIVE_TYPES = UnmodifiableCollections.toSet(
062            Double.class, Long.class, Integer.class, Boolean.class, String.class, Number.class);
063
064    /**
065     * The default render options used by configurate.
066     */
067    private static final ConfigRenderOptions DEFAULT_RENDER_OPTIONS = ConfigRenderOptions.defaults()
068            .setOriginComments(false)
069            .setJson(false);
070
071    /**
072     * An instance of {@link ConfigOrigin} for configurate.
073     */
074    private static final ConfigOrigin CONFIGURATE_ORIGIN = ConfigOriginFactory.newSimple("configurate-hocon");
075
076    /**
077     * Creates a new {@link HoconConfigurationLoader} builder.
078     *
079     * @return a new builder
080     */
081    @NonNull
082    public static Builder builder() {
083        return new Builder();
084    }
085
086    /**
087     * Builds a {@link HoconConfigurationLoader}.
088     */
089    public static final class Builder extends AbstractConfigurationLoader.Builder<Builder, HoconConfigurationLoader> {
090        private ConfigRenderOptions render = DEFAULT_RENDER_OPTIONS;
091
092        Builder() {
093        }
094
095        /**
096         * Set whether output from this loader will be pretty-printed or not.
097         *
098         * <p>Output will always print with a fixed indent of 4 spaces per
099         * level. This is a limitation of the underlying library, so it may
100         * become customizable at some point in the future.</p>
101         *
102         * @param prettyPrinting whether to pretty-print
103         * @return this builder
104         */
105        public Builder prettyPrinting(final boolean prettyPrinting) {
106            this.render = this.render.setFormatted(prettyPrinting);
107            return this;
108        }
109
110        /**
111         * Set whether comments should be emmitted.
112         *
113         * <p>Comments will always be loaded from files and
114         * stored in memory.</p>
115         *
116         * @param emitComments whether to emit comments
117         * @return this builder
118         */
119        public Builder emitComments(final boolean emitComments) {
120            this.render = this.render.setComments(emitComments);
121            return this;
122        }
123
124        /**
125         * Set whether output generated by this loader should be
126         * json-compatible.
127         *
128         * <p>Whatever format input is received in, this will output
129         * JSON. To be fully spec-compliant, comment output must also
130         * be disabled.</p>
131         *
132         * @param jsonCompatible to emit json-format output
133         * @return this builder
134         */
135        public Builder emitJsonCompatible(final boolean jsonCompatible) {
136            this.render = this.render.setJson(jsonCompatible);
137            return this;
138        }
139
140        @NonNull
141        @Override
142        public HoconConfigurationLoader build() {
143            defaultOptions(o -> o.nativeTypes(NATIVE_TYPES));
144            return new HoconConfigurationLoader(this);
145        }
146    }
147
148    private final ConfigRenderOptions render;
149
150    private HoconConfigurationLoader(final Builder build) {
151        super(build, new CommentHandler[] {CommentHandlers.HASH, CommentHandlers.DOUBLE_SLASH});
152        this.render = build.render;
153    }
154
155    @Override
156    protected void loadInternal(final CommentedConfigurationNode node, final BufferedReader reader) throws ParsingException {
157        Config hoconConfig;
158        try {
159            hoconConfig = ConfigFactory.parseReader(reader);
160            hoconConfig = hoconConfig.resolve();
161        } catch (final ConfigException ex) {
162            throw new ParsingException(node, ex.origin().lineNumber(), 0, ex.origin().description(), null, ex);
163        }
164
165        for (Map.Entry<String, ConfigValue> ent : hoconConfig.root().entrySet()) {
166            readConfigValue(ent.getValue(), node.node(ent.getKey()));
167        }
168    }
169
170    private static void readConfigValue(final ConfigValue value, final CommentedConfigurationNode node) {
171        if (!value.origin().comments().isEmpty()) {
172            node.comment(value.origin().comments().stream()
173                    .map(input -> input.replace("\r", ""))
174                    .collect(Collectors.joining("\n")));
175        }
176
177        switch (value.valueType()) {
178            case OBJECT:
179                final ConfigObject object = (ConfigObject) value;
180                if (object.isEmpty()) {
181                    node.raw(Collections.emptyMap());
182                } else {
183                    for (Map.Entry<String, ConfigValue> ent : object.entrySet()) {
184                        readConfigValue(ent.getValue(), node.node(ent.getKey()));
185                    }
186                }
187                break;
188            case LIST:
189                final ConfigList list = (ConfigList) value;
190                if (list.isEmpty()) {
191                    node.raw(Collections.emptyList());
192                } else {
193                    for (int i = 0; i < list.size(); ++i) {
194                        readConfigValue(list.get(i), node.node(i));
195                    }
196                }
197                break;
198            case NULL:
199                return;
200            default:
201                node.raw(value.unwrapped());
202        }
203    }
204
205    @Override
206    protected void saveInternal(final ConfigurationNode node, final Writer writer) throws ConfigurateException {
207        try {
208            if (!node.isMap()) {
209                if (node.virtual() || node.raw() == null) {
210                    writer.write(SYSTEM_LINE_SEPARATOR);
211                    return;
212                } else {
213                    throw new ConfigurateException(node, "HOCON can only write nodes that are in map format!");
214                }
215            }
216            final ConfigValue value = fromValue(node);
217            final String renderedValue = value.render(this.render);
218            writer.write(renderedValue);
219        } catch (final IOException io) {
220            throw new ConfigurateException(node, io);
221        }
222    }
223
224    private static ConfigValue fromValue(final ConfigurationNode node) {
225        ConfigValue ret;
226        if (node.isMap()) {
227            final Map<String, ConfigValue> children = node.options().mapFactory().create();
228            for (Map.Entry<Object, ? extends ConfigurationNode> ent : node.childrenMap().entrySet()) {
229                children.put(String.valueOf(ent.getKey()), fromValue(ent.getValue()));
230            }
231            ret = newConfigObject(children);
232        } else if (node.isList()) {
233            final List<ConfigValue> children = new ArrayList<>();
234            for (ConfigurationNode ent : node.childrenList()) {
235                children.add(fromValue(ent));
236            }
237            ret = newConfigList(children);
238
239        } else {
240            ret = ConfigValueFactory.fromAnyRef(node.rawScalar(), CONFIGURATE_ORIGIN.description());
241        }
242        if (node instanceof CommentedConfigurationNodeIntermediary<?>) {
243            final CommentedConfigurationNodeIntermediary<?> commentedNode = (CommentedConfigurationNodeIntermediary<?>) node;
244            final @Nullable String origComment = commentedNode.comment();
245            if (origComment != null) {
246                ret = ret.withOrigin(ret.origin().withComments(Arrays.asList(CONFIGURATE_LINE_PATTERN.split(origComment))));
247            }
248        }
249        return ret;
250    }
251
252    static ConfigValue newConfigObject(final Map<String, ConfigValue> vals) {
253        try {
254            return CONFIG_OBJECT_CONSTRUCTOR.newInstance(CONFIGURATE_ORIGIN, vals);
255        } catch (InstantiationException | IllegalAccessException | InvocationTargetException e) {
256            throw new RuntimeException(e); // rethrow
257        }
258
259    }
260
261    static ConfigValue newConfigList(final List<ConfigValue> vals) {
262        try {
263            return CONFIG_LIST_CONSTRUCTOR.newInstance(CONFIGURATE_ORIGIN, vals);
264        } catch (InstantiationException | IllegalAccessException | InvocationTargetException e) {
265            throw new RuntimeException(e); // rethrow
266        }
267    }
268
269    @NonNull
270    @Override
271    public CommentedConfigurationNode createNode(@NonNull ConfigurationOptions options) {
272        options = options.nativeTypes(NATIVE_TYPES);
273        return CommentedConfigurationNode.root(options);
274    }
275
276    // -- Comment handling -- this might have to be updated as the hocon dep changes
277    // (But tests should detect this breakage)
278    private static final Constructor<? extends ConfigValue> CONFIG_OBJECT_CONSTRUCTOR;
279    private static final Constructor<? extends ConfigValue> CONFIG_LIST_CONSTRUCTOR;
280
281    static {
282        final Class<? extends ConfigValue> objectClass;
283        final Class<? extends ConfigValue> listClass;
284        try {
285            objectClass = Class.forName("com.typesafe.config.impl.SimpleConfigObject").asSubclass(ConfigValue.class);
286            listClass = Class.forName("com.typesafe.config.impl.SimpleConfigList").asSubclass(ConfigValue.class);
287        } catch (final ClassNotFoundException e) {
288            throw new ExceptionInInitializerError(e);
289        }
290
291        try {
292            CONFIG_OBJECT_CONSTRUCTOR = objectClass.getDeclaredConstructor(ConfigOrigin.class, Map.class);
293            CONFIG_OBJECT_CONSTRUCTOR.setAccessible(true);
294            CONFIG_LIST_CONSTRUCTOR = listClass.getDeclaredConstructor(ConfigOrigin.class, List.class);
295            CONFIG_LIST_CONSTRUCTOR.setAccessible(true);
296        } catch (final NoSuchMethodException e) {
297            throw new ExceptionInInitializerError(e);
298        }
299    }
