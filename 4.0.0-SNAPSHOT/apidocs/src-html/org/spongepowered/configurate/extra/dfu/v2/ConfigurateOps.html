017package org.spongepowered.configurate.extra.dfu.v2;
018
019import static java.util.Objects.requireNonNull;
020
021import com.google.common.collect.ImmutableMap;
022import com.mojang.datafixers.DSL;
023import com.mojang.datafixers.Dynamic;
024import com.mojang.datafixers.types.DynamicOps;
025import com.mojang.datafixers.types.Type;
026import org.checkerframework.checker.nullness.qual.Nullable;
027import org.spongepowered.configurate.CommentedConfigurationNode;
028import org.spongepowered.configurate.ConfigurationNode;
029import org.spongepowered.configurate.ConfigurationOptions;
030import org.spongepowered.configurate.serialize.TypeSerializerCollection;
031
032import java.util.Map;
033import java.util.Optional;
034import java.util.function.Function;
035import java.util.function.Supplier;
036import java.util.stream.Stream;
037
038/**
039 * Implementation of DataFixerUpper's DynamicOps.
040 *
041 * <p>When possible, the first node's {@link ConfigurationNode#copy()}  method
042 * will be used to create a new node to contain results. Otherwise, the provided
043 * factory will be used. The default factory creates a
044 * {@link CommentedConfigurationNode} with the {@link TypeSerializerCollection#defaults() default TypeSerializer collection},
045 * but a custom factory may be provided.
046 *
047 */
048public final class ConfigurateOps implements DynamicOps<ConfigurationNode> {
049
050    private static final ConfigurateOps INSTANCE = new ConfigurateOps(CommentedConfigurationNode::root);
051
052    private final Supplier<? extends ConfigurationNode> factory;
053
054    /**
055     * Get the shared instance of this class, which creates new nodes using
056     * the default factory.
057     *
058     * @return the shared instance
059     */
060    public static DynamicOps<ConfigurationNode> instance() {
061        return INSTANCE;
062    }
063
064    /**
065     * Create a new instance of the ops, with a custom node factory.
066     *
067     * @param factory the factory function
068     * @return a new ops instance
069     */
070    public static DynamicOps<ConfigurationNode> withNodeFactory(final Supplier<? extends ConfigurationNode> factory) {
071        return new ConfigurateOps(factory);
072    }
073
074    /**
075     * Wrap a ConfigurationNode in a {@link Dynamic} instance. The returned Dynamic will use the same type
076     * serializer collection as the original node for its operations.
077     *
078     * @param node the node to wrap
079     * @return a wrapped node
080     */
081    public static Dynamic<ConfigurationNode> wrap(final ConfigurationNode node) {
082        if (node.options().serializers().equals(TypeSerializerCollection.defaults())) {
083            return new Dynamic<>(instance(), node);
084        } else {
085            final ConfigurationOptions opts = node.options();
086            return new Dynamic<>(withNodeFactory(() -> CommentedConfigurationNode.root(opts)), node);
087        }
088    }
089
090    ConfigurateOps(final Supplier<? extends ConfigurationNode> factory) {
091        this.factory = factory;
092    }
093
094    private static String unwrapKey(final ConfigurationNode node) {
095        return requireNonNull(node.getString(), "Kep nodes must have a value!");
096    }
097
098    @Override
099    public ConfigurationNode empty() {
100        return this.factory.get();
101    }
102
103    @Override
104    public Type<?> getType(final ConfigurationNode input) {
105        if (input == null) {
106            throw new NullPointerException("input is null");
107        }
108
109        if (input.isMap()) {
110            return DSL.compoundList(DSL.remainderType(), DSL.remainderType());
111
112        } else if (input.isList()) {
113            return DSL.list(DSL.remainderType());
114        } else {
115            final @Nullable Object value = input.rawScalar();
116            if (value == null) {
117                return DSL.nilType();
118            } else if (value instanceof String) {
119                return DSL.string();
120            } else if (value instanceof Boolean) {
121                return DSL.bool();
122            } else if (value instanceof Short) {
123                return DSL.shortType();
124            } else if (value instanceof Integer) {
125                return DSL.intType();
126            } else if (value instanceof Long) {
127                return DSL.longType();
128            } else if (value instanceof Float) {
129                return DSL.floatType();
130            } else if (value instanceof Double) {
131                return DSL.doubleType();
132            } else if (value instanceof Byte) {
133                return DSL.byteType();
134            } else {
135                throw new IllegalArgumentException("Scalar value '" + input + "' has an unknown type: " + value.getClass().getName());
136            }
137        }
138    }
139
140    @Override
141    public Optional<Number> getNumberValue(final ConfigurationNode input) {
142        if (!(input.isMap() || input.isList())) {
143            final @Nullable Object raw = input.rawScalar();
144            if (raw instanceof Number) {
145                return Optional.of((Number) raw);
146            } else if (raw instanceof Boolean) {
147                return Optional.of(input.getBoolean() ? 1 : 0);
148            }
149        }
150
151        return Optional.empty();
152    }
153
154    @Override
155    public ConfigurationNode createNumeric(final Number i) {
156        return empty().raw(i);
157    }
158
159    @Override
160    public ConfigurationNode createBoolean(final boolean value) {
161        return empty().raw(value);
162    }
163
164    @Override
165    public Optional<String> getStringValue(final ConfigurationNode input) {
166        return Optional.ofNullable(input.getString());
167    }
168
169    @Override
170    public ConfigurationNode createString(final String value) {
171        return empty().raw(value);
172    }
173
174    @Override
175    public ConfigurationNode mergeInto(final ConfigurationNode input, final ConfigurationNode value) {
176        if (input.isList()) {
177            final ConfigurationNode ret = input.copy();
178            ret.appendListNode().from(value);
179            return ret;
180        }
181        return input;
182    }
183
184    @Override
185    public ConfigurationNode mergeInto(final ConfigurationNode input, final ConfigurationNode key, final ConfigurationNode value) {
186        return input.copy().node(unwrapKey(key)).from(value);
187    }
188
189    /**
190     * Merge into a newly created node.
191     *
192     * @param first the primary node
193     * @param second the second node, with values that will override those in
194     *               the first node
195     * @return a newly created node
196     */
197    @Override
198    public ConfigurationNode merge(final ConfigurationNode first, final ConfigurationNode second) {
199        return first.copy().mergeFrom(second);
200
201    }
202
203    @Override
204    public Optional<Map<ConfigurationNode, ConfigurationNode>> getMapValues(final ConfigurationNode input) {
205        if (input.isMap()) {
206            final ImmutableMap.Builder<ConfigurationNode, ConfigurationNode> builder = ImmutableMap.builder();
207            for (final Map.Entry<Object, ? extends ConfigurationNode> entry : input.childrenMap().entrySet()) {
208                builder.put(empty().raw(entry.getKey()), entry.getValue().copy());
209            }
210            return Optional.of(builder.build());
211        }
212
213        return Optional.empty();
214    }
215
216    @Override
217    public ConfigurationNode createMap(final Map<ConfigurationNode, ConfigurationNode> map) {
218        final ConfigurationNode ret = empty();
219
220        for (final Map.Entry<ConfigurationNode, ConfigurationNode> entry : map.entrySet()) {
221            ret.node(unwrapKey(entry.getKey())).from(entry.getValue());
222        }
223
224        return ret;
225    }
226
227    @Override
228    public Optional<Stream<ConfigurationNode>> getStream(final ConfigurationNode input) {
229        if (input.isList()) {
230            final Stream<ConfigurationNode> stream = input.childrenList().stream().map(it -> it);
231            return Optional.of(stream);
232        }
233
234        return Optional.empty();
235    }
236
237    @Override
238    public ConfigurationNode createList(final Stream<ConfigurationNode> input) {
239        final ConfigurationNode ret = empty();
240        input.forEach(it -> ret.appendListNode().from(it));
241        return ret;
242    }
243
244    @Override
245    public ConfigurationNode remove(final ConfigurationNode input, final String key) {
246        if (input.isMap()) {
247            final ConfigurationNode ret = input.copy();
248            ret.node(key).raw(null);
249            return ret;
250        }
251
252        return input;
253    }
254
255    @Override
256    public Optional<ConfigurationNode> get(final ConfigurationNode input, final String key) {
257        final ConfigurationNode ret = input.node(key);
258        return ret.virtual() ? Optional.empty() : Optional.of(ret);
259    }
260
261    @Override
262    public Optional<ConfigurationNode> getGeneric(final ConfigurationNode input, final ConfigurationNode key) {
263        final ConfigurationNode ret = input.node(unwrapKey(key));
264        return ret.virtual() ? Optional.empty() : Optional.of(ret);
265    }
266
267    @Override
268    public ConfigurationNode set(final ConfigurationNode input, final String key, final ConfigurationNode value) {
269        final ConfigurationNode ret = input.copy();
270        ret.node(key).from(value);
271        return ret;
272    }
273
274    @Override
275    public ConfigurationNode update(final ConfigurationNode input, final String key, final Function<ConfigurationNode, ConfigurationNode> function) {
276        if (input.node(key).virtual()) {
277            return input;
278        }
279
280        final ConfigurationNode ret = input.copy();
281
282        final ConfigurationNode child = ret.node(key);
283        child.from(function.apply(child));
284        return ret;
285    }
286
287    @Override
288    public ConfigurationNode updateGeneric(final ConfigurationNode input, final ConfigurationNode wrappedKey,
289                                           final Function<ConfigurationNode, ConfigurationNode> function) {
290        final Object key = unwrapKey(wrappedKey);
291        if (input.node(key).virtual()) {
292            return input;
293        }
294
295        final ConfigurationNode ret = input.copy();
296
297        final ConfigurationNode child = ret.node(key);
298        child.from(function.apply(child));
299        return ret;
300    }
301
302    @Override
303    public String toString() {
304        return "Configurate";
305    }
