044/**
045 * Implementation of DataFixerUpper's DynamicOps.
046 *
047 * <p>The {@link DynamicOps} interface should be thought of essentially as a way
048 * to perform operations on a type without having to directly implement that
049 * interface on the type. Rather than taking an object that implements an
050 * interface, DFU methods take the implementation of the DynamicOps interface
051 * plus the type implemented onto.
052 *
053 * <p>When possible, the first node's {@link ConfigurationNode#copy()} method
054 * will be used to create a new node to contain results. Otherwise, the provided
055 * factory will be used. The default factory creates a
056 * {@link CommentedConfigurationNode} with the default serializer collection
057 * but a custom factory may be provided.
058 *
059 * <p>DynamicOps has the following primitive types (as determined by those
060 * codecs that implement {@link com.mojang.serialization.codecs.PrimitiveCodec}):
061 * <dl>
062 *     <dt>boolean</dt>
063 *     <dd>literal boolean, or numeric 1 for true, 0 for false</dd>
064 *     <dt>byte</dt>
065 *     <dd>numeric value, {@link Number#byteValue() coerced} to a byte.
066 *     If {@link #compressMaps()}, a string may be parsed as a byte as well.</dd>
067 *     <dt>short</dt>
068 *     <dd>numeric value, {@link Number#shortValue() coerced} to a short.
069 *     If {@link #compressMaps()}, a string may be parsed as a short as well.</dd>
070 *     <dt>int</dt>
071 *     <dd>numeric value, {@link Number#intValue() coerced} to an integer.
072 *     If {@link #compressMaps()}, a string may be parsed as an integer as well.</dd>
073 *     <dt>long</dt>
074 *     <dd>numeric value, {@link Number#longValue() coerced} to a long.
075 *     If {@link #compressMaps()}, a string may be parsed as a long as well.</dd>
076 *     <dt>float</dt>
077 *     <dd>numeric value, {@link Number#floatValue() coerced} to a float.
078 *     If {@link #compressMaps()}, a string may be parsed as a float as well.</dd>
079 *     <dt>double</dt>
080 *     <dd>numeric value, {@link Number#doubleValue() coerced} to a double.
081 *     If {@link #compressMaps()}, a string may be parsed as a double as well.</dd>
082 *     <dt>{@link String}</dt>
083 *     <dd>Any scalar value, as {@link Object#toString() a string}</dd>
084 *     <dt>{@link java.nio.ByteBuffer}</dt>
085 *     <dd>An array of bytes. Either a native byte array in the node,
086 *         or (by default impl) a list of bytes</dd>
087 *     <dt>{@link java.util.stream.IntStream}</dt>
088 *     <dd>A sequence of integers. Either a native int array in the node,
089 *         or (by default impl) a list of integers</dd>
090 *     <dt>{@link java.util.stream.LongStream}</dt>
091 *     <dd>A sequence of longs. Either a native long array in the node,
092 *         or (by default impl) a list of longs</dd>
093 * </dl>
094 */

187    /**
188     * Whether data passed through this ops will be compressed or not.
189     *
190     * <p>In the context of DFU, <pre>compressed</pre> means that in situations
191     * where values are of a {@link com.mojang.serialization.Keyable} type
192     * (as is with types like Minecraft Registries)
193     * rather than fully encoding each value, its index into the container
194     * is encoded.
195     *
196     * <p>While data encoded this way may take less space to store, the
197     * compressed data will also require an explicit mapping of indices to
198     * values. If this is not stored with the node, the indices of values must
199     * be preserved to correctly deserialize compressed values.
200     *
201     * <p>For example, for an enum new values could only be appended, not added
202     * in the middle of the constants.
203     *
204     * @return whether maps are compressed
205     */
206    @Override
207    public boolean compressMaps() {
208        return this.compressed;
209    }

211    /**
212     * Extract a value from a node used as a {@code key} in DFU methods.
213     *
214     * <p>This currently only attempts to interpret the key as a single level
215     * down. However, we may want to try to extract an array or iterable of path
216     * elements to be able to traverse multiple levels.
217     *
218     * @param node data source
219     * @return a key, asserted non-null
220     */
221    static Object keyFrom(final ConfigurationNode node) {
222        if (node.isList() || node.isMap()) {
223            throw new IllegalArgumentException("Key nodes must have scalar values");
224        }
225        return requireNonNull(node.raw(), "The provided key node must have a value");
226    }

228    /**
229     * Guard source node according to ops instance's copying policy.
230     *
231     * @implNote currently, this will make a deep copy of the node.
232     *
233     * @param untrusted original node
234     * @return a node with equivalent data
235     */
236    ConfigurationNode guardOutputRead(final ConfigurationNode untrusted) {
237        switch (this.readProtection) {
238            case COPY_DEEP: return untrusted.copy();
239            case NONE: return untrusted;
240            default: throw new IllegalArgumentException("Unexpected state");
241        }
242    }
243
244    ConfigurationNode guardInputWrite(final ConfigurationNode untrusted) {
245        switch (this.writeProtection) {
246            case COPY_DEEP: return untrusted.copy();
247            case NONE: return untrusted;
248            default: throw new IllegalArgumentException("Unexpected state");
249        }
250    }

272    // If the destination ops is another Configurate ops instance, just directly pass the node through
273    @SuppressWarnings("unchecked")
274    private <U> @Nullable U convertSelf(final DynamicOps<U> outOps, final ConfigurationNode input) {
275        if (outOps instanceof ConfigurateOps) {
276            return (U) input;
277        } else {
278            return null;
279        }
280    }

282    /**
283     * Create a copy of the source node converted to a different data structure.
284     *
285     * <p>Value types will be preserved as much as possible, but a reverse
286     * conversion will most likely be lossy
287     *
288     * @param targetOps output type
289     * @param source source value
290     * @param <U> output type
291     * @return output value
292     */
293    @Override
294    public <U> U convertTo(final DynamicOps<U> targetOps, final ConfigurationNode source) {
295        final @Nullable U self = convertSelf(requireNonNull(targetOps, "targetOps"), requireNonNull(source, "source"));
296        if (self != null) {
297            return self;
298        }
299
300        if (source.isMap()) {
301            return convertMap(targetOps, source);
302        } else if (source.isList()) {
303            return convertList(targetOps, source);
304        } else {
305            final @Nullable Object value = source.rawScalar();
306            if (value == null) {
307                return targetOps.empty();
308            } else if (value instanceof String) {
309                return targetOps.createString((String) value);
310            } else if (value instanceof Boolean) {
311                return targetOps.createBoolean((Boolean) value);
312            } else if (value instanceof Short) {
313                return targetOps.createShort((Short) value);
314            } else if (value instanceof Integer) {
315                return targetOps.createInt((Integer) value);
316            } else if (value instanceof Long) {
317                return targetOps.createLong((Long) value);
318            } else if (value instanceof Float) {
319                return targetOps.createFloat((Float) value);
320            } else if (value instanceof Double) {
321                return targetOps.createDouble((Double) value);
322            } else if (value instanceof Byte) {
323                return targetOps.createByte((Byte) value);
324            } else if (value instanceof byte[]) {
325                return targetOps.createByteList(ByteBuffer.wrap((byte[]) value));
326            } else if (value instanceof int[]) {
327                return targetOps.createIntList(IntStream.of((int[]) value));
328            } else if (value instanceof long[]) {
329                return targetOps.createLongList(LongStream.of((long[]) value));
330            } else {
331                throw new IllegalArgumentException("Scalar value '" + source + "' has an unknown type: " + value.getClass().getName());
332            }
333        }
334    }

336    /**
337     * Get the value of the provided node if it is a number or boolean.
338     *
339     * <p>If {@link #compressMaps()} is true, values may be coerced from
340     * another type.
341     *
342     * @param input data source
343     * @return extracted number
344     */
345    @Override
346    public DataResult<Number> getNumberValue(final ConfigurationNode input) {
347        if (!(input.isMap() || input.isList())) {
348            final @Nullable Object value = input.rawScalar();
349            if (value instanceof Number) {
350                return DataResult.success((Number) value);
351            } else if (value instanceof Boolean) {
352                return DataResult.success((boolean) value ? 1 : 0);
353            }
354
355            if (compressMaps()) {
356                final int result = input.getInt(Integer.MIN_VALUE);
357                if (result == Integer.MIN_VALUE) {
358                    return DataResult.error("Value is not a number");
359                }
360                return DataResult.success(result);
361            }
362        }
363
364        return DataResult.error("Not a number: " + input);
365    }

437     * Appends element {@code value} to list node {@code input}.
438     *
439     * @param input base node. Must be empty or of list type
440     * @param value value to add as element to the list
441     * @return success with modified node, or error if {@code input} contains a
442     *          non-{@link ConfigurationNode#isList() list} value
443     */
444    @Override
445    public DataResult<ConfigurationNode> mergeToList(final ConfigurationNode input, final ConfigurationNode value) {
446        if (input.isList() || input.empty()) {
447            final ConfigurationNode ret = guardOutputRead(input);
448            ret.appendListNode().from(value);
449            return DataResult.success(ret);
450        }
451
452        return DataResult.error("mergeToList called on a node which is not a list: " + input, input);
453    }

537    /**
538     * Get a consumer that takes an action to perform on every element of
539     * list node {@code input}.
540     *
541     * <p>As an example, to print out every node in a list
542     *      (minus error checking):
543     *     <pre>
544     *         getList(listNode).result().get()
545     *             .accept(element -&gt; System.out.println(element);
546     *     </pre>
547     *
548     * @param input data source
549     * @return result, that if successful will take an action to perform on
550     *          every element
551     */
552    @Override
553    public DataResult<Consumer<Consumer<ConfigurationNode>>> getList(final ConfigurationNode input) {
554        if (input.isList()) {
555            return DataResult.success(action -> {
556                for (ConfigurationNode child : input.childrenList()) {
557                    action.accept(guardOutputRead(child));
558                }
559            });
560        } else {
561            return DataResult.error("Input node is not a list");
562        }
563    }

696    public ConfigurationNode set(final ConfigurationNode input, final String key, final ConfigurationNode value) {
697        final ConfigurationNode ret = guardInputWrite(input);
698        ret.node(key).from(value);
699        return ret;
700    }

702    /**
703     * Copies the input node and transform its child at {@code key}.
704     *
705     * <p>Return a copy of the input node with the child at {@code key}
706     * transformed by the provided function
707     *
708     * <p>If there is no value at {@code key}, the input node will be
709     * returned unmodified.
710     *
711     * @param input base value
712     * @param key key to change
713     * @param function function to process the node at {@code wrappedKey}
714     * @return an updated copy of input node
715     */
716    @Override
717    public ConfigurationNode update(final ConfigurationNode input, final String key, final Function<ConfigurationNode, ConfigurationNode> function) {
718        if (input.node(key).virtual()) {
719            return input;
720        }
721
722        final ConfigurationNode ret = guardInputWrite(input);
723        final ConfigurationNode child = ret.node(key);
724        child.from(function.apply(child));
725        return ret;
726    }

728    /**
729     * Copies the input node and transform the node at {@code wrappedKey}.
730     *
731     * <p>Return a copy of the input node with the child at {@code wrappedKey}
732     * transformed by the provided function
733     *
734     * <p>If there is no value at {@code wrappedKey}, the input node will be
735     * returned unmodified.
736     *
737     * <p>Keys will be interpreted as a single Object, and can only
738     * currently access direct children.
739     *
740     * @param input base value
741     * @param wrappedKey key to change
742     * @param function function to process the node at {@code wrappedKey}
743     * @return an updated copy of input node
744     */
745    @Override
746    public ConfigurationNode updateGeneric(final ConfigurationNode input, final ConfigurationNode wrappedKey,
747            final Function<ConfigurationNode, ConfigurationNode> function) {
748        final Object key = keyFrom(wrappedKey);
749        if (input.node(key).virtual()) {
750            return input;
751        }
752
753        final ConfigurationNode ret = guardInputWrite(input);
754
755        final ConfigurationNode child = ret.node(key);
756        child.from(function.apply(child));
757        return ret;
758    }
