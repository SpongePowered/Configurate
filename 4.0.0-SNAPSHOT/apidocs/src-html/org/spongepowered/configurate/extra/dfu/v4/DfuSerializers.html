017package org.spongepowered.configurate.extra.dfu.v4;
018
019import static java.util.Objects.requireNonNull;
020
021import com.mojang.serialization.Codec;
022import com.mojang.serialization.Lifecycle;
023import io.leangen.geantyref.TypeToken;
024import org.checkerframework.checker.nullness.qual.Nullable;
025import org.spongepowered.configurate.serialize.TypeSerializer;
026import org.spongepowered.configurate.serialize.TypeSerializerCollection;
027
028/**
029 * A bridge between Configurate and DataFixerUpper serialization types.
030 */
031public final class DfuSerializers {
032
033    private DfuSerializers() {
034    }
035
036    /**
037     * Create a new serializer wrapping the provided {@link Codec}.
038     *
039     * @param codec codec to use for the serialization operation
040     * @param <V> value type
041     * @return a new serializer
042     */
043    public static <V> TypeSerializer<V> serializer(final Codec<V> codec) {
044        return new CodecSerializer<>(requireNonNull(codec, "codec"));
045    }
046
047    /**
048     * Create a new codec that uses the default type serializer collection
049     * to serialize an object of the provided type.
050     *
051     * @param type token representing a value type
052     * @param <S> value type
053     * @return a codec for the type, or null if an appropriate
054     *      {@link TypeSerializer} could not be found.
055     */
056    public static <S> @Nullable Codec<S> codec(final TypeToken<S> type) {
057        return codec(requireNonNull(type, "type"), TypeSerializerCollection.defaults());
058    }
059
060    /**
061     * Create a new codec based on a Configurate {@link TypeSerializer}.
062     *
063     * @param type type to serialize
064     * @param collection source for values
065     * @param <V> value type
066     * @return a codec, or null if an appropriate {@link TypeSerializer}
067     *      could not be found for the TypeToken.
068     */
069    public static <V> @Nullable Codec<V> codec(final TypeToken<V> type, final TypeSerializerCollection collection) {
070        final @Nullable TypeSerializer<V> serial = collection.get(requireNonNull(type, "type"));
071        if (serial == null) {
072            return null;
073        }
074        return new TypeSerializerCodec<>(type, serial, ConfigurateOps.forSerializers(collection)).withLifecycle(Lifecycle.stable());
075    }
076
077}
