017package org.spongepowered.configurate.extra.dfu.v3;
018
019import static java.util.Objects.requireNonNull;
020
021import com.mojang.serialization.Dynamic;
022import org.spongepowered.configurate.CommentedConfigurationNode;
023import org.spongepowered.configurate.ConfigurationNode;
024import org.spongepowered.configurate.ConfigurationNodeFactory;
025import org.spongepowered.configurate.ConfigurationOptions;
026import org.spongepowered.configurate.serialize.TypeSerializerCollection;
027
028/**
029 * A builder for {@link ConfigurateOps} instances.
030 */
031public final class ConfigurateOpsBuilder {
032
033    private ConfigurationNodeFactory<? extends ConfigurationNode> nodeSupplier = CommentedConfigurationNode.factory();
034    private boolean compressed;
035    private ConfigurateOps.Protection readProtection = ConfigurateOps.Protection.COPY_DEEP;
036    private ConfigurateOps.Protection writeProtection = ConfigurateOps.Protection.COPY_DEEP;
037
038    ConfigurateOpsBuilder() {}
039
040    /**
041     * Set the node factory for the returned ops.
042     *
043     * <p>The default node factory wil create {@link CommentedConfigurationNode}
044     * instances using Confabricate's minecraft serializers.
045     *
046     * @param supplier source for new nodes created to store values in
047     *     the {@code create*} methods
048     * @return this builder
049     */
050    public ConfigurateOpsBuilder factory(final ConfigurationNodeFactory<? extends ConfigurationNode> supplier) {
051        this.nodeSupplier = requireNonNull(supplier, "nodeSupplier");
052        return this;
053    }
054
055    /**
056     * Set a node factory that will use the provided collection.
057     *
058     * <p>This will replace any set {@link #factory(ConfigurationNodeFactory)}.
059     *
060     * @param collection type serializers to use for nodes.
061     * @return this builder
062     */
063    public ConfigurateOpsBuilder factoryFromSerializers(final TypeSerializerCollection collection) {
064        requireNonNull(collection, "collection");
065        return factory(options -> CommentedConfigurationNode.root(options.serializers(collection)));
066    }
067
068    /**
069     * Set the node factory based on the options of the provided node.
070     *
071     * @param node node to use
072     * @return this builder
073     */
074    public ConfigurateOpsBuilder factoryFromNode(final ConfigurationNode node) {
075        final ConfigurationOptions options = requireNonNull(node, "node").options();
076        return factory(new ConfigurationNodeFactory<ConfigurationNode>() {
077            @Override
078            public ConfigurationNode createNode(final ConfigurationOptions options) {
079                return CommentedConfigurationNode.root(options);
080            }
081
082            @Override public ConfigurationOptions defaultOptions() {
083                return options;
084            }
085        });
086    }
087
088    /**
089     * Set whether {@link com.mojang.serialization.Keyable} values should be compressed.
090     *
091     * @param compressed whether to compress values
092     * @return this builder
093     * @see ConfigurateOps#compressMaps() for more about what compression is
094     */
095    public ConfigurateOpsBuilder compressed(final boolean compressed) {
096        this.compressed = compressed;
097        return this;
098    }
099
100    /**
101     * Set how nodes returned from read methods will be protected
102     * from modification.
103     *
104     * <p>For read protection, the protection level refers to how the attached
105     * node will be affected by modifications made to the nodes returned from
106     * {@code get*} methods.
107     *
108     * @param readProtection protection level
109     * @return this builder
110     */
111    public ConfigurateOpsBuilder readProtection(final ConfigurateOps.Protection readProtection) {
112        this.readProtection = requireNonNull(readProtection, "readProtection");
113        return this;
114    }
115
116    /**
117     * Set how nodes provided to mutator methods will be protected
118     * from modification.
119     *
120     * <p>For write protection, the protection level refers to how the provided
121     * {@code prefix} node will be protected from seeing changes to the
122     * operation
123     *
124     * @param writeProtection protection level
125     * @return this builder
126     */
127    public ConfigurateOpsBuilder writeProtection(final ConfigurateOps.Protection writeProtection) {
128        this.writeProtection = requireNonNull(writeProtection, "writeProtection");
129        return this;
130    }
131
132    /**
133     * Set how nodes will be protected from both read and write modifications.
134     *
135     * @param protection protection level
136     * @return this builder
137     * @see #readProtection(ConfigurateOps.Protection) for how this level
138     *      affects value reads
139     * @see #writeProtection(ConfigurateOps.Protection) for how this level
140     *      affects value writes
141     */
142    public ConfigurateOpsBuilder readWriteProtection(final ConfigurateOps.Protection protection) {
143        requireNonNull(protection, "protection");
144        this.readProtection = protection;
145        this.writeProtection = protection;
146        return this;
147    }
148
149    /**
150     * Create a new ops instance.
151     *
152     * <p>All options have defaults provided and all setters validate their
153     * input, so by the time this method is reached the builder will be in a
154     * valid state.
155     *
156     * @return the new instance
157     */
158    public ConfigurateOps build() {
159        return new ConfigurateOps(this.nodeSupplier, this.compressed, this.readProtection, this.writeProtection);
160    }
161
162    /**
163     * Build a new ops instance, returned as part of a {@linkplain Dynamic}.
164     *
165     * <p>Returned ops instances will not take type serializers or other options
166     * from the provided node. For that, use {@link #factoryFromNode(ConfigurationNode)}.
167     *
168     * @param node wrapped node
169     * @return new dynamic
170     */
171    public Dynamic<ConfigurationNode> buildWrapping(final ConfigurationNode node) {
172        return new Dynamic<>(build(), node);
173    }
174
175}
