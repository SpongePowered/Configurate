017package org.spongepowered.configurate.extra.dfu.v4;
018
019import static java.util.Objects.requireNonNull;
020
021import com.mojang.serialization.Dynamic;
022import org.spongepowered.configurate.CommentedConfigurationNode;
023import org.spongepowered.configurate.ConfigurationNode;
024import org.spongepowered.configurate.ConfigurationNodeFactory;
025import org.spongepowered.configurate.ConfigurationOptions;
026import org.spongepowered.configurate.serialize.TypeSerializerCollection;
027
028/**
029 * A builder for {@link ConfigurateOps} instances.
030 */
031public final class ConfigurateOpsBuilder {
032
033    private ConfigurationNodeFactory<? extends ConfigurationNode> nodeSupplier = CommentedConfigurationNode.factory();
034    private boolean compressed;
035    private ConfigurateOps.Protection readProtection = ConfigurateOps.Protection.COPY_DEEP;
036    private ConfigurateOps.Protection writeProtection = ConfigurateOps.Protection.COPY_DEEP;
037
038    ConfigurateOpsBuilder() {}
039
040    /**
041     * Set the node factory for the returned ops.
042     *
043     * <p>The default node factory wil create {@link CommentedConfigurationNode}
044     * instances using Confabricate's minecraft serializers.
045     *
046     * @param supplier source for new nodes created to store values in
047     *     the {@code create*} methods
048     * @return this builder
049     */
050    public ConfigurateOpsBuilder factory(final ConfigurationNodeFactory<? extends ConfigurationNode> supplier) {
051        this.nodeSupplier = requireNonNull(supplier, "nodeSupplier");
052        return this;
053    }
054
055    /**
056     * Set a node factory that will use the provided collection.
057     *
058     * <p>This will replace any set {@link #factory(ConfigurationNodeFactory)}.
059     *
060     * @param collection type serializers to use for nodes.
061     * @return this builder
062     */
063    public ConfigurateOpsBuilder factoryFromSerializers(final TypeSerializerCollection collection) {
064        requireNonNull(collection, "collection");
065        return factory(options -> CommentedConfigurationNode.root(options.serializers(collection)));
066    }
067
068    /**
069     * Set the node factory based on the options of the provided node.
070     *
071     * <p>This will replace any set {@link #factory(ConfigurationNodeFactory)}.
072     *
073     * @param node node to use
074     * @return this builder
075     */
076    public ConfigurateOpsBuilder factoryFromNode(final ConfigurationNode node) {
077        final ConfigurationOptions options = requireNonNull(node, "node").options();
078        return factory(new ConfigurationNodeFactory<ConfigurationNode>() {
079            @Override
080            public ConfigurationNode createNode(final ConfigurationOptions options) {
081                return CommentedConfigurationNode.root(options);
082            }
083
084            @Override public ConfigurationOptions defaultOptions() {
085                return options;
086            }
087        });
088    }
089
090    /**
091     * Set whether {@link com.mojang.serialization.Keyable} values should be compressed.
092     *
093     * @param compressed whether to compress values
094     * @return this builder
095     * @see ConfigurateOps#compressMaps() for more about what compression is
096     */
097    public ConfigurateOpsBuilder compressed(final boolean compressed) {
098        this.compressed = compressed;
099        return this;
100    }
101
102    /**
103     * Set how nodes returned from read methods will be protected
104     * from modification.
105     *
106     * <p>For read protection, the protection level refers to how the attached
107     * node will be affected by modifications made to the nodes returned from
108     * {@code get*} methods.
109     *
110     * @param readProtection protection level
111     * @return this builder
112     */
113    public ConfigurateOpsBuilder readProtection(final ConfigurateOps.Protection readProtection) {
114        this.readProtection = requireNonNull(readProtection, "readProtection");
115        return this;
116    }
117
118    /**
119     * Set how nodes provided to mutator methods will be protected
120     * from modification.
121     *
122     * <p>For write protection, the protection level refers to how the provided
123     * {@code prefix} node will be protected from seeing changes to the
124     * operation
125     *
126     * @param writeProtection protection level
127     * @return this builder
128     */
129    public ConfigurateOpsBuilder writeProtection(final ConfigurateOps.Protection writeProtection) {
130        this.writeProtection = requireNonNull(writeProtection, "writeProtection");
131        return this;
132    }
133
134    /**
135     * Set how nodes will be protected from both read and write modifications.
136     *
137     * @param protection protection level
138     * @return this builder
139     * @see #readProtection(ConfigurateOps.Protection) for how this level
140     *      affects value reads
141     * @see #writeProtection(ConfigurateOps.Protection) for how this level
142     *      affects value writes
143     */
144    public ConfigurateOpsBuilder readWriteProtection(final ConfigurateOps.Protection protection) {
145        requireNonNull(protection, "protection");
146        this.readProtection = protection;
147        this.writeProtection = protection;
148        return this;
149    }
150
151    /**
152     * Create a new ops instance.
153     *
154     * <p>All options have defaults provided and all setters validate their
155     * input, so by the time this method is reached the builder will be in a
156     * valid state.
157     *
158     * @return the new instance
159     */
160    public ConfigurateOps build() {
161        return new ConfigurateOps(this.nodeSupplier, this.compressed, this.readProtection, this.writeProtection);
162    }
163
164    /**
165     * Build a new ops instance, returned as part of a {@linkplain Dynamic}.
166     *
167     * <p>Returned ops instances will not take type serializers or other options
168     * from the provided node. For that, use {@link #factoryFromNode(ConfigurationNode)}.
169     *
170     * @param node wrapped node
171     * @return new dynamic
172     */
173    public Dynamic<ConfigurationNode> buildWrapping(final ConfigurationNode node) {
174        return new Dynamic<>(build(), node);
175    }
176
177}
