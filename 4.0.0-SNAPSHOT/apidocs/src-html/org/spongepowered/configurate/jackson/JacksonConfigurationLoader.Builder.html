047/**
048 * A loader for JSON-formatted configurations, using the jackson library for
049 * parsing and generation.
050 */
051public final class JacksonConfigurationLoader extends AbstractConfigurationLoader<BasicConfigurationNode> {
052
053    private static final Set<Class<?>> NATIVE_TYPES = UnmodifiableCollections.toSet(Map.class, List.class, Double.class, Float.class,
054            Long.class, Integer.class, Boolean.class, String.class, byte[].class);
055
056    /**
057     * Creates a new {@link JacksonConfigurationLoader} builder.
058     *
059     * @return a new builder
060     */
061    public static Builder builder() {
062        return new Builder();
063    }
064
065    /**
066     * Builds a {@link JacksonConfigurationLoader}.
067     */
068    public static final class Builder extends AbstractConfigurationLoader.Builder<Builder, JacksonConfigurationLoader> {
069        private final JsonFactoryBuilder factory = new JsonFactoryBuilder();
070        private int indent = 2;
071        private FieldValueSeparatorStyle fieldValueSeparatorStyle = FieldValueSeparatorStyle.SPACE_AFTER;
072
073        Builder() {
074            this.factory.enable(JsonReadFeature.ALLOW_JAVA_COMMENTS)
075                    .enable(JsonReadFeature.ALLOW_YAML_COMMENTS)
076                    .enable(JsonReadFeature.ALLOW_BACKSLASH_ESCAPING_ANY_CHARACTER)
077                    .enable(JsonReadFeature.ALLOW_UNQUOTED_FIELD_NAMES)
078                    .enable(JsonReadFeature.ALLOW_SINGLE_QUOTES)
079                    .enable(JsonReadFeature.ALLOW_NON_NUMERIC_NUMBERS)
080                    .enable(JsonReadFeature.ALLOW_UNESCAPED_CONTROL_CHARS);
081        }
082
083        /**
084         * Gets the {@link JsonFactory} used to configure the implementation.
085         *
086         * @return the json factory
087         */
088        public JsonFactoryBuilder factoryBuilder() {
089            return this.factory;
090        }
091
092        /**
093         * Sets the level of indentation the resultant loader should use.
094         *
095         * @param indent the indent level
096         * @return this builder (for chaining)
097         */
098        public Builder indent(final int indent) {
099            this.indent = indent;
100            return this;
101        }
102
103        /**
104         * Gets the level of indentation to be used by the resultant loader.
105         *
106         * @return the indent level
107         */
108        public int indent() {
109            return this.indent;
110        }
111
112        /**
113         * Sets the field value separator style the resultant loader should use.
114         *
115         * @param style the style
116         * @return this builder (for chaining)
117         */
118        public Builder fieldValueSeparatorStyle(final FieldValueSeparatorStyle style) {
119            this.fieldValueSeparatorStyle = style;
120            return this;
121        }
122
123        /**
124         * Gets the field value separator style to be used by the built loader.
125         *
126         * @return the style
127         */
128        public FieldValueSeparatorStyle fieldValueSeparatorStyle() {
129            return this.fieldValueSeparatorStyle;
130        }
131
132        @Override
133        public JacksonConfigurationLoader build() {
134            defaultOptions(o -> o.nativeTypes(NATIVE_TYPES));
135            return new JacksonConfigurationLoader(this);
136        }
137    }
138
139    private final JsonFactory factory;
140    private final int indent;
141    private final FieldValueSeparatorStyle fieldValueSeparatorStyle;
142
143    private JacksonConfigurationLoader(final Builder builder) {
144        super(builder, new CommentHandler[]{CommentHandlers.DOUBLE_SLASH, CommentHandlers.SLASH_BLOCK, CommentHandlers.HASH});
145        this.factory = builder.factoryBuilder().build();
146        this.factory.disable(JsonParser.Feature.AUTO_CLOSE_SOURCE);
147        this.indent = builder.indent();
148        this.fieldValueSeparatorStyle = builder.fieldValueSeparatorStyle();
149    }
150
151    private static final int MAX_CTX_LENGTH = 80;
152
153    @Override
154    protected void loadInternal(final BasicConfigurationNode node, final BufferedReader reader) throws ParsingException {
155        try (JsonParser parser = this.factory.createParser(reader)) {
156            parser.nextToken();
157            parseValue(parser, node);
158        } catch (final StreamReadException ex) {
159            throw newException(node, ex.getLocation(), ex.getRequestPayloadAsString(), ex.getMessage(), ex.getCause());
160        } catch (final IOException ex) {
161            throw new ParsingException(node, -1, -1, null, null, ex);
162        }
163    }
164
165    private static void parseValue(final JsonParser parser, final ConfigurationNode node) throws IOException, ParsingException {
166        try {
167            final JsonToken token = parser.getCurrentToken();
168            switch (token) {
169                case START_OBJECT:
170                    parseObject(parser, node);
171                    break;
172                case START_ARRAY:
173                    parseArray(parser, node);
174                    break;
175                case VALUE_NUMBER_FLOAT:
176                    final double doubleVal = parser.getDoubleValue();
177                    if ((float) doubleVal != doubleVal) {
178                        node.raw(parser.getDoubleValue());
179                    } else {
180                        node.raw(parser.getFloatValue());
181                    }
182                    break;
183                case VALUE_NUMBER_INT:
184                    final long longVal = parser.getLongValue();
185                    if ((int) longVal != longVal) {
186                        node.raw(parser.getLongValue());
187                    } else {
188                        node.raw(parser.getIntValue());
189                    }
190                    break;
191                case VALUE_STRING:
192                    node.raw(parser.getText());
193                    break;
194                case VALUE_TRUE:
195                case VALUE_FALSE:
196                    node.raw(parser.getBooleanValue());
197                    break;
198                case VALUE_NULL: // Ignored values
199                case FIELD_NAME:
200                    break;
201                default:
202                    final JsonLocation loc = parser.getTokenLocation();
203                    throw new ParsingException(node, loc.getLineNr(), loc.getColumnNr(), parser.getText(), "Unsupported token type: " + token, null);
204            }
205        } catch (final StreamReadException ex) {
206            throw newException(node, ex.getLocation(), ex.getRequestPayloadAsString(), ex.getMessage(), ex.getCause());
207        }
208    }
209
210    private static void parseArray(final JsonParser parser, final ConfigurationNode node) throws IOException, ParsingException {
211        boolean written = false;
212        JsonToken token;
213        while ((token = parser.nextToken()) != null) {
214            switch (token) {
215                case END_ARRAY:
216                    // ensure the type is preserved
217                    if (!written) {
218                        node.raw(Collections.emptyList());
219                    }
220                    return;
221                default:
222                    parseValue(parser, node.appendListNode());
223                    written = true;
224            }
225        }
226        throw newException(node, parser.getCurrentLocation(), null, "Reached end of stream with unclosed array!", null);
227    }
228
229    private static void parseObject(final JsonParser parser, final ConfigurationNode node) throws IOException, ParsingException {
230        boolean written = false;
231        JsonToken token;
232        while ((token = parser.nextToken()) != null) {
233            if (token == JsonToken.END_OBJECT) { // ensure the type is preserved
234                if (!written) {
235                    node.raw(Collections.emptyMap());
236                }
237                return;
238            } else {
239                parseValue(parser, node.node(parser.getCurrentName()));
240                written = true;
241            }
242        }
243        throw newException(node, parser.getCurrentLocation(), null, "Reached end of stream with unclosed object!", null);
244    }
245
246    @Override
247    public void saveInternal(final ConfigurationNode node, final Writer writer) throws ConfigurateException {
248        try (JsonGenerator generator = this.factory.createGenerator(writer)) {
249            generator.setPrettyPrinter(new ConfiguratePrettyPrinter(this.indent, this.fieldValueSeparatorStyle));
250            node.visit(JacksonVisitor.INSTANCE.get(), generator);
251            writer.write(SYSTEM_LINE_SEPARATOR); // Jackson doesn't add a newline at the end of files by default
252        } catch (final IOException ex) {
253            throw new ConfigurateException(ex);
254        }
255    }
256
257    @Override
258    public BasicConfigurationNode createNode(final @NonNull ConfigurationOptions options) {
259        return BasicConfigurationNode.root(options.nativeTypes(NATIVE_TYPES));
260    }
261
262    private static ParsingException newException(final ConfigurationNode node,
263            final JsonLocation position,
264            final @Nullable String content,
265            final @Nullable String message,
266            final @Nullable Throwable cause) {
267        @Nullable String context = content == null ? null : content.substring((int) position.getCharOffset());
268        if (context != null) {
269            int nextLine = context.indexOf('\n');
270            if (nextLine == -1) {
271                nextLine = context.length();
272            } else if (context.charAt(nextLine - 1) == '\r') {
273                nextLine--;
274            }
275
276            if (nextLine > MAX_CTX_LENGTH) {
277                context = context.substring(0, MAX_CTX_LENGTH) + "...";
278            } else {
279                context = context.substring(0, nextLine);
280            }
281        }
282        // no newline: set to length
283        // too long: truncate
284        // otherwise: trim to position of next newline
285        return new ParsingException(node, position.getLineNr(), position.getColumnNr(), context, message, cause);
286    }
