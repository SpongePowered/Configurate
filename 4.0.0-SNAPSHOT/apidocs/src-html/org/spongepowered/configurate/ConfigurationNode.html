039/**
040 * A node in the configuration tree.
041 *
042 * <p>All aspects of a configurations structure are represented using instances
043 * of {@link ConfigurationNode}, and the links between them.</p>
044 *
045 * <p>{@link ConfigurationNode}s can hold different types of value. They can:</p>
046 *
047 * <ul>
048 *     <li>Hold a single "scalar" value (accessed by {@link #raw()}</li>
049 *     <li>Represent a "list" of child {@link ConfigurationNode}s (accessed by {@link #isList()} and {@link #childrenList()})</li>
050 *     <li>Represent a "map" of child {@link ConfigurationNode}s (accessed by {@link #isMap()} and {@link #childrenMap()})</li>
051 *     <li>Hold no value at all (when {@link #virtual()} is true)</li>
052 * </ul>
053 *
054 * <p>The overall configuration stems from a single "root" node, which is
055 * provided by the {@link ConfigurationLoader}, or by other means programmatically.</p>
056 *
057 * <p>This is effectively the main class of Configurate.</p>
058 *
059 * @since 4.0.0
060 */
061public interface ConfigurationNode {

072     *
073     * <p>The key determines this {@link ConfigurationNode}s position within
074     * the overall configuration structure.</p>
075     *
076     * <p>If this node is currently {@link #virtual() virtual}, this method's
077     * result may be inaccurate.</p>
078     *
079     * <p>Note that this method only returns the nearest "link" in the
080     * hierarchy, and does not return a representation of the full path. See
081     * {@link #path()} for that.</p>
082     *
083     * <p>The {@link ConfigurationNode}s returned as values from
084     * {@link #childrenMap()} will have keys derived from their pairing in
085     * the map node.</p>
086     *
087     * <p>The {@link ConfigurationNode}s returned from
088     * {@link #childrenList()} will have keys derived from their position
089     * (index) in the list node.</p>
090     *
091     * @return the key of this node
092     * @since 4.0.0
093     */

269    /**
270     * Create a collector that appends values to this node as map children.
271     *
272     * <p>This collector does not accept values in parallel.</p>
273     *
274     * @param valueType marker for value type
275     * @param <V> value type
276     * @return a new collector
277     * @since 4.0.0
278     */
279    default <V> Collector<Map.Entry<?, V>, ? extends ConfigurationNode, ? extends ConfigurationNode> toMapCollector(final TypeToken<V> valueType) {
280        return Collector.of(() -> this, (node, entry) -> {
281            try {
282                node.node(entry.getKey()).set(valueType, entry.getValue());
283            } catch (SerializationException e) {
284                throw new IllegalArgumentException(e);
285            }
286        }, ConfigurationNode::mergeFrom);
287    }

655     * If this node has list values, this function unwraps them and converts
656     * them to an appropriate type based on the provided function.
657     *
658     * <p>If this node has a scalar value, this function treats it as a list
659     * with one value.</p>
660     *
661     * @param elementType expected type
662     * @param defSupplier function that will be called to calculate a default
663     *                    value only if there is no existing value of the
664     *                    correct type
665     * @param <V> expected type
666     * @return an immutable copy of the values contained that could be
667     *         successfully converted, or {@code def} if no values could be
668     *         converted.
669     * @throws SerializationException if any value fails to be converted to the
670     *                                requested type
671     * @since 4.0.0
672     */
673    @SuppressWarnings({"unchecked", "checkstyle:NoGetSetPrefix"})
674    default <V> List<V> getList(Class<V> elementType, Supplier<List<V>> defSupplier) throws SerializationException {
675        final Type type = TypeFactory.parameterizedClass(List.class, elementType);
676        final List<V> ret = (List<V>) get(type, defSupplier);
677        return ret.isEmpty() ? storeDefault(this, type, defSupplier.get()) : ret;
678    }
