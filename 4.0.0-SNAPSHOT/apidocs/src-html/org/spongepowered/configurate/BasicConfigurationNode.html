017package org.spongepowered.configurate;
018
019import org.spongepowered.configurate.util.CheckedConsumer;
020
021/**
022 * A standard configuration node, without any additional options.
023 */
024public interface BasicConfigurationNode extends ScopedConfigurationNode<BasicConfigurationNode> {
025
026    /**
027     * Create a new factory providing {@link BasicConfigurationNode} instances.
028     *
029     * <p>The returned factory will create nodes with default options.</p>
030     *
031     * @return a new factory
032     */
033    static ConfigurationNodeFactory<BasicConfigurationNode> factory() {
034        return BasicConfigurationNode::root;
035    }
036
037    /**
038     * Create a new empty root node.
039     *
040     * <p>This node will use the {@link ConfigurationOptions#defaults() default options}</p>
041     *
042     * <p>A root node is always attached, and has no parent and an
043     * empty path.</p>
044     *
045     * @return a new empty node
046     */
047    static BasicConfigurationNode root() {
048        return root(ConfigurationOptions.defaults());
049    }
050
051    /**
052     * Create a new root node with the provided initializer which may throw.
053     *
054     * <p>This node will use the {@link ConfigurationOptions#defaults() default options}</p>
055     *
056     * <p>A root node is always attached, and has no parent and an
057     * empty path.</p>
058     *
059     * @param <E> error type thrown
060     * @param maker action to be applied to the newly created node
061     * @return a new initialized node
062     * @throws E when thrown from inner action
063     */
064    static <E extends Exception> BasicConfigurationNode root(final CheckedConsumer<? super BasicConfigurationNode, E> maker) throws E {
065        return root().act(maker);
066    }
067
068    /**
069     * Create a new empty root node with the provided options.
070     *
071     * <p>A root node is always attached, and has no parent and an
072     * empty path.</p>
073     *
074     * @param options options to apply.
075     * @return a new empty node
076     */
077    static BasicConfigurationNode root(final ConfigurationOptions options) {
078        return new BasicConfigurationNodeImpl(null, null, options);
079    }
080
081    /**
082     * Create a new root node with the provided options and initializer.
083     *
084     * <p>A root node is always attached, and has no parent and an
085     * empty path.</p>
086     *
087     * @param <E> thrown type
088     * @param options options to apply.
089     * @param maker action to be applied to the newly created node
090     * @return a new initialized node
091     * @throws E when thrown from inner action
092     */
093    static <E extends Exception> BasicConfigurationNode root(final ConfigurationOptions options,
094            final CheckedConsumer<? super BasicConfigurationNode, E> maker) throws E {
095        return root(options).act(maker);
096    }
097
098}
