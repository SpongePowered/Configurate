017package org.spongepowered.configurate.transformation;
018
019import io.leangen.geantyref.TypeToken;
020import org.checkerframework.checker.nullness.qual.Nullable;
021import org.spongepowered.configurate.ConfigurateException;
022import org.spongepowered.configurate.ConfigurationNode;
023
024import java.util.function.Supplier;
025
026/**
027 * Represents an action to be performed that transforms a node in the
028 * configuration tree.
029 */
030@FunctionalInterface
031public interface TransformAction {
032
033    /**
034     * Create a transform action that will remove the node at a specified path.
035     *
036     * @return new action
037     */
038    static TransformAction remove() {
039        return (path, value) -> {
040            value.raw(null);
041            return null;
042        };
043    }
044
045    /**
046     * Rename a node
047     *
048     * <p>This transformation cannot be applied to the root node.
049     *
050     * @param newKey the new key
051     * @return new action
052     */
053    static TransformAction rename(Object newKey) {
054        return (path, value) -> {
055            final Object[] arr = path.array();
056            if (arr.length == 0) {
057                throw new ConfigurateException(value, "The root node cannot be renamed!");
058            }
059            arr[arr.length - 1] = newKey;
060            return arr;
061        };
062    }
063
064    /**
065     * Create a transform action that will change the value of a node to one of
066     * the specified type.
067     *
068     * @param type value type
069     * @param value value
070     * @param <V> value type
071     * @return new transformation action
072     */
073    static <V> TransformAction set(TypeToken<V> type, @Nullable V value) {
074        return (path, node) -> {
075            node.set(type, value);
076            return null;
077        };
078    }
079
080    /**
081     * Create a transform action that will change the value of a node to one of
082     * the specified type.
083     *
084     * @param type value type
085     * @param valueSupplier supplier returning a value on each call
086     * @param <V> value type
087     * @return new transformation action
088     */
089    static <V> TransformAction set(TypeToken<V> type, Supplier<@Nullable V> valueSupplier) {
090        return (path, value) -> {
091            value.set(type, valueSupplier.get());
092            return null;
093        };
094    }
095
096    /**
097     * Create a transform action that will change the value of a node to one of
098     * the specified type.
099     *
100     * @param type value type
101     * @param valueSupplier supplier returning a value on each call
102     * @param <V> value type
103     * @return new transformation action
104     */
105    static <V> TransformAction set(Class<V> type, Supplier<V> valueSupplier) {
106        return (path, value) -> {
107            value.set(type, valueSupplier.get());
108            return null;
109        };
110    }
111
112
113    /**
114     * Called at a certain path, with the node at that path.
115     *
116     * <p><strong>Caution:</strong> The state of the <code>path</code> is
117     * only guaranteed to be accurate during a run of the transform function.
118     * Use {@link NodePath#copy()} if the path's state needs to
119     * be stored.
120     *
121     * @param path the path of the given node
122     * @param value the node at the input path. May be modified
123     * @return a modified path, or null if the path is to stay the same
124     */
125    @Nullable Object @Nullable[] visitPath(NodePath path, ConfigurationNode value) throws ConfigurateException;
126
127}
