017package org.spongepowered.configurate.transformation;
018
019import static java.util.Objects.requireNonNull;
020
021import org.checkerframework.checker.nullness.qual.NonNull;
022import org.spongepowered.configurate.ConfigurateException;
023import org.spongepowered.configurate.ConfigurationNode;
024
025import java.util.NavigableMap;
026import java.util.TreeMap;
027import java.util.function.Consumer;
028
029/**
030 * Represents a set of transformations on a configuration.
031 */
032@FunctionalInterface
033public interface ConfigurationTransformation {
034
035    /**
036     * A special object that represents a wildcard in a path provided to a
037     * configuration transformer.
038     */
039    Object WILDCARD_OBJECT = new Object();
040
041    /**
042     * Get an empty transformation.
043     *
044     * <p>This transformation will perform no actions.</p>
045     *
046     * @return empty transformation
047     */
048    static ConfigurationTransformation empty() {
049        return node -> {};
050    }
051
052    /**
053     * Create a new builder to create a basic configuration transformation.
054     *
055     * @return a new transformation builder.
056     */
057    static Builder builder() {
058        return new Builder();
059    }
060
061    /**
062     * This creates a builder for versioned transformations.
063     *
064     * @return a new builder for versioned transformations
065     */
066    static VersionedBuilder versionedBuilder() {
067        return new VersionedBuilder();
068    }
069
070    /**
071     * Creates a chain of {@link ConfigurationTransformation}s.
072     *
073     * @param transformations the transformations
074     * @return a new transformation chain
075     */
076    static ConfigurationTransformation chain(final ConfigurationTransformation... transformations) {
077        if (requireNonNull(transformations, "transformations").length == 0) {
078            throw new IllegalArgumentException("Cannot chain an empty array of transformations!");
079        }
080
081        if (transformations.length == 1) {
082            return transformations[0];
083        } else {
084            return new ChainedConfigurationTransformation(transformations);
085        }
086    }
087
088    /**
089     * Apply this transformation to a given node.
090     *
091     * @param node the target node
092     */
093    void apply(ConfigurationNode node) throws ConfigurateException;
094
095    /**
096     * Builds a basic {@link ConfigurationTransformation}.
097     */
098    final class Builder {
099        private MoveStrategy strategy = MoveStrategy.OVERWRITE;
100        private final NavigableMap<NodePath, TransformAction> actions;
101
102        Builder() {
103            this.actions = new TreeMap<>(new NodePathComparator());
104        }
105
106        /**
107         * Adds an action to the transformation.
108         *
109         * @param path the path to apply the action at
110         * @param action the action
111         * @return this builder (for chaining)
112         */
113        public Builder addAction(final NodePath path, final TransformAction action) {
114            this.actions.put(requireNonNull(path, "path"), requireNonNull(action, "action"));
115            return this;
116        }
117
118        /**
119         * Gets the move strategy to be used by the resultant transformation.
120         *
121         * @return the move strategy
122         */
123        public MoveStrategy moveStrategy() {
124            return this.strategy;
125        }
126
127        /**
128         * Sets the mode strategy to be used by the resultant transformation.
129         *
130         * @param strategy the strategy
131         * @return this builder (for chaining)
132         */
133        public Builder moveStrategy(final MoveStrategy strategy) {
134            this.strategy = requireNonNull(strategy, "strategy");
135            return this;
136        }
137
138        /**
139         * Builds the transformation.
140         *
141         * @return the transformation
142         */
143        public ConfigurationTransformation build() {
144            if (this.actions.isEmpty()) {
145                throw new IllegalArgumentException("At least one action must be specified to build a transformation");
146            }
147            return new SingleConfigurationTransformation(this.actions, this.strategy);
148        }
149    }
150
151    /**
152     * Builds a versioned {@link ConfigurationTransformation}.
153     */
154    final class VersionedBuilder {
155        private NodePath versionKey = NodePath.path("version");
156        private final NavigableMap<Integer, ConfigurationTransformation> versions = new TreeMap<>();
157
158        VersionedBuilder() {}
159
160        /**
161         * Sets the path of the version key within the configuration.
162         *
163         * @param versionKey the path to the version key
164         * @return this builder (for chaining)
165         */
166        public VersionedBuilder versionKey(final Object... versionKey) {
167            this.versionKey = NodePath.of(versionKey);
168            return this;
169        }
170
171        /**
172         * Adds a transformation to this builder for the given version.
173         *
174         * <p>The version must be between 0 and {@link Integer#MAX_VALUE}, and a version cannot be specified multiple times.
175         *
176         * @param version the version
177         * @param transformation the transformation
178         * @return this builder (for chaining)
179         */
180        @NonNull
181        public VersionedBuilder addVersion(final int version, final @NonNull ConfigurationTransformation transformation) {
182            if (version < 0) {
183                throw new IllegalArgumentException("Version must be at least 0");
184            }
185            if (this.versions.putIfAbsent(version, requireNonNull(transformation, "transformation")) != null) {
186                throw new IllegalArgumentException("Version '" + version + "' has been specified multiple times.");
187            }
188            return this;
189        }
190
191        /**
192         * Adds a new series of transformations for a version.
193         *
194         * <p>The version must be between 0 and {@link Integer#MAX_VALUE}.
195         *
196         * @param version the version
197         * @param transformations the transformations. To perform a version
198         *                        upgrade, these transformations will be
199         *                        executed in order.
200         * @return this builder
201         */
202        public @NonNull VersionedBuilder addVersion(final int version, final @NonNull ConfigurationTransformation... transformations) {
203            return this.addVersion(version, chain(transformations));
204        }
205
206        /**
207         * Create and add a new transformation to this builder.
208         *
209         * <p>The transformation will be created from the builder passed to
210         * the callback function</p>
211         *
212         * <p>The version must be between 0 and {@link Integer#MAX_VALUE}
213         *
214         * @param version the version
215         * @param maker the transformation
216         * @return this builder
217         */
218        public @NonNull VersionedBuilder makeVersion(final int version, final @NonNull Consumer<? super Builder> maker) {
219            final Builder builder = builder();
220            maker.accept(builder);
221            return this.addVersion(version, builder.build());
222        }
223
224        /**
225         * Builds the transformation.
226         *
227         * @return the transformation
228         */
229        public ConfigurationTransformation.@NonNull Versioned build() {
230            if (this.versions.isEmpty()) {
231                throw new IllegalArgumentException("At least one version must be specified to build a transformation");
232            }
233            return new VersionedTransformation(this.versionKey, this.versions);
234        }
235    }
236
237    /**
238     * A transformation that is aware of node versions.
239     */
240    interface Versioned extends ConfigurationTransformation {
241        int VERSION_UNKNOWN = -1;
242
243        /**
244         * Get the path the node's current version is located at.
245         *
246         * @return version path
247         */
248        NodePath versionKey();
249
250        /**
251         * Get the latest version that nodes can be updated to.
252         *
253         * @return the most recent version
254         */
255        int latestVersion();
256
257        /**
258         * Get the version of a node hierarchy.
259         *
260         * <p>Note that the node checked here must be the same node passed to
261         * {@link #apply(ConfigurationNode)}, not any node in a hierarchy.
262         *
263         * <p>If the node value is not present or not coercible to an integer,
264         * {@link #VERSION_UNKNOWN} will be returned. When the transformation is
265         * executed, every version transformation will be applied.
266         *
267         * @param node node to check
268         * @return version, or {@link #VERSION_UNKNOWN} if no value is present
269         */
270        default int version(final ConfigurationNode node) {
271            return node.node(versionKey()).getInt(VERSION_UNKNOWN);
272        }
273    }
