017package org.spongepowered.configurate.transformation;
018
019import org.checkerframework.checker.nullness.qual.NonNull;
020
021import java.util.Collection;
022import java.util.Iterator;
023
024/**
025 * Represents the path to a given node.
026 */
027public interface NodePath extends Iterable<Object>, Cloneable {
028
029    /**
030     * Gets a specific element from the path array
031     *
032     * @param i The index to get
033     * @return Object at the index
034     */
035    Object get(int i);
036
037    /**
038     * Gets the length of the path
039     *
040     * @return Length of the path array
041     */
042    int size();
043
044    /**
045     * Returns a copy of the original path array
046     *
047     * @return the copied array
048     */
049    Object[] getArray();
050
051    /**
052     * Create a new path with the provided element appended to the end
053     *
054     * @param childKey The new key to append
055     * @return A new path object reflecting the extended path
056     */
057    NodePath withAppendedChild(@NonNull Object childKey);
058
059    /**
060     * Returns an iterator over the path.
061     *
062     * @return An iterator of the path
063     */
064    @NonNull
065    @Override
066    Iterator<Object> iterator();
067
068    /**
069     * Create a node path reference
070     *
071     * @param path The path to reference. The provided array will be copied.
072     * @return The path instance
073     */
074    static NodePath create(Object[] path) {
075        return new NodePathImpl(path, true);
076    }
077
078    /**
079     * Create a node path reference
080     *
081     * This overload takes varargs, and is designed to be imported statically to create paths
082     *
083     * @param elements The path to reference. The provided array will be copied.
084     * @return The path instance
085     */
086    static NodePath path(Object... elements) {
087        return create(elements);
088    }
089
090    /**
091     * Create a node path reference
092     *
093     * @param path A collection containing elements of the path to reference
094     * @return The path instance
095     */
096    static NodePath create(Collection<?> path) {
097        return new NodePathImpl(path.toArray(), false);
098    }
099
100    /**
101     * Create a new node path with the same data as this path
102     *
103     * @return The resulting path
104     */
105    NodePath clone();
106}
