017package org.spongepowered.configurate.transformation;
018
019import org.spongepowered.configurate.ConfigurationNode;
020
021/**
022 * Strategy to use when moving a node from one path to another.
023 */
024public enum MoveStrategy {
025
026    /**
027     * Moves nodes using {@link ConfigurationNode#mergeFrom(ConfigurationNode)}.
028     */
029    MERGE {
030        @Override
031        public <T extends ConfigurationNode> void move(final T source, final T target) {
032            target.mergeFrom(source);
033        }
034    },
035
036    /**
037     * Moves nodes using {@link ConfigurationNode#set(Object)}.
038     */
039    OVERWRITE {
040        @Override
041        public <T extends ConfigurationNode> void move(final T source, final T target) {
042            target.from(source);
043        }
044    };
045
046    /**
047     * Moves <code>source</code> to <code>target</code>.
048     *
049     * @param <T> the type of node being processed
050     * @param source the source node
051     * @param target the target node
052     */
053    public abstract <T extends ConfigurationNode> void move(T source, T target);
054
055}
