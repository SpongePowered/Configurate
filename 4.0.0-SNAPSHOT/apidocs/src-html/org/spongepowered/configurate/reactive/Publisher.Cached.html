017package org.spongepowered.configurate.reactive;
018
019import static java.util.Objects.requireNonNull;
020
021import org.checkerframework.checker.nullness.qual.Nullable;
022import org.spongepowered.configurate.util.CheckedFunction;
023import org.spongepowered.configurate.util.CheckedSupplier;
024
025import java.util.concurrent.Executor;
026import java.util.concurrent.ForkJoinPool;
027
028/**
029 * Something that can publish events.
030 *
031 * <p>Each subscriber is responsible for removing itself from this stream, by
032 * using the Disposable returned upon subscription.
033 *
034 * @param <V> the type of notification received by subscribers
035 */
036public interface Publisher<V> {
037
038    /**
039     * Execute an action returning a single value on the common {@link ForkJoinPool},
040     * and pass the result to any subscribers.
041     *
042     * <p>Subscribers who only begin subscribing after the operation has been
043     * completed will receive the result of the operation.
044     *
045     * @param action the action to perform
046     * @param <V> returned value type
047     * @param <E> exception thrown
048     * @return a publisher
049     */
050    static <V, E extends Exception> Publisher<V> execute(CheckedSupplier<V, E> action) {
051        return execute(action, ForkJoinPool.commonPool());
052    }
053
054    /**
055     * Execute an action returning a single value on the provided {@link Executor},
056     * and pass the result to any subscribers.
057     *
058     * <p>Subscribers who only begin subscribing after the operation has been
059     * completed will receive the result of the operation.
060     *
061     * @param action the action to perform
062     * @param executor the executor to perform this operation on
063     * @param <V> returned value type
064     * @param <E> exception thrown
065     * @return a publisher
066     */
067    static <V, E extends Exception> Publisher<V> execute(CheckedSupplier<V, E> action, Executor executor) {
068        return new ExecutePublisher<>(requireNonNull(action, "action"), requireNonNull(executor, "executor"));
069    }
070
071    /**
072     * Subscribe to updates from this Publisher. If this is already closed, the
073     * Subscriber will receive an error event with an IllegalStateException, and
074     * the returned {@link Disposable} will be a no-op.
075     *
076     * @param subscriber the listener to register
077     * @return a disposable that can be used to cancel this subscription
078     */
079    Disposable subscribe(Subscriber<? super V> subscriber);
080
081    /**
082     * Return whether or not this Publisher has any subscribers.
083     *
084     * <p>In a concurrent environment, this value could change from the time
085     * of calling.
086     *
087     * @return if there are subscribers
088     */
089    boolean hasSubscribers();
090
091    /**
092     * Create a new publisher that will transform events published.
093     *
094     * @param mapper transformer function
095     * @param <R> output value type
096     * @return a new publisher
097     */
098    default <R> Publisher<R> map(CheckedFunction<? super V, ? extends R, TransactionFailedException> mapper) {
099        return new MappedProcessor<>(mapper, this);
100    }
101
102    /**
103     * Return a publisher that will track its most recent value. The provided
104     * processor won't have a value until one is submitted to this publisher.
105     *
106     * @return a publisher based on this one
107     */
108    default Cached<V> cache() {
109        return cache(null);
110    }
111
112    /**
113     * Create a cached publisher with an initial value.
114     *
115     * @param initialValue value to initialize the returned publisher with
116     * @return publisher that will cache future responses
117     */
118    default Cached<V> cache(@Nullable V initialValue) {
119        return new CachedPublisher<>(this, initialValue);
120    }
121
122    /**
123     * Get the executor used to handle published events.
124     *
125     * @return the executor
126     */
127    Executor executor();
128
129    /**
130     * A publisher that caches the last value received.
131     *
132     * @param <V> value type
133     */
134    interface Cached<V> extends Publisher<V> {
135
136        V get();
137
138        void submit(V value);
139
140    }
141
142}
