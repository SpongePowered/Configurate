017package org.spongepowered.configurate.reactive;
018
019/**
020 * A listener to events that may be called by an {@link Publisher}.
021 *
022 * <p>For every publisher this subscriber is subscribed to, the subscriber will
023 * only process one event at a time -- effectively, within a single publisher
024 * this subscriber does not have to be aware of concurrent effects.
025 *
026 * @param <V> the value that will be received
027 */
028@FunctionalInterface
029public interface Subscriber<V> {
030
031    /**
032     * Called to submit a new item.
033     *
034     * @param item the item available
035     */
036    void submit(V item);
037
038    /**
039     * When an error occurs while subscribing to an {@link Publisher}, or is
040     * thrown during the execution of {@link #submit(Object)} that is not
041     * otherwise handled, this method will be called with the error. The
042     * associated {@link Publisher} will not send further update signals after
043     * an error is thrown.
044     *
045     * @param thrown the exception thrown
046     */
047    default void onError(Throwable thrown) {
048        final Thread t = Thread.currentThread();
049        t.getUncaughtExceptionHandler().uncaughtException(t, thrown);
050    }
051
052    /**
053     * When the {@link Publisher} this is subscribed to closes without error,
054     * this method will be called.
055     */
056    default void onClose() {
057        // no-op
058    }
059
060}
