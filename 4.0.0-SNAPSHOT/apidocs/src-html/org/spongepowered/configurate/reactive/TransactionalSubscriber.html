017package org.spongepowered.configurate.reactive;
018
019/**
020 * A subscriber that is transaction-aware. As opposed to standard Subscribers
021 * which receive simple value events, transactional subscribers receive a series
022 * of events: first, a {@code beginTransaction}, followed by a {@code commit}
023 * or {@code rollback}.
024 *
025 * <p>A transactional subscriber will only be fully effective when controlled by
026 * a {@link Processor.Transactional transaction-aware processor}. A
027 * transaction-aware processor will roll back all subscribers if the transaction
028 * fails on any one subscriber, where a standard publisher will not.
029 *
030 * @param <V> the value handled by this subscriber
031 */
032public interface TransactionalSubscriber<V> extends Subscriber<V> {
033
034    @Override
035    default void submit(V item) {
036        try {
037            beginTransaction(item);
038            commit();
039        } catch (TransactionFailedException ex) {
040            rollback();
041        } catch (Exception ex) {
042            rollback();
043            throw ex;
044        }
045    }
046
047    /**
048     * Receive a new value, and validate it. The received value must not be made
049     * available outside of to other transaction-aware viewers until
050     * {@link #commit()} has been called.
051     *
052     * @param newValue the new value
053     * @throws TransactionFailedException if the new value does not validate
054     */
055    void beginTransaction(V newValue) throws TransactionFailedException;
056
057    /**
058     * Expose a transaction's result.
059     *
060     * <p>This method will be called on all transactional subscribers in a
061     * system have received and validated any new data. Calling this method when
062     * a transaction is not in progress should result in a noop.
063     */
064    void commit();
065
066    /**
067     * Called when a transaction has failed, to revert any prepared changes.
068     *
069     * <p>This event indicates that it is safe for clients to discard any
070     * prepared information from an in-progress transaction. If there is no
071     * transaction in progress, this must be a no-op.
072     */
073    void rollback();
074
075}
