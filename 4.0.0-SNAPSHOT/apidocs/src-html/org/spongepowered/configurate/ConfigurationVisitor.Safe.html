017package org.spongepowered.configurate;
018
019/**
020 * A visitor to traverse node hierarchies in a depth-first order.
021 *
022 * <p>Instances of stateful implementations may be reusable by taking advantage
023 * of the state object. During each visitation, a visitor will experience the
024 * node tree as a sequence of events, described by the following
025 * pseudo-grammar:</p>
026 *
027 * <pre>
028 * mappingNode: enterMappingNode node* exitMappingNode
029 * listNode: enterListNode node* exitListNode
030 *
031 * node: enterNode
032 *      (mappingNode
033 *       | listNode
034 *       | enterScalarNode)
035 *
036 * visit: newState?
037 *        beginVisit
038 *        node*
039 *        endVisit
040 * </pre>
041 *
042 * <p>If the starting node has no value, no node events will be received.
043 * Otherwise, the first event received will be for the starting node itself, and
044 * will continue from there.</p>
045 *
046 * <p>The children to visit for list and mapping nodes will only be collected
047 * after both the {@code enterNode} and {@code enter(List|Mapping)Node} methods
048 * have been executed for the node, and changes to the node values may be made
049 * to control which nodes will be visited.</p>
050 *
051 * <p>Any exceptions thrown within the visitor will result in the visitation
052 * ending immediately and the exception being rethrown within the visit
053 * method.</p>
054 *
055 * <p>There are a few specializations of the visitor interface available:
056 * {@link Stateless} carries no state and can act as a functional interface
057 * type, and {@link Safe} which throws no checked exceptions and therefore can
058 * be visited without having to handle any exceptions.</p>
059 *
060 * @param <S> a state object that will be used for one visit
061 * @param <T> the terminal value, that can be returned at the end of the visit
062 * @param <E> exception type that may be thrown
063 * @see ScopedConfigurationNode#visit(ConfigurationVisitor) to execute this
064 *      configuration visitation
065 */
066public interface ConfigurationVisitor<S, T, E extends Exception> {
067
068    /**
069     * Called to provide a state object if a visit is initiated without one
070     * already existing.
071     *
072     * @return a new state object to be passed through the rest of this visit
073     * @throws E when thrown by implementation
074     */
075    S newState() throws E;
076
077    /**
078     * Called at the beginning of the visit with a state object created.
079     *
080     * @param node the root node
081     * @param state the state
082     * @throws E when thrown by implementation
083     */
084    void beginVisit(ConfigurationNode node, S state) throws E;
085
086    /**
087     * Called once per node, for every node.
088     *
089     * @param node the current node
090     * @param state provided state
091     * @throws E when thrown by implementation
092     */
093    void enterNode(ConfigurationNode node, S state) throws E;
094
095    /**
096     * Called after {@link #enterNode(ConfigurationNode, Object)} for mapping
097     * nodes.
098     *
099     * @param node current node
100     * @param state provided state
101     * @throws E when thrown by implementation
102     */
103    void enterMappingNode(ConfigurationNode node, S state) throws E;
104
105    /**
106     * Called after {@link #enterNode(ConfigurationNode, Object)} for list nodes.
107     *
108     * @param node current node
109     * @param state provided state
110     * @throws E when thrown by implementation
111     */
112    void enterListNode(ConfigurationNode node, S state) throws E;
113
114    /**
115     * Called after {@link #enterNode(ConfigurationNode, Object)} for scalar nodes.
116     *
117     * @param node current node
118     * @param state provided state
119     * @throws E when thrown by implementation
120     */
121    void enterScalarNode(ConfigurationNode node, S state) throws E;
122
123    /**
124     * Called for a list node after the node and any of its children have
125     * been visited.
126     *
127     * @param node the node that has been visited
128     * @param state provided state
129     * @throws E when thrown by implementation
130     */
131    void exitMappingNode(ConfigurationNode node, S state) throws E;
132
133    /**
134     * Called for a list node after the node and any of its children have
135     * been visited.
136     *
137     * @param node the node that has been visited
138     * @param state provided state
139     * @throws E when thrown by implementation
140     */
141    void exitListNode(ConfigurationNode node, S state) throws E;
142
143    /**
144     * Called after every node has been visited, to allow for cleanup
145     * and validation.
146     *
147     * @param state provided state
148     * @return a terminal value
149     * @throws E when thrown by implementation
150     */
151    T endVisit(S state) throws E;
152
153    /**
154     * Stateless specialization of visitors, where both the state and terminal
155     * type are Void.
156     *
157     */
158    @FunctionalInterface
159    interface Stateless<E extends Exception> extends ConfigurationVisitor<Void, Void, E> {
160        @Override
161        default Void newState() {
162            return null;
163        }
164
165        @Override
166        default void beginVisit(ConfigurationNode node, Void state) throws E {
167            beginVisit(node);
168        }
169
170        /**
171         * Called at the beginning of the visit with a state object created.
172         *
173         * @param node the root node
174         * @throws E as required by implementation
175         */
176        default void beginVisit(ConfigurationNode node) throws E {}
177
178        @Override
179        default void enterNode(ConfigurationNode node, Void state) throws E {
180            enterNode(node);
181        }
182
183        /**
184         * Called once per node, for every node.
185         *
186         * @param node the current node
187         * @throws E as required by implementation
188         */
189        void enterNode(ConfigurationNode node) throws E;
190
191        @Override
192        default void enterMappingNode(ConfigurationNode node, Void state) throws E {
193            enterMappingNode(node);
194        }
195
196        /**
197         * Called after {@link #enterNode(ConfigurationNode, Object)} for
198         * mapping nodes.
199         *
200         * @param node current node
201         * @throws E when thrown by implementation
202         */
203        default void enterMappingNode(ConfigurationNode node) throws E {}
204
205        @Override
206        default void enterListNode(ConfigurationNode node, Void state) throws E {
207            enterListNode(node);
208        }
209
210        /**
211         * Called after {@link #enterNode(ConfigurationNode, Object)} for list
212         * nodes.
213         *
214         * @param node current node
215         * @throws E when thrown by implementation
216         */
217        default void enterListNode(ConfigurationNode node) throws E {
218        }
219
220        @Override
221        default void enterScalarNode(ConfigurationNode node, Void state) throws E {
222            enterScalarNode(node);
223        }
224
225        /**
226         * Called after {@link #enterNode(ConfigurationNode, Object)} for scalar
227         * nodes.
228         *
229         * @param node current node
230         * @throws E when thrown by implementation
231         */
232        default void enterScalarNode(ConfigurationNode node) throws E {
233        }
234
235        @Override
236        default void exitMappingNode(ConfigurationNode node, Void state) throws E {
237            exitMappingNode(node);
238        }
239
240        /**
241         * Called for a mapping node after the node and any of its children have
242         * been visited.
243         *
244         * @param node the node that has been visited
245         * @throws E when thrown by implementation
246         */
247        default void exitMappingNode(ConfigurationNode node) throws E {}
248
249        @Override
250        default void exitListNode(ConfigurationNode node, Void state) throws E {
251            exitListNode(node);
252        }
253
254        /**
255         * Called for a list node after the node and any of its children have
256         * been visited.
257         *
258         * @param node the node that has been visited
259         * @throws E when thrown by implementation
260         */
261        default void exitListNode(ConfigurationNode node) throws E {}
262
263        @Override
264        default Void endVisit(Void state) throws E {
265            endVisit();
266            return null;
267        }
268
269        /**
270         * Called after every node has been visited, to allow for cleanup
271         * and validation.
272         *
273         * @throws E when thrown by implementation
274         */
275        default void endVisit() throws E {}
276    }
277
278    /**
279     * A subinterface for visitors that do not throw any checked exceptions
280     * during their execution.
281     *
282     * @param <S> state type
283     * @param <T> terminal value type
284     */
285    interface Safe<S, T> extends ConfigurationVisitor<S, T, VisitorSafeNoopException> {
286
287        @Override
288        S newState();
289
290        @Override
291        void beginVisit(ConfigurationNode node, S state);
292
293        @Override
294        void enterNode(ConfigurationNode node, S state);
295
296        @Override
297        void enterMappingNode(ConfigurationNode node, S state);
298
299        @Override
300        void enterListNode(ConfigurationNode node, S state);
301
302        @Override
303        void enterScalarNode(ConfigurationNode node, S state);
304
305        @Override
306        void exitMappingNode(ConfigurationNode node, S state);
307
308        @Override
309        void exitListNode(ConfigurationNode node, S state);
310
311        @Override
312        T endVisit(S state);
313
314    }
